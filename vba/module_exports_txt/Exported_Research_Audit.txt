' --- PRVKY PRO PŘIPOJENÍ DISKU (ZABRÁNĚNÍ AMBIGUOUS NAME) ---
Private Function MountVirtualDrive(ByVal driveLetter As String, ByVal targetPath As String) As Boolean
    Dim wsh As Object, fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    If fso.DriveExists(driveLetter) Then
        MountVirtualDrive = True
        Exit Function
    End If
    Set wsh = CreateObject("WScript.Shell")
    If Right(targetPath, 1) = "\" Then targetPath = left(targetPath, Len(targetPath) - 1)
    Dim cmd As String: cmd = "subst " & driveLetter & " """ & targetPath & """"
    On Error Resume Next
    wsh.Run cmd, 0, True
    On Error GoTo 0
    MountVirtualDrive = fso.DriveExists(driveLetter)
End Function

Private Sub UnmountVirtualDrive(driveLetter As String)
    Dim wsh As Object: Set wsh = CreateObject("WScript.Shell")
    On Error Resume Next
    wsh.Run "subst " & driveLetter & " /d", 0, True
    On Error GoTo 0
End Sub

' --- UTILITY PRO ZABRÁNĚNÍ AMBIGUOUS NAME ---
Private Sub ToggleSpeed(ByVal speedUp As Boolean, Optional wApp As Object = Nothing)
    On Error Resume Next
    If speedUp Then
        Application.ScreenUpdating = False
        Application.Calculation = -4135 ' xlCalculationManual
        Application.EnableEvents = False
        Application.DisplayStatusBar = True
        
        If Not wApp Is Nothing Then
            wApp.ScreenUpdating = False
            wApp.DisplayAlerts = 0 ' wdAlertsNone
            wApp.Options.CheckSpellingAsYouType = False
            wApp.Options.CheckGrammarAsYouType = False
            wApp.Options.Pagination = False
        End If
    Else
        Application.ScreenUpdating = True
        Application.Calculation = -4105 ' xlCalculationAutomatic
        Application.EnableEvents = True
        Application.DisplayStatusBar = True
        
        If Not wApp Is Nothing Then
            wApp.ScreenUpdating = True
            wApp.DisplayAlerts = -1 ' wdAlertsAll
            wApp.Options.CheckSpellingAsYouType = True
            wApp.Options.CheckGrammarAsYouType = True
            wApp.Options.Pagination = True
        End If
    End If
End Sub

Private Function CleanFileName(ByVal rawName As String) As String
    Dim s As String: s = rawName
    s = Replace(s, ":", "_"): s = Replace(s, "/", " "): s = Replace(s, "\", "")
    s = Replace(s, "*", ""): s = Replace(s, "?", ""): s = Replace(s, """", "")
    s = Replace(s, "<", ""): s = Replace(s, ">", ""): s = Replace(s, "|", "")
    CleanFileName = Application.Trim(s)
End Function

Private Function LoadFileCache(fso As Object, folderPath As String) As Object
    Dim dict As Object
    Set dict = CreateObject("Scripting.Dictionary")
    dict.CompareMode = vbTextCompare
    
    If Right(folderPath, 1) <> "\" Then folderPath = folderPath & "\"
    
    If fso.FolderExists(folderPath) Then
        Dim parentFolder As Object
        Set parentFolder = fso.GetFolder(folderPath)
        Dim fileObj As Object
        For Each fileObj In parentFolder.Files
            dict(fileObj.Name) = True
        Next fileObj
    End If
    Set LoadFileCache = dict
End Function

Private Function LoadFolderCache(fso As Object, folderPath As String) As Object
    Dim dict As Object
    Set dict = CreateObject("Scripting.Dictionary")
    dict.CompareMode = vbTextCompare
    
    If Right(folderPath, 1) <> "\" Then folderPath = folderPath & "\"
    
    If fso.FolderExists(folderPath) Then
        Dim parentFolder As Object
        Set parentFolder = fso.GetFolder(folderPath)
        Dim subFolder As Object
        For Each subFolder In parentFolder.SubFolders
            dict(subFolder.Name) = True
        Next subFolder
    End If
    Set LoadFolderCache = dict
End Function

Private Function GetWordCountFromDoc(wordApp As Object, docPath As String) As Long
    Dim doc As Object
    Dim wCount As Long: wCount = 0
    On Error Resume Next
    Set doc = wordApp.Documents.Open(FileName:=docPath, ReadOnly:=True, Visible:=False, AddToRecentFiles:=False)
    If Not doc Is Nothing Then
        wCount = doc.BuiltinDocumentProperties(15) ' wdPropertyWords
        If wCount <= 0 Then
            wCount = doc.Range.ComputeStatistics(0) ' wdStatisticWords
        End If
        doc.Close SaveChanges:=False
    End If
    On Error GoTo 0
    GetWordCountFromDoc = wCount
End Function

Private Function GetRatingHTML(ByVal rating As Double) As String
    If rating <= 0 Then
        GetRatingHTML = "<span class=""badge badge-normal"" style=""color: var(--text-secondary);"">X/10</span>"
    Else
        Dim rColor As String
        If rating >= 9 Then
            rColor = "#22c55e"
        ElseIf rating >= 8 Then
            rColor = "#84cc16"
        ElseIf rating >= 7 Then
            rColor = "#eab308"
        ElseIf rating >= 6 Then
            rColor = "#f97316"
        Else
            rColor = "#ef4444"
        End If
        GetRatingHTML = "<strong style=""color: " & rColor & ";"">" & Format(rating, "0") & "/10</strong>"
    End If
End Function

Private Function GetWordCountProgressHTML(ByVal wCount As Long, Optional ByVal maxVal As Long = 6200) As String

    wPct = CLng((wCount / maxVal) * 100)
    If wPct > 100 Then wPct = 100
    Dim wClass As String
    If wPct < 30 Then
        wClass = "critical"
    ElseIf wPct < 60 Then
        wClass = "high"
    ElseIf wPct < 90 Then
        wClass = "medium"
    Else
        wClass = "low"
    End If
    Dim res As String
    res = "<div class=""progress-container"">" & _
          "<span class=""word-count-text"">" & wCount & " slov</span>" & _
          "<div class=""progress-bar-bg"">" & _
          "<div class=""progress-bar " & wClass & """ style=""width: " & wPct & "%;""></div>" & _
          "</div></div>"
    GetWordCountProgressHTML = res
End Function

Private Function NormalizeForMatch(ByVal s As String) As String
    s = LCase(s)
    s = Replace(s, ":", ""): s = Replace(s, "/", ""): s = Replace(s, "\", "")
    s = Replace(s, "_", ""): s = Replace(s, "-", ""): s = Replace(s, " ", "")
    s = Replace(s, ",", ""): s = Replace(s, ".", ""): s = Replace(s, ";", "")
    s = Replace(s, "!", ""): s = Replace(s, "?", ""): s = Replace(s, "*", "")
    s = Replace(s, "<", ""): s = Replace(s, ">", ""): s = Replace(s, "|", "")
    s = Replace(s, """", ""): s = Replace(s, "'", "")
    s = Replace(s, ChrW(&H201C), ""): s = Replace(s, ChrW(&H201D), "")
    s = Replace(s, ChrW(&H2018), ""): s = Replace(s, ChrW(&H2019), "")
    s = Replace(s, "ě", "e"): s = Replace(s, "á", "a"): s = Replace(s, "í", "i")
    s = Replace(s, "ó", "o"): s = Replace(s, "ú", "u"): s = Replace(s, "ů", "u")
    s = Replace(s, "ý", "y"): s = Replace(s, "é", "e"): s = Replace(s, "š", "s")
    s = Replace(s, "č", "c"): s = Replace(s, "ř", "r"): s = Replace(s, "ž", "z")
    s = Replace(s, "ť", "t"): s = Replace(s, "ď", "d"): s = Replace(s, "ň", "n")
    s = Replace(s, "adventures", "adventure")
    s = Replace(s, "s0", "s") ' Normalizace sezón, např. S04 -> s4
    s = Replace(s, ChrW(65533), "e")
    NormalizeForMatch = s
End Function

Private Function FindMatchingFileOnDisk(ByVal animeName As String, ByVal seriesName As String, dictFiles As Object) As String
    Dim cleanA As String: cleanA = CleanFileName(animeName) & ".docx"
    If dictFiles.Exists(cleanA) Then
        FindMatchingFileOnDisk = cleanA
        Exit Function
    End If
    
    If seriesName <> "" Then
        Dim cleanS As String: cleanS = CleanFileName(seriesName) & ".docx"
        If dictFiles.Exists(cleanS) Then
            FindMatchingFileOnDisk = cleanS
            Exit Function
        End If
    End If
    
    Dim normA As String: normA = NormalizeForMatch(animeName)
    Dim normS As String: normS = NormalizeForMatch(seriesName)
    
    Dim k As Variant
    For Each k In dictFiles.keys
        Dim fName As String: fName = CStr(k)
        If Len(fName) >= 5 And LCase(Right(fName, 5)) = ".docx" Then
            Dim normF As String: normF = NormalizeForMatch(left(fName, Len(fName) - 5))
            
            If normF <> "" Then
                If normA = normF Then
                    FindMatchingFileOnDisk = fName: Exit Function
                End If
                If Len(normA) > 5 Then
                    If InStr(1, normF, normA, vbTextCompare) = 1 Then
                        Dim tailFA As String: tailFA = Mid(normF, Len(normA) + 1)
                        If normA = normS And Len(tailFA) >= 2 And left(tailFA, 1) = "s" And IsNumeric(Mid(tailFA, 2, 1)) Then
                            ' Excel nemá sezónu, ale soubor ano -> zamítnout (např. "Slime" nebere "Slime S02")
                        Else
                            FindMatchingFileOnDisk = fName: Exit Function
                        End If
                    End If
                End If
            End If
        End If
    Next k
    FindMatchingFileOnDisk = ""
End Function

Private Function FindMatchingFolderOnDisk(ByVal animeName As String, ByVal seriesName As String, dictFolders As Object) As String
    Dim cleanA As String: cleanA = CleanFileName(animeName)
    If dictFolders.Exists(cleanA) Then
        FindMatchingFolderOnDisk = cleanA
        Exit Function
    End If
    
    If seriesName <> "" Then
        Dim cleanS As String: cleanS = CleanFileName(seriesName)
        If dictFolders.Exists(cleanS) Then
            FindMatchingFolderOnDisk = cleanS
            Exit Function
        End If
    End If
    
    Dim normA As String: normA = NormalizeForMatch(animeName)
    Dim normS As String: normS = NormalizeForMatch(seriesName)
    
    Dim k As Variant
    For Each k In dictFolders.keys
        Dim fName As String: fName = CStr(k)
        Dim normF As String: normF = NormalizeForMatch(fName)
        
        If normF <> "" Then
            If normA = normF Then
                FindMatchingFolderOnDisk = fName: Exit Function
            End If
            If Len(normA) > 5 Then
                If InStr(1, normF, normA, vbTextCompare) = 1 Then
                    Dim tailFA As String: tailFA = Mid(normF, Len(normA) + 1)
                    If normA = normS And Len(tailFA) >= 2 And left(tailFA, 1) = "s" And IsNumeric(Mid(tailFA, 2, 1)) Then
                        ' Excel nemá sezónu, ale složka ano -> zamítnout
                    Else
                        FindMatchingFolderOnDisk = fName: Exit Function
                    End If
                End If
            End If
        End If
    Next k
    FindMatchingFolderOnDisk = ""
End Function

' --- QUICKSORT PRO AnimeImprovementData ---
Private Sub QuickSortImprovements(arr() As AnimeImprovementData, ByVal lo As Long, ByVal hi As Long)
    If lo >= hi Then Exit Sub
    Dim pivot As Double: pivot = arr((lo + hi) \ 2).PriorityScore
    Dim i As Long: i = lo
    Dim j As Long: j = hi
    Dim tmp As AnimeImprovementData
    Do While i <= j
        Do While arr(i).PriorityScore > pivot: i = i + 1: Loop
        Do While arr(j).PriorityScore < pivot: j = j - 1: Loop
        If i <= j Then
            tmp = arr(i): arr(i) = arr(j): arr(j) = tmp
            i = i + 1: j = j - 1
        End If
    Loop
    If lo < j Then QuickSortImprovements arr, lo, j
    If i < hi Then QuickSortImprovements arr, i, hi
End Sub

'--------------------------------------------------------------------------------------------------
' VYTVOŘENÍ A OTEVŘENÍ HTML AUDIT REPORTU (PREMIUM DESIGN)
'--------------------------------------------------------------------------------------------------
Public Sub GenerovatHTMLAudit()
    Dim ws As Worksheet
    Dim lastRow As Long, r As Long
    Dim fso As Object, folderObj As Object, fileObj As Object
    Dim dictExcel As Object, dictFiles As Object
    Dim dictTop10 As Object, dictHM As Object
    Dim wordApp As Object, doc As Object
    Dim createdWord As Boolean: createdWord = False
    Dim startTimer As Double: startTimer = Timer
    
    ' Výstupní cesta k souboru HTML (do Stažených souborů)
    Dim htmlPath As String
    htmlPath = Environ("USERPROFILE") & "\Downloads\Audit_Report.html"
    
    ' Nastavení a optimalizace
    ToggleSpeed True
    
    ' --- 1. Připojení disku ---
    If Not MountVirtualDrive(VIRTUAL_DRIVE, PATH_ROOT_LONG) Then
        MsgBox "Nepodařilo se připojit virtuální disk " & VIRTUAL_DRIVE, vbCritical
        GoTo CleanExit
    End If
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set dictExcel = CreateObject("Scripting.Dictionary")
    dictExcel.CompareMode = vbTextCompare
    
    ' Načtení dat z listu ANIME LIST
    Set ws = ThisWorkbook.Sheets("ANIME LIST")
    lastRow = ws.Cells(ws.Rows.count, "C").End(xlUp).row
    
    ' Pre-cache názvů sérií z komentářů ve sloupci C
    Dim commentsC As Object
    Set commentsC = CreateObject("Scripting.Dictionary")
    For r = 2 To lastRow
        If Not ws.Cells(r, "C").Comment Is Nothing Then
            commentsC(r) = GetSeriesNameFromComment(ws.Cells(r, "C"))
        Else
            commentsC(r) = ""
        End If
    Next r
    
    ' Načtení TOP 10 a HM
    Call NactiTop10AHM(dictTop10, dictHM)
    
    ' Načtení cache souborů
    Set dictFiles = LoadFileCache(fso, PATH_SRC_ANALYSIS)
    
    ' Inicializace Wordu pro audit počtu slov
    On Error Resume Next
    Set wordApp = GetObject(, "Word.Application")
    If wordApp Is Nothing Then
        Set wordApp = CreateObject("Word.Application")
        createdWord = True
    End If
    On Error GoTo ErrorHandler
    
    wordApp.Visible = False
    wordApp.DisplayAlerts = 0
    ToggleSpeed True, wordApp
    
    ' Načtení hlavních dat z Excelu
    Dim arrData As Variant
    If lastRow >= 2 Then
        arrData = ws.Range(ws.Cells(2, 3), ws.Cells(lastRow, 11)).Value ' Sloupce C až K (C=3, K=11)
    End If
    
    ' Datové kolekce pro HTML
    Dim missingList As Object: Set missingList = CreateObject("System.Collections.ArrayList")
    Dim extraList As Object: Set extraList = CreateObject("System.Collections.ArrayList")
    Dim completeList As Object: Set completeList = CreateObject("System.Collections.ArrayList")
    Dim newAnimeList As Object: Set newAnimeList = CreateObject("System.Collections.ArrayList")
    Dim wipAnimeList As Object: Set wipAnimeList = CreateObject("System.Collections.ArrayList")
    Dim improvementList() As AnimeImprovementData
    Dim impCount As Long: impCount = 0
    Dim totalWordCount As Long: totalWordCount = 0
    Dim docCount As Long: docCount = 0
    
    Dim top5L(1 To 5) As String, top5LW(1 To 5) As Long
    Dim top5S(1 To 5) As String, top5SW(1 To 5) As Long
    Dim i_top As Long, j_top As Long
    For i_top = 1 To 5
        top5LW(i_top) = 0: top5SW(i_top) = 9999999
    Next i_top
    
    Dim dictWIP As Object
    Set dictWIP = LoadFolderCache(fso, PATH_WIP_ANALYSIS)
    
    If IsArray(arrData) Then
        ReDim improvementList(1 To UBound(arrData, 1))
        
        For r = 1 To UBound(arrData, 1)
            ' Status bar progress
            If r Mod 10 = 0 Then
                Application.StatusBar = "Skenování rozborů pro HTML audit: " & r & " / " & UBound(arrData, 1)
                DoEvents
            End If
            
            Dim animeName As String: animeName = Trim(CStr(arrData(r, 1))) ' Sloupec C
            Dim ratingVal As Double: ratingVal = 0
            If IsNumeric(arrData(r, 9)) Then ratingVal = CDbl(arrData(r, 9)) ' Sloupec K (Rating)
            
            If animeName <> "" Then
                Dim seriesName As String
                If commentsC.Exists(r + 1) Then
                    seriesName = CStr(commentsC(r + 1))
                Else
                    seriesName = ""
                End If
                
                Dim releaseDateStr As String: releaseDateStr = Trim(CStr(arrData(r, 4))) ' Sloupec F
                If releaseDateStr <> "" Then
                    On Error Resume Next
                    Dim rDate As Date: rDate = CDate(releaseDateStr)
                    Dim dateOk As Boolean: dateOk = (err.Number = 0)
                    err.Clear
                    On Error GoTo ErrorHandler
                    If dateOk And rDate > CDate(NEW_ANIME_CUTOFF) Then
                        newAnimeList.Add Array(animeName, releaseDateStr, ratingVal)
                    End If
                End If
                
                Dim matchedWIP As String
                matchedWIP = FindMatchingFolderOnDisk(animeName, seriesName, dictWIP)
                If matchedWIP <> "" Then
                    wipAnimeList.Add Array(animeName, matchedWIP)
                End If
                
                ' Označíme, že anime existuje v Excelu
                If Not dictExcel.Exists(animeName) Then
                    dictExcel.Add animeName, animeName
                End If
                
                Dim matchedFile As String
                matchedFile = FindMatchingFileOnDisk(animeName, seriesName, dictFiles)
                Dim wCount As Long: wCount = 0
                
                If matchedFile <> "" Then
                    dictFiles(matchedFile) = False ' Označíme jako využité
                    
                    ' Přečteme počet slov dokumentu
                    Dim analysisPath As String
                    analysisPath = PATH_SRC_ANALYSIS & matchedFile
                    
                    wCount = GetWordCountFromDoc(wordApp, analysisPath)
                    On Error GoTo ErrorHandler
                    
                    ' Akumulace statistik
                    If wCount > 0 Then
                        totalWordCount = totalWordCount + wCount
                        docCount = docCount + 1
                        
                        For i_top = 1 To 5
                            If wCount > top5LW(i_top) Then
                                For j_top = 5 To i_top + 1 Step -1
                                    top5LW(j_top) = top5LW(j_top - 1)
                                    top5L(j_top) = top5L(j_top - 1)
                                Next j_top
                                top5LW(i_top) = wCount
                                top5L(i_top) = animeName
                                Exit For
                            End If
                        Next i_top
                        
                        For i_top = 1 To 5
                            If wCount < top5SW(i_top) Then
                                For j_top = 5 To i_top + 1 Step -1
                                    top5SW(j_top) = top5SW(j_top - 1)
                                    top5S(j_top) = top5S(j_top - 1)
                                Next j_top
                                top5SW(i_top) = wCount
                                top5S(i_top) = animeName
                                Exit For
                            End If
                        Next i_top
                    End If
                    
                    ' Výpočet hustoty (slov na epizodu)
                    Dim eps As Variant: eps = arrData(r, 7) ' Sloupec I
                    Dim mins As Variant: mins = arrData(r, 8) ' Sloupec J
                    Dim density As Long: density = 0
                    If IsNumeric(eps) And IsNumeric(mins) And val(eps) > 0 And val(mins) > 0 Then
                        Dim normLen As Double: normLen = (CDbl(eps) * CDbl(mins)) / 24
                        If normLen > 0 Then density = CLng(wCount / normLen)
                    End If
                    
                    ' Zkontrolujeme, zda je rozbor kompletní nebo k vylepšení
                    If wCount >= WORD_THRESHOLD_COMPLETE Then
                        ' Kompletní rozbor
                        completeList.Add Array(animeName, wCount, ratingVal, seriesName, density)
                    Else
                        ' K vylepšení
                        impCount = impCount + 1
                        improvementList(impCount).Name = animeName
                        improvementList(impCount).rating = ratingVal
                        improvementList(impCount).WordCount = wCount
                        improvementList(impCount).density = density
                        
                        ' Zjistíme, zda je v TOP 10 nebo HM
                        Dim t10Rank As Long: t10Rank = 0
                        Dim inHM As Boolean: inHM = False
                        
                        ' Match podle názvu nebo série
                        If dictTop10.Exists(animeName) Then
                            t10Rank = dictTop10(animeName)
                        ElseIf seriesName <> "" And dictTop10.Exists(seriesName) Then
                            t10Rank = dictTop10(seriesName)
                        End If
                        
                        If dictHM.Exists(animeName) Then
                            inHM = True
                        ElseIf seriesName <> "" And dictHM.Exists(seriesName) Then
                            inHM = True
                        End If
                        
                        improvementList(impCount).Top10Rank = t10Rank
                        improvementList(impCount).IsHM = inHM
                        improvementList(impCount).IsWIP = (matchedWIP <> "")
                        improvementList(impCount).FileName = matchedFile
                        
                        ' Výpočet prioritního skóre
                        Dim score As Double: score = ratingVal * 10
                        If t10Rank > 0 Then
                            score = score + (11 - t10Rank) * 10
                        End If
                        If inHM Then
                            score = score + 50
                        End If
                        
                        improvementList(impCount).PriorityScore = score
                        
                        ' Třída priority
                        If score >= 150 Then
                            improvementList(impCount).PriorityClass = "Critical"
                        ElseIf score >= 100 Then
                            improvementList(impCount).PriorityClass = "High"
                        ElseIf score >= 70 Then
                            improvementList(impCount).PriorityClass = "Medium"
                        Else
                            improvementList(impCount).PriorityClass = "Low"
                        End If
                    End If
                Else
                    ' Chybí rozbor na disku
                    missingList.Add Array(animeName, ratingVal, seriesName)
                End If
            End If
        Next r
    End If
    
    ' Skenování disku na přebývající soubory (Extra files)
    Dim k As Variant
    Dim extCount As Long: extCount = 0
    For Each k In dictFiles.keys
        Dim fName As String: fName = CStr(k)
        If dictFiles(k) = True Then ' Nevyužité -> Extra
            If LCase(Right(fName, 5)) = ".docx" And left(fName, 2) <> "~$" Then
                extCount = extCount + 1
                If extCount Mod 5 = 0 Then
                    Application.StatusBar = "Analyzuji soubory navíc pro HTML audit: " & extCount
                    DoEvents
                End If
                
                ' Přečteme slova i pro přebývající soubory
                Dim exPath As String: exPath = PATH_SRC_ANALYSIS & fName
                Dim extraWCount As Long
                extraWCount = GetWordCountFromDoc(wordApp, exPath)
                
                extraList.Add Array(fName, extraWCount)
            End If
        End If
    Next k
    
    ' Word bude bezpečně ukončen v CleanExit
    
    ' Třídění rozborů k vylepšení (QuickSort podle PriorityScore sestupně)
    If impCount > 1 Then QuickSortImprovements improvementList, 1, impCount
    
    ' --- GENEROVÁNÍ HTML SOUBORU ---
    Dim f As Long
    f = FreeFile
    Open htmlPath For Output As #f
    
    ' Hlavička HTML
    Print #f, "<!DOCTYPE html>"
    Print #f, "<html lang=""cs"">"
    Print #f, "<head>"

' POMOCNÝ SUB PRO NAČTENÍ TOP 10 A HM Z SHAPES LISTU OBECNÉ INFORMACE
'--------------------------------------------------------------------------------------------------
Private Sub NactiTop10AHM(ByRef dictTop10 As Object, ByRef dictHM As Object)
    Dim wsTarget As Worksheet
    Dim shp As Shape
    Dim i As Long
    
    Set wsTarget = ThisWorkbook.Sheets("OBECNÉ INFORMACE")
    Set dictTop10 = CreateObject("Scripting.Dictionary")
    Set dictHM = CreateObject("Scripting.Dictionary")
    
    dictTop10.CompareMode = vbTextCompare
    dictHM.CompareMode = vbTextCompare
    
    ' Top 10
    For i = 1 To 10
        Set shp = Nothing
        On Error Resume Next
        Set shp = wsTarget.Shapes("Top10_Anime_" & i)
        On Error GoTo 0
        If Not shp Is Nothing Then
            If Trim(shp.AlternativeText) <> "" Then
                If Not dictTop10.Exists(Trim(shp.AlternativeText)) Then
                    dictTop10.Add Trim(shp.AlternativeText), i
                End If
            End If
        End If
    Next i
    
    ' HM
    For i = 1 To HM_MAX_SLOTS
        Set shp = Nothing
        On Error Resume Next
        Set shp = wsTarget.Shapes("HM_Anime_" & i)
        On Error GoTo 0
        If Not shp Is Nothing Then
            If Trim(shp.AlternativeText) <> "" Then
                If Not dictHM.Exists(Trim(shp.AlternativeText)) Then
                    dictHM.Add Trim(shp.AlternativeText), True
                End If
            End If
        End If
    Next i
End Sub

' POMOCNÁ FUNKCE PRO EXTRAKCI NÁZVU SÉRIE Z KOMENTÁŘE
'--------------------------------------------------------------------------------------------------
Private Function GetSeriesNameFromComment(cell As Range) As String
    If Not cell.Comment Is Nothing Then
        Dim cleanText As String
        cleanText = Replace(cell.Comment.text, vbCr, "")
        Dim lines As Variant
        lines = Split(cleanText, vbLf)
        Dim line As Variant
        For Each line In lines
            If InStr(1, Trim(line), "Název série:", vbTextCompare) = 1 Then
                GetSeriesNameFromComment = Trim(Mid(line, InStr(1, line, ":") + 1))
                Exit Function
            End If
        Next line
    End If
    GetSeriesNameFromComment = ""
End Function
