Option Explicit

' --- DEKLARACE WINDOWS API (64-bit kompatibilni) ---
#If VBA7 Then
    Declare PtrSafe Sub Sleep Lib "kernel32" (ByVal dwMilliseconds As Long)
    Declare PtrSafe Function GetForegroundWindow Lib "user32" () As LongPtr
    ' Funkce pro zruseni maximalizace
    Declare PtrSafe Function ShowWindow Lib "user32" (ByVal hwnd As LongPtr, ByVal nCmdShow As Long) As Long
    Declare PtrSafe Function SetWindowPos Lib "user32" ( _
        ByVal hwnd As LongPtr, ByVal hWndInsertAfter As LongPtr, _
        ByVal x As Long, ByVal y As Long, ByVal cx As Long, ByVal cy As Long, _
        ByVal wFlags As Long) As Long
#End If

' --- KONSTANTY PRO NOVOU FUNKCI ---
Private Const ENABLE_MULTIPLE_WINDOWS_OPTION As Boolean = True
Private Const WINDOWS_MULTIPLIER As Long = 5

' --- HLAVNI MAKRO ---
Sub OtevriGeminiNaTretinyVedlejsiMonitor()
    Dim i As Integer
    Dim url As String
    #If VBA7 Then
        Dim hwnd As LongPtr
    #Else
        Dim hwnd As Long
    #End If
    Dim poziceX As Long
    Dim sirkaOkna As Long
    Dim vyskaOkna As Long
    Dim korekceOkraje As Long
    Dim chromePath As String
    
    ' Nove promenne pro rozsireny pocet oken a stridani
    Dim totalWindows As Long
    Dim g As Long
    Dim col As Long
    Dim currentMonitorX As Long
    
    ' Konstanta pro obnoveni okna do normalni velikosti
    Const SW_RESTORE As Long = 9
    
    ' Nastaveni parametru
    url = "https://gemini.google.com/"
    korekceOkraje = 7     ' Neviditelny stin okna ve Windows (obvykle 7 nebo 8 pixelu)
    
    ' Urceni poctu oken (vychozi 3)
    totalWindows = 3
    If ENABLE_MULTIPLE_WINDOWS_OPTION Then
        Dim msgText As String
        Dim msgTitle As String
        
        ' Sestaveni ceskeho textu pomoci ChrW pro spravne zobrazeni bez ohledu na kodovani Windows (systemove locale)
        msgText = "Chcete otev" & ChrW(345) & "t standardn" & ChrW(237) & " 3 okna (klikn" & ChrW(283) & "te Ano), " & _
                  "nebo roz?" & ChrW(237) & ChrW(345) & "en" & ChrW(253) & "ch " & (3 * WINDOWS_MULTIPLIER) & " oken (klikn" & ChrW(283) & "te Ne)?"
        msgTitle = "V" & ChrW(253) & "b" & ChrW(283) & "r po" & ChrW(269) & "tu oken"
        
        Dim response As VbMsgBoxResult
        response = MsgBox(msgText, vbYesNo + vbQuestion, msgTitle)
        If response = vbNo Then
            totalWindows = 3 * WINDOWS_MULTIPLIER
        End If
    End If
    
    ' Sirka je 640 (tretina z 1920) + rozsireni o levy i pravy neviditelny okraj
    sirkaOkna = 640 + (korekceOkraje * 2)
    vyskaOkna = 1040      ' 1080 minus odhadovana vyska spodni listy Windows
    
    chromePath = """C:\Program Files\Google\Chrome\Application\chrome.exe"""
    
    For i = 1 To totalWindows
        ' 1. Krok: Otevreni noveho samostatneho okna Chromu
        shell chromePath & " --new-window " & url, vbNormalFocus
        
        ' 2. Krok: Pauza 1200 milisekund (1,2 vteriny), aby okno stihlo vyskocit
        Sleep 1200
        
        ' 3. Krok: Zachyceni tohoto noveho okna (kter? je ted aktivni vpredu)
        hwnd = GetForegroundWindow()
        
        ' 4. Krok: Vynucene zruseni maximalizace okna (splasknuti do normalniho rezimu)
        ShowWindow hwnd, SW_RESTORE
        
        ' Drobna pauza, aby system stihl okno vizualne zmensit pred jeho presunem
        Sleep 200
        
        ' 5. Krok: Vypocet stridani monitoru (skupiny po 3 oknech)
        g = Int((i - 1) / 3)
        If g Mod 2 = 0 Then
            currentMonitorX = 1920 ' Vedlejsi monitor (napravo)
        Else
            currentMonitorX = 0    ' Hlavni monitor (nalevo)
        End If
        
        col = (i - 1) Mod 3
        poziceX = currentMonitorX + (col * 640) - korekceOkraje
        
        ' 6. Krok: Posunuti a zmena velikosti okna
        ' Parametry: hwnd, 0 (Z-order), X, Y (0 = uplne nahore), Sirka, Vyska, 0 (Flags)
        SetWindowPos hwnd, 0, poziceX, 0, sirkaOkna, vyskaOkna, 0
    Next i
End Sub
