' 2) Hlavní procedura - SPOUŠTĚČ
Public Sub AktualizujVsechnyAnimePrehledy()
    On Error GoTo ChybaHlavnihoMakra

    Dim wsData As Worksheet
    Dim lastRow As Long
    Dim arrData As Variant
    
    Set wsData = ThisWorkbook.Sheets("ANIME LIST")
    
    ' --- OPTIMALIZACE PROSTŘEDÍ ---
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual
    
    lastRow = wsData.Cells(wsData.Rows.count, "C").End(xlUp).row
    
    If lastRow < 2 Then
        MsgBox "Žádná data v Anime Listu.", vbExclamation
        GoTo CleanUpHlavnihoMakra
    End If
    
    ' NAČTENÍ DAT DO PAMĚTI (Range C2 až Q_poslední)
    arrData = wsData.Range("C2:Q" & lastRow).Value
    
    ' --- Krok 1: Poslední zhlédnutá ---
    VytvorPrehled_PosledniZhlednute arrData
    
    ' --- Krok 2: Souhrn sérií ---
    VytvorPrehled_NejdelsiSerie arrData, wsData
    
    ' --- Krok 3: Nejrychlejší Binge ---
    VytvorPrehled_NejrychlejsiBinge arrData

CleanUpHlavnihoMakra:
    Application.ScreenUpdating = True
    Application.Calculation = xlCalculationAutomatic
    Exit Sub

ChybaHlavnihoMakra:
    Application.ScreenUpdating = True
    Application.Calculation = xlCalculationAutomatic
    MsgBox "Chyba při aktualizaci: " & err.Description, vbCritical
End Sub

'====================================================================================
' *** POMOCNÁ FUNKCE PRO ŘAZENÍ ***
'====================================================================================
Private Sub BubbleSort2D(ByRef arr As Variant, ByVal SortColIndex As Long, Optional descending As Boolean = True)
    Dim i As Long, j As Long, k As Long
    Dim Temp As Variant
    Dim condition As Boolean
    
    For i = LBound(arr, 1) To UBound(arr, 1) - 1
        For j = i + 1 To UBound(arr, 1)
            If descending Then
                condition = (arr(j, SortColIndex) > arr(i, SortColIndex))
            Else
                condition = (arr(j, SortColIndex) < arr(i, SortColIndex))
            End If
            
            If condition Then
                For k = LBound(arr, 2) To UBound(arr, 2)
                    Temp = arr(i, k)
                    arr(i, k) = arr(j, k)
                    arr(j, k) = Temp
                Next k
            End If
        Next j
    Next i
End Sub

'====================================================================================
' *** SEKCE A: PŘEHLED POSLEDNÍCH ZHLÉDNUTÝCH ***
'====================================================================================
Private Sub VytvorPrehled_PosledniZhlednute(arrMain As Variant)
    Dim wsTarget As Worksheet
    Dim arrFiltered() As Variant
    Dim i As Long, count As Long, r As Long
    
    Dim tbHeader As Shape, tbContent As Shape
    Dim defaultTopPos As Double, defaultLeftPos As Double
    Dim borderColor As Long: borderColor = RGB(0, 0, 0)
    
    Set wsTarget = ThisWorkbook.Sheets("OBECNÉ INFORMACE")
    
    ' 1. Filtrace dat v paměti
    ReDim arrFiltered(1 To UBound(arrMain, 1), 1 To 5)
    count = 0
    
    For i = 1 To UBound(arrMain, 1)
        Dim valEnd As Variant: valEnd = arrMain(i, 11) ' M
        Dim valName As String: valName = CStr(arrMain(i, 1)) ' C
        
        If IsDate(valEnd) And Len(valName) > 0 Then
            count = count + 1
            arrFiltered(count, 1) = valName
            arrFiltered(count, 2) = arrMain(i, 10) ' Start
            arrFiltered(count, 3) = valEnd     ' End
            If IsNumeric(arrMain(i, 13)) Then arrFiltered(count, 4) = CLng(arrMain(i, 13)) Else arrFiltered(count, 4) = 0
            If IsNumeric(arrMain(i, 9)) Then arrFiltered(count, 5) = CInt(arrMain(i, 9)) Else arrFiltered(count, 5) = -1
        End If
    Next i
    
    If count = 0 Then Exit Sub
    
    Dim arrFinal() As Variant
    ReDim arrFinal(1 To count, 1 To 5)
    For i = 1 To count
        For r = 1 To 5
            arrFinal(i, r) = arrFiltered(i, r)
        Next r
    Next i
    
    ' 2. Seřazení podle DATA KONCE
    BubbleSort2D arrFinal, 3, True
    
    ' 3. Vykreslování
    With wsTarget.Range("BG27")
        defaultTopPos = .top
        defaultLeftPos = .left
    End With
    
    On Error Resume Next
    wsTarget.Shapes("lineAnimeOverviewHeaderTop").Delete
    wsTarget.Shapes("lineAnimeOverviewHeaderLeft").Delete
    wsTarget.Shapes("lineAnimeOverviewHeaderRight").Delete
    wsTarget.Shapes("lineAnimeOverviewContentLeft").Delete
    wsTarget.Shapes("lineAnimeOverviewContentRight").Delete
    wsTarget.Shapes("lineAnimeOverviewContentBottom").Delete
    On Error GoTo 0

    ' Hlavička
    On Error Resume Next
    Set tbHeader = wsTarget.Shapes("txtAnimeOverviewHeader")
    On Error GoTo 0
    
    If tbHeader Is Nothing Then
        Set tbHeader = wsTarget.Shapes.AddTextbox(msoTextOrientationHorizontal, defaultLeftPos, defaultTopPos, 300, 30)
        tbHeader.Name = "txtAnimeOverviewHeader"
    End If
    
    With tbHeader
        .line.Visible = msoFalse
        .TextFrame2.TextRange.text = "Poslední zhlédnutá Anime"
        With .TextFrame2.TextRange.Font
            .Bold = msoTrue: .size = 18: .Name = "Aptos Narrow"
        End With
    End With
    
    DrawLine wsTarget, "lineAnimeOverviewHeaderTop", tbHeader.left, tbHeader.top, tbHeader.left + tbHeader.Width, tbHeader.top, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeOverviewHeaderLeft", tbHeader.left, tbHeader.top, tbHeader.left, tbHeader.top + tbHeader.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeOverviewHeaderRight", tbHeader.left + tbHeader.Width, tbHeader.top, tbHeader.left + tbHeader.Width, tbHeader.top + tbHeader.Height, borderColor, 1.5

    ' Obsah
    On Error Resume Next
    Set tbContent = wsTarget.Shapes("txtAnimeOverviewContent")
    On Error GoTo 0
    
    If tbContent Is Nothing Then
        Set tbContent = wsTarget.Shapes.AddTextbox(msoTextOrientationHorizontal, defaultLeftPos, defaultTopPos + 30, 400, 300)
        tbContent.Name = "txtAnimeOverviewContent"
    End If
    
    With tbContent
        .line.Visible = msoFalse
        .TextFrame2.TextRange.text = ""
    End With
    
    Dim limit As Long: limit = IIf(count > POCET_ZOBRAZOVANYCH_POSLEDNICH, POCET_ZOBRAZOVANYCH_POSLEDNICH, count)
    Dim tr As Object
    
    For i = 1 To limit
        If i > 1 Then tbContent.TextFrame2.TextRange.Characters.InsertAfter vbCrLf & vbCrLf
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(i & ". ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrFinal(i, 1) & vbCrLf)
        FormatTxt tr, True, True
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("Začátek sledování: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(Format(arrFinal(i, 2), "DD.MM.YYYY"))
        FormatTxt tr, True, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / konec sledování: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(Format(arrFinal(i, 3), "DD.MM.YYYY") & vbCrLf)
        FormatTxt tr, True, False
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("Délka sledování: ")
        FormatTxt tr, False, False
        
        Dim mins As Long: mins = arrFinal(i, 4)
        Dim hrs As Double: hrs = Round(mins / 60, 1)
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(mins & " minut (" & Replace(hrs, ".", ",") & " " & SklonovaniHodin(hrs) & ")")
        FormatTxt tr, True, False
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / hodnocení: ")
        FormatTxt tr, False, False
        
        Dim rt As String
        If arrFinal(i, 5) >= 0 Then rt = arrFinal(i, 5) & "/10" Else rt = "?/10"
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(rt)
        FormatTxt tr, True, False
    Next i
    
    DrawLine wsTarget, "lineAnimeOverviewContentLeft", tbContent.left, tbContent.top, tbContent.left, tbContent.top + tbContent.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeOverviewContentRight", tbContent.left + tbContent.Width, tbContent.top, tbContent.left + tbContent.Width, tbContent.top + tbContent.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeOverviewContentBottom", tbContent.left, tbContent.top + tbContent.Height, tbContent.left + tbContent.Width, tbContent.top + tbContent.Height, borderColor, 1.5

End Sub

'====================================================================================
' *** SEKCE B: SOUHRN NEJDELŠÍCH SÉRIÍ (Oprava dle původního modulu) ***
'====================================================================================
Private Sub VytvorPrehled_NejdelsiSerie(arrMain As Variant, wsData As Worksheet)
    Dim wsTarget As Worksheet
    Dim dictSeries As Object
    Set dictSeries = CreateObject("Scripting.Dictionary")
    
    Dim i As Long
    Dim sName As String
    Dim rngComments As Range
    
    Set wsTarget = ThisWorkbook.Sheets("OBECNÉ INFORMACE")
    Set rngComments = wsData.Range("C2").Resize(UBound(arrMain, 1), 1)
    
    ' --- 1. AGREGACE DAT ---
    For i = 1 To UBound(arrMain, 1)
        sName = GetSeriesNameFromComment(rngComments.Cells(i, 1))
        
        If sName <> "" Then
            ' Používáme CDbl, aby to četlo desetinné čárky jako v původním modulu
            Dim eps As Long: If IsNumeric(arrMain(i, 7)) Then eps = CLng(arrMain(i, 7)) Else eps = 0
            Dim timePerEp As Double: If IsNumeric(arrMain(i, 8)) Then timePerEp = CDbl(arrMain(i, 8)) Else timePerEp = 0
            Dim rating As Double: If IsNumeric(arrMain(i, 9)) Then rating = CDbl(arrMain(i, 9)) Else rating = -1
            
            If Not dictSeries.Exists(sName) Then
                ' [TotalMin, TotalEp, SumRating, CountRating, NumParts]
                dictSeries.Add sName, Array(0#, 0, 0#, 0, 0)
            End If
            
            Dim stats As Variant
            stats = dictSeries(sName)
            
            stats(0) = stats(0) + (CDbl(timePerEp) * CDbl(eps)) ' Minuty
            stats(1) = stats(1) + eps               ' Epizody
            If rating >= 0 Then
                stats(2) = stats(2) + rating        ' Suma hodnocení
                stats(3) = stats(3) + 1             ' Počet hodnocení
            End If
            stats(4) = stats(4) + 1                 ' Počet částí
            
            dictSeries(sName) = stats
        End If
    Next i
    
    If dictSeries.count = 0 Then Exit Sub
    
    ' --- 2. PŘEVOD NA POLE A ŘAZENÍ ---
    Dim arrExport() As Variant
    ReDim arrExport(1 To dictSeries.count, 1 To 5)
    
    Dim key As Variant, r As Long
    r = 0
    For Each key In dictSeries.keys
        r = r + 1
        Dim dStats As Variant: dStats = dictSeries(key)
        
        arrExport(r, 1) = key           ' Jméno
        arrExport(r, 2) = dStats(0)     ' Minuty (Double)
        arrExport(r, 3) = dStats(1)     ' Epizody
        If dStats(3) > 0 Then arrExport(r, 4) = dStats(2) / dStats(3) Else arrExport(r, 4) = 0
        arrExport(r, 5) = dStats(4)     ' Parts
    Next key
    
    BubbleSort2D arrExport, 2, True
    
    ' --- 3. VYKRESLOVÁNÍ (Podle původního designu) ---
    Dim tbHeader As Shape, tbContent As Shape
    Dim defaultTopPos As Double, defaultLeftPos As Double
    Dim borderColor As Long: borderColor = RGB(0, 0, 0)
    
    With wsTarget.Range("BN1")
        defaultTopPos = .top + 50
        defaultLeftPos = .left - 10
    End With
    
    ' Úklid
    On Error Resume Next
    wsTarget.Shapes("lineAnimeSummaryHeaderTop").Delete
    wsTarget.Shapes("lineAnimeSummaryHeaderLeft").Delete
    wsTarget.Shapes("lineAnimeSummaryHeaderRight").Delete
    wsTarget.Shapes("lineAnimeSummaryContentLeft").Delete
    wsTarget.Shapes("lineAnimeSummaryContentRight").Delete
    wsTarget.Shapes("lineAnimeSummaryContentBottom").Delete
    On Error GoTo 0
    
    ' Hlavička (S VÝPLNÍ)
    On Error Resume Next
    Set tbHeader = wsTarget.Shapes("txtAnimeSummaryHeader")
    On Error GoTo 0
    
    If tbHeader Is Nothing Then
        Set tbHeader = wsTarget.Shapes.AddTextbox(msoTextOrientationHorizontal, defaultLeftPos, defaultTopPos, 400, 30)
        tbHeader.Name = "txtAnimeSummaryHeader"
        tbHeader.Fill.Visible = msoTrue
        tbHeader.Fill.ForeColor.RGB = RGB(128, 0, 0)
        tbHeader.Fill.Transparency = 0
        tbHeader.TextFrame2.TextRange.Font.Fill.Visible = msoTrue
        tbHeader.TextFrame2.TextRange.Font.Fill.Solid
        tbHeader.TextFrame2.TextRange.Font.Fill.ForeColor.RGB = RGB(255, 255, 255)
    End If
    
    With tbHeader
        .line.Visible = msoFalse
        .TextFrame2.TextRange.text = "Nejdelší zhlédnuté Anime série"
        .TextFrame2.TextRange.Font.Bold = msoTrue
        .TextFrame2.TextRange.Font.size = 18
        .TextFrame2.TextRange.Font.Name = "Aptos Narrow"
        .TextFrame2.VerticalAnchor = msoAnchorMiddle
        .TextFrame2.HorizontalAnchor = msoAlignCenter
    End With
    
    DrawLine wsTarget, "lineAnimeSummaryHeaderTop", tbHeader.left, tbHeader.top, tbHeader.left + tbHeader.Width, tbHeader.top, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeSummaryHeaderLeft", tbHeader.left, tbHeader.top, tbHeader.left, tbHeader.top + tbHeader.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeSummaryHeaderRight", tbHeader.left + tbHeader.Width, tbHeader.top, tbHeader.left + tbHeader.Width, tbHeader.top + tbHeader.Height, borderColor, 1.5
    
    ' Obsah (S VÝPLNÍ a PŮVODNÍM FORMÁTOVÁNÍM)
    Dim limit As Long: limit = IIf(UBound(arrExport, 1) > POCET_ZOBRAZOVANYCH_SERII, POCET_ZOBRAZOVANYCH_SERII, UBound(arrExport, 1))

    ' Výpočet výchozí výšky POUZE pro případ, že pole ještě neexistuje
    Dim calcHeight As Double
    calcHeight = (limit * 3 * 14) + (limit * 10)
    If calcHeight < 100 Then calcHeight = 100
    
    On Error Resume Next
    Set tbContent = wsTarget.Shapes("txtAnimeSummaryContent")
    On Error GoTo 0
    
    If tbContent Is Nothing Then
        ' Pokud neexistuje, vytvoříme ho s výchozí vypočítanou výškou
        Set tbContent = wsTarget.Shapes.AddTextbox(msoTextOrientationHorizontal, defaultLeftPos, defaultTopPos + 30, 400, calcHeight)
        tbContent.Name = "txtAnimeSummaryContent"
        tbContent.Fill.Visible = msoTrue
        tbContent.Fill.ForeColor.RGB = RGB(128, 0, 0)
        tbContent.Fill.Transparency = 0
        tbContent.TextFrame2.TextRange.Font.Fill.Visible = msoTrue
        tbContent.TextFrame2.TextRange.Font.Fill.Solid
        tbContent.TextFrame2.TextRange.Font.Fill.ForeColor.RGB = RGB(255, 255, 255)
    Else
        ' Pokud existuje, NEMĚNÍME VÝŠKU.
        ' Pouze zajistíme, že začíná přesně pod hlavičkou.
        tbContent.left = tbHeader.left
        tbContent.top = tbHeader.top + tbHeader.Height
    End If
    
    With tbContent
        .line.Visible = msoFalse
        .TextFrame2.TextRange.text = ""
    End With
    
    Dim tr As Object
    For i = 1 To limit
        If i > 1 Then tbContent.TextFrame2.TextRange.Characters.InsertAfter vbCrLf & vbCrLf
        
        ' 1. VÝPOČET DAT (PŘESNĚ JAKO V PŮVODNÍM MODULU)
        
        ' Minuty: Převedeme na Long (zaokrouhlení/oříznutí), aby to bylo celé číslo
        Dim totalMinutes As Long: totalMinutes = arrExport(i, 2)
        
        ' Hodiny: Počítáme z totalMinutes
        Dim totalHoursDecimal As Double: totalHoursDecimal = totalMinutes / 60
        Dim totalHoursFormatted As String
        
        ' Formátování hodin (kopie logiky původního modulu)
        If totalHoursDecimal > 0 Then
             If totalHoursDecimal = Int(totalHoursDecimal) Then
                 totalHoursFormatted = Format(totalHoursDecimal, "0")
             Else
                 totalHoursFormatted = Format(totalHoursDecimal, "0.0")
                 totalHoursFormatted = Replace(totalHoursFormatted, Application.DecimalSeparator, ",")
                 If Right(totalHoursFormatted, 2) = ",0" Then
                     totalHoursFormatted = left(totalHoursFormatted, Len(totalHoursFormatted) - 2)
                 End If
             End If
        End If
        
        ' Dny: Počítáme z totalHoursDecimal
        Dim totalDaysDecimal As Double: totalDaysDecimal = totalHoursDecimal / 24
        Dim totalDaysFormatted As String
        
        ' Formátování dnů (kopie logiky původního modulu)
        If totalDaysDecimal > 0 Then
             If totalDaysDecimal = Int(totalDaysDecimal) Then
                 totalDaysFormatted = Format(totalDaysDecimal, "0")
             Else
                 totalDaysFormatted = Format(totalDaysDecimal, "0.0")
                 totalDaysFormatted = Replace(totalDaysFormatted, Application.DecimalSeparator, ",")
                 If Right(totalDaysFormatted, 2) = ",0" Then
                     totalDaysFormatted = left(totalDaysFormatted, Len(totalDaysFormatted) - 2)
                 End If
             End If
        End If
        
        ' Sestavení textu pro délku
        Dim durationTextFormatted As String
        durationTextFormatted = totalMinutes & " minut"
        If totalHoursDecimal > 0 Then durationTextFormatted = durationTextFormatted & " (" & totalHoursFormatted & " hodin)"
        If totalDaysDecimal > 0 Then durationTextFormatted = durationTextFormatted & " (" & totalDaysFormatted & " " & SklonovaniDni(totalDaysDecimal) & ")"
        
        ' Hodnocení (kopie logiky původního modulu - 0.#)
        Dim avgVal As Double: avgVal = arrExport(i, 4)
        Dim avgRatingFormatted As String
        If avgVal > 0 Then
            If avgVal = Int(avgVal) Then
                avgRatingFormatted = Format(avgVal, "0")
            Else
                avgRatingFormatted = Replace(Format(avgVal, "0.#"), ".", ",")
            End If
        Else
            avgRatingFormatted = "??"
        End If
        
        ' 2. VYPISOVÁNÍ DO TEXTBOXU
        
        ' Řádek 1: Název
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(i & ". ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrExport(i, 1) & vbCrLf)
        FormatTxt tr, True, True
        
        ' Řádek 2: Délka
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("Délka sledování: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(durationTextFormatted & vbCrLf)
        FormatTxt tr, True, False
        
        ' Řádek 3: Info
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("Počet dílů: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrExport(i, 5) & " / ")
        FormatTxt tr, True, False
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("počet epizod: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrExport(i, 3) & " / ")
        FormatTxt tr, True, False
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("průměrné hodnocení: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(avgRatingFormatted & "/10")
        FormatTxt tr, True, False
    Next i
    
    DrawLine wsTarget, "lineAnimeSummaryContentLeft", tbContent.left, tbContent.top, tbContent.left, tbContent.top + tbContent.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeSummaryContentRight", tbContent.left + tbContent.Width, tbContent.top, tbContent.left + tbContent.Width, tbContent.top + tbContent.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeSummaryContentBottom", tbContent.left, tbContent.top + tbContent.Height, tbContent.left + tbContent.Width, tbContent.top + tbContent.Height, borderColor, 1.5

End Sub

'====================================================================================
' *** SEKCE C: NEJRYCHLEJŠÍ BINGE ***
'====================================================================================
Private Sub VytvorPrehled_NejrychlejsiBinge(arrMain As Variant)
    Dim wsTarget As Worksheet
    Dim i As Long, count As Long, r As Long
    
    Dim tbHeader As Shape, tbContent As Shape
    Dim defaultTopPos As Double, defaultLeftPos As Double
    Dim borderColor As Long: borderColor = RGB(0, 0, 0)
    
    Set wsTarget = ThisWorkbook.Sheets("OBECNÉ INFORMACE")
    
    ' --- 1. FILTRACE DAT ---
    ' Binge pravidla:
    '   - Musí mít Start i End datum
    '   - Celková délka anime > 180 minut
    '   - Tempo sledování >= 77 minut/den
    
    Dim arrFiltered() As Variant
    ReDim arrFiltered(1 To UBound(arrMain, 1), 1 To 5)
    count = 0
    
    For i = 1 To UBound(arrMain, 1)
        Dim valStart As Variant: valStart = arrMain(i, 10) ' L = Start
        Dim valEnd As Variant: valEnd = arrMain(i, 11)     ' M = End
        Dim valName As String: valName = CStr(arrMain(i, 1)) ' C = Název
        
        If IsDate(valStart) And IsDate(valEnd) And Len(valName) > 0 Then
            ' Výpočet celkových minut
            Dim eps As Long: If IsNumeric(arrMain(i, 7)) Then eps = CLng(arrMain(i, 7)) Else eps = 0
            Dim timePerEp As Double: If IsNumeric(arrMain(i, 8)) Then timePerEp = CDbl(arrMain(i, 8)) Else timePerEp = 0
            Dim totalMin As Double: totalMin = CDbl(eps) * timePerEp
            
            ' Pravidlo 1: Celková délka > 180 minut
            If totalMin > 180 Then
                ' Počet dnů (min 1 den = stejný start/end)
                Dim pocetDni As Long: pocetDni = DateDiff("d", CDate(valStart), CDate(valEnd)) + 1
                If pocetDni < 1 Then pocetDni = 1
                
                ' Pravidlo 2: Tempo >= 77 minut/den
                Dim minutZaDen As Double: minutZaDen = totalMin / pocetDni
                
                If minutZaDen >= 77 Then
                    count = count + 1
                    arrFiltered(count, 1) = valName      ' Název
                    arrFiltered(count, 2) = pocetDni     ' Počet dnů
                    arrFiltered(count, 3) = totalMin     ' Celkové minuty
                    arrFiltered(count, 4) = eps          ' Počet epizod
                    arrFiltered(count, 5) = minutZaDen   ' Minut za den (pro řazení)
                End If
            End If
        End If
    Next i
    
    If count = 0 Then Exit Sub
    
    ' Kopie do pole správné velikosti
    Dim arrBinge() As Variant
    ReDim arrBinge(1 To count, 1 To 5)
    For i = 1 To count
        For r = 1 To 5
            arrBinge(i, r) = arrFiltered(i, r)
        Next r
    Next i
    
    ' --- 2. ŘAZENÍ ---
    ' Sestupně podle minut za den (nejvíce hodin/den = nejrychlejší binge nahoře)
    BubbleSort2D arrBinge, 5, True
    
    ' --- 3. VYKRESLOVÁNÍ ---
    With wsTarget.Range("BW1")
        defaultTopPos = .top + 50
        defaultLeftPos = .left - 10
    End With
    
    ' Úklid starých čar
    On Error Resume Next
    wsTarget.Shapes("lineAnimeBingeHeaderTop").Delete
    wsTarget.Shapes("lineAnimeBingeHeaderLeft").Delete
    wsTarget.Shapes("lineAnimeBingeHeaderRight").Delete
    wsTarget.Shapes("lineAnimeBingeContentLeft").Delete
    wsTarget.Shapes("lineAnimeBingeContentRight").Delete
    wsTarget.Shapes("lineAnimeBingeContentBottom").Delete
    On Error GoTo 0
    
    ' Hlavička (S VÝPLNÍ)
    On Error Resume Next
    Set tbHeader = wsTarget.Shapes("txtAnimeBingeHeader")
    On Error GoTo 0
    
    If tbHeader Is Nothing Then
        Set tbHeader = wsTarget.Shapes.AddTextbox(msoTextOrientationHorizontal, defaultLeftPos, defaultTopPos, 400, 30)
        tbHeader.Name = "txtAnimeBingeHeader"
        tbHeader.Fill.Visible = msoTrue
        tbHeader.Fill.ForeColor.RGB = RGB(128, 0, 0)
        tbHeader.Fill.Transparency = 0
        tbHeader.TextFrame2.TextRange.Font.Fill.Visible = msoTrue
        tbHeader.TextFrame2.TextRange.Font.Fill.Solid
        tbHeader.TextFrame2.TextRange.Font.Fill.ForeColor.RGB = RGB(255, 255, 255)
    End If
    
    With tbHeader
        .line.Visible = msoFalse
        .TextFrame2.TextRange.text = "Nejrychlejší Binge"
        .TextFrame2.TextRange.Font.Bold = msoTrue
        .TextFrame2.TextRange.Font.size = 18
        .TextFrame2.TextRange.Font.Name = "Aptos Narrow"
        .TextFrame2.VerticalAnchor = msoAnchorMiddle
        .TextFrame2.HorizontalAnchor = msoAlignCenter
    End With
    
    DrawLine wsTarget, "lineAnimeBingeHeaderTop", tbHeader.left, tbHeader.top, tbHeader.left + tbHeader.Width, tbHeader.top, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeBingeHeaderLeft", tbHeader.left, tbHeader.top, tbHeader.left, tbHeader.top + tbHeader.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeBingeHeaderRight", tbHeader.left + tbHeader.Width, tbHeader.top, tbHeader.left + tbHeader.Width, tbHeader.top + tbHeader.Height, borderColor, 1.5
    
    ' Obsah (S VÝPLNÍ)
    Dim limit As Long: limit = IIf(count > POCET_ZOBRAZOVANYCH_BINGE, POCET_ZOBRAZOVANYCH_BINGE, count)
    
    Dim calcHeight As Double
    calcHeight = (limit * 2 * 14) + (limit * 10)
    If calcHeight < 100 Then calcHeight = 100
    
    On Error Resume Next
    Set tbContent = wsTarget.Shapes("txtAnimeBingeContent")
    On Error GoTo 0
    
    If tbContent Is Nothing Then
        Set tbContent = wsTarget.Shapes.AddTextbox(msoTextOrientationHorizontal, defaultLeftPos, defaultTopPos + 30, 400, calcHeight)
        tbContent.Name = "txtAnimeBingeContent"
        tbContent.Fill.Visible = msoTrue
        tbContent.Fill.ForeColor.RGB = RGB(128, 0, 0)
        tbContent.Fill.Transparency = 0
        tbContent.TextFrame2.TextRange.Font.Fill.Visible = msoTrue
        tbContent.TextFrame2.TextRange.Font.Fill.Solid
        tbContent.TextFrame2.TextRange.Font.Fill.ForeColor.RGB = RGB(255, 255, 255)
    Else
        tbContent.left = tbHeader.left
        tbContent.top = tbHeader.top + tbHeader.Height
    End If
    
    With tbContent
        .line.Visible = msoFalse
        .TextFrame2.TextRange.text = ""
    End With
    
    Dim tr As Object
    For i = 1 To limit
        If i > 1 Then tbContent.TextFrame2.TextRange.Characters.InsertAfter vbCrLf & vbCrLf
        
        ' Výpočty
        Dim bingeDays As Long: bingeDays = arrBinge(i, 2)
        Dim bingeMinutes As Double: bingeMinutes = arrBinge(i, 3)
        Dim bingeHours As Double: bingeHours = Round(bingeMinutes / 60, 1)
        Dim bingeEps As Long: bingeEps = arrBinge(i, 4)
        
        ' Formátování hodin
        Dim bingeHoursFormatted As String
        If bingeHours = Int(bingeHours) Then
            bingeHoursFormatted = Format(bingeHours, "0")
        Else
            bingeHoursFormatted = Replace(Format(bingeHours, "0.0"), Application.DecimalSeparator, ",")
            If Right(bingeHoursFormatted, 2) = ",0" Then
                bingeHoursFormatted = left(bingeHoursFormatted, Len(bingeHoursFormatted) - 2)
            End If
        End If
        
        ' Řádek 1: Název
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(i & ". ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrBinge(i, 1) & vbCrLf)
        FormatTxt tr, True, True
        
        ' Řádek 2: Počet dnů / počet epizod / počet hodin
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("Počet dnů: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(bingeDays)
        FormatTxt tr, True, False
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / počet epizod: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(bingeEps & " EP")
        FormatTxt tr, True, False
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / počet hodin: ")
        FormatTxt tr, False, False
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(bingeHoursFormatted & " h")
        FormatTxt tr, True, False
    Next i
    
    DrawLine wsTarget, "lineAnimeBingeContentLeft", tbContent.left, tbContent.top, tbContent.left, tbContent.top + tbContent.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeBingeContentRight", tbContent.left + tbContent.Width, tbContent.top, tbContent.left + tbContent.Width, tbContent.top + tbContent.Height, borderColor, 1.5
    DrawLine wsTarget, "lineAnimeBingeContentBottom", tbContent.left, tbContent.top + tbContent.Height, tbContent.left + tbContent.Width, tbContent.top + tbContent.Height, borderColor, 1.5

End Sub

Function GetSeriesNameFromComment(cell As Range) As String
    Dim commentText As String
    Dim lines As Variant, line As Variant

    If Not cell.Comment Is Nothing Then
        commentText = cell.Comment.text
        lines = Split(commentText, vbLf)
        For Each line In lines
            If InStr(1, Trim(line), "Název série:", vbTextCompare) = 1 Then
                GetSeriesNameFromComment = Trim(Mid(line, InStr(1, line, ":") + 1))
                Exit Function
            End If
        Next line
    End If
    
    If ZARADIT_POLOZKY_BEZ_POZNAMKY Then
        GetSeriesNameFromComment = CStr(cell.Value)
    Else
        GetSeriesNameFromComment = ""
    End If
End Function

Private Function SklonovaniDni(ByVal n As Variant) As String
    Dim d As Double: d = Abs(CDbl(n))
    ' Zaokrouhlení na 1 des. místo (odpovídá formátování zobrazení)
    Dim rounded As Double: rounded = Round(d, 1)
    Dim v As Long: v = CLng(Int(rounded))
    
    ' Desetinné číslo (např. 1,1 dne, 2,5 dne)
    If rounded <> Int(rounded) Then
        SklonovaniDni = "dne"
    ElseIf v = 1 Then
        SklonovaniDni = "den"
    ElseIf v >= 2 And v <= 4 Then
        SklonovaniDni = "dny"
    Else
        SklonovaniDni = "dní"
    End If
End Function

Private Function SklonovaniHodin(ByVal n As Variant) As String
    Dim d As Double: d = Abs(CDbl(n))
    Dim v As Long: v = CLng(Int(d))
    
    ' Desetinné číslo (např. 2,2 hodin)
    If d <> Int(d) Then
        SklonovaniHodin = "hodin"
    ElseIf v = 1 Then
        SklonovaniHodin = "hodina"
    ElseIf v >= 2 And v <= 4 Then
        SklonovaniHodin = "hodiny"
    Else
        SklonovaniHodin = "hodin"
    End If
End Function

Private Sub DrawLine(ws As Worksheet, sName As String, x1 As Double, y1 As Double, x2 As Double, y2 As Double, col As Long, w As Double)
    Dim ln As Shape
    Set ln = ws.Shapes.AddLine(x1, y1, x2, y2)
    With ln
        .Name = sName
        .line.ForeColor.RGB = col
        .line.weight = w
    End With
End Sub

Private Sub FormatTxt(rng As Object, bBold As Boolean, bItalic As Boolean)
    With rng.Font
        .Name = "Aptos Narrow"
        .size = 11
        .Bold = bBold
        .Italic = bItalic
    End With
End Sub
