Sub GenerateTechnicalSheetReport()
    ' ==========================================
    ' TECHNICKÝ REPORT - OBJEKTY A MAKRA (PLNÝ TEXT)
    ' Autor: Gemini (Upraveno: Bez limitu znaků)
    ' ==========================================
    
    Dim wdApp As Object
    Dim wdDoc As Object
    Dim ws As Worksheet
    Dim sPath As String, sFileName As String
    Dim shp As Shape
    Dim tbl As ListObject
    Dim pt As PivotTable
    Dim wdTable As Object
    Dim r As Long
    Dim macroName As String
    Dim objType As String
    Dim fullText As String
    
    ' Konstanty Word
    Const wdFormatDocumentDefault = 16
    Const wdStyleHeading1 = -2
    Const wdStyleHeading3 = -4
    Const wdPageBreak = 7
    Const wdAutoFitWindow = 2
    
    ' 1. Nastavení cesty
    sPath = "C:\Users\macou\Desktop\Anime list - experimenting\"
    If Dir(sPath, vbDirectory) = "" Then Exit Sub
    
    ' 2. Start Wordu
    On Error Resume Next
    Set wdApp = GetObject(, "Word.Application")
    If err.Number <> 0 Then Set wdApp = CreateObject("Word.Application")
    On Error GoTo 0
    
    wdApp.Visible = True
    Set wdDoc = wdApp.Documents.Add
    
    ' Úvod
    With wdApp.Selection
        .Style = wdStyleHeading1
        .TypeText text:="Technický Blueprint - Objekty a Funkce (Full Data)"
        .TypeParagraph
        .TypeText text:="Generováno: " & Format(Now, "dd.mm.yyyy hh:mm")
        .TypeParagraph
        .TypeParagraph
    End With
    
    ' 3. Smyčka přes listy
    For Each ws In ThisWorkbook.Worksheets
        
        With wdApp.Selection
            ' --- Nadpis Listu ---
            .Style = wdStyleHeading1
            .TypeText text:="List: " & ws.Name & " (" & ws.CodeName & ")"
            .TypeParagraph
            
            ' --- A) ANALÝZA OBJEKTŮ ---
            If ws.Shapes.count > 0 Then
                .Style = wdStyleHeading3
                .TypeText text:="Objekty (" & ws.Shapes.count & ")"
                .TypeParagraph
                
                ' Tabulka
                Set wdTable = wdDoc.Tables.Add(.Range, ws.Shapes.count + 1, 5)
                wdTable.Borders.Enable = True
                wdTable.Rows(1).Range.Font.Bold = True
                wdTable.Rows(1).Shading.BackgroundPatternColor = 14737632
                
                ' Hlavičky
                wdTable.cell(1, 1).Range.text = "Název Objektu"
                wdTable.cell(1, 2).Range.text = "Typ"
                wdTable.cell(1, 3).Range.text = "Makro"
                wdTable.cell(1, 4).Range.text = "Pozice"
                wdTable.cell(1, 5).Range.text = "Data (Alt Text / Obsah)"
                
                r = 2
                For Each shp In ws.Shapes
                    ' Typ
                    Select Case shp.Type
                        Case 1: objType = "AutoShape"
                        Case 3: objType = "Graf"
                        Case 4: objType = "Komentář" ' msoComment
                        Case 8: objType = "Tlačítko"
                        Case 12: objType = "ActiveX"
                        Case 13: objType = "Obrázek"
                        Case 17: objType = "TextBox"
                        Case Else: objType = "Jiné (" & shp.Type & ")"
                    End Select
                    
                    ' Makro
                    macroName = ""
                    On Error Resume Next
                    macroName = shp.OnAction
                    On Error GoTo 0
                    
                    ' --- ZÍSKÁNÍ DAT (Tady byla změna) ---
                    fullText = ""
                    
                    ' 1. Priorita: Alternativní text (metadata)
                    fullText = shp.AlternativeText
                    
                    ' --- VYLEPŠENÍ PRO GRAFY ---
' Pokud je to graf, zkusíme získat jeho název (ChartTitle)
If shp.Type = 3 Then ' 3 = msoChart
    On Error Resume Next
    If shp.HasChart Then
         If shp.Chart.HasTitle Then fullText = shp.Chart.ChartTitle.text
    End If
    On Error GoTo 0
End If
' ---------------------------
                    
                    ' 2. Priorita: Pokud je AltText prázdný, zkusíme viditelný text
                    If Trim(fullText) = "" Then
                        On Error Resume Next
                        ' Zkusíme přečíst text z rámečku (funguje pro TextBoxy a Tlačítka)
                        fullText = shp.TextFrame.Characters.text
                        ' Zkusíme přečíst text z ActiveX
                        If fullText = "" Then fullText = shp.OLEFormat.Object.Caption
                        On Error GoTo 0
                    End If
                    
                    ' Pojistka proti pádu Wordu u extrémně dlouhých textů (např. Base64 obrázky)
                    If Len(fullText) > 2000 Then
                        fullText = left(fullText, 2000) & " ... [TEXT PŘÍLIŠ DLOUHÝ - OŘÍZNUTO]"
                    End If
                    
                    ' Zápis
                    wdTable.cell(r, 1).Range.text = shp.Name
                    wdTable.cell(r, 2).Range.text = objType
                    wdTable.cell(r, 3).Range.text = macroName
                    ' --- BEZPEČNÉ ZÍSKÁNÍ POZICE ---
                    Dim objPos As String
                    objPos = "N/A" ' Výchozí hodnota pro případ chyby
                    
                    On Error Resume Next ' Dočasně ignorovat chyby
                    objPos = shp.TopLeftCell.Address(False, False)
                    On Error GoTo 0 ' Vrátit hlášení chyb zpět
                    
                    wdTable.cell(r, 4).Range.text = objPos
                    ' -------------------------------
                    wdTable.cell(r, 5).Range.text = fullText ' Zde je nyní kompletní text
                    
                    r = r + 1
                Next shp
                
                wdTable.AutoFitBehavior wdAutoFitWindow
                .EndKey Unit:=6
                .TypeParagraph
            End If
            
            ' --- B) ANALÝZA TABULEK ---
            If ws.ListObjects.count > 0 Then
                .Style = wdStyleHeading3
                .TypeText text:="Tabulky (ListObjects)"
                .TypeParagraph
                Set wdTable = wdDoc.Tables.Add(.Range, ws.ListObjects.count + 1, 3)
                wdTable.Borders.Enable = True
                wdTable.Rows(1).Range.Font.Bold = True
                wdTable.cell(1, 1).Range.text = "Název"
                wdTable.cell(1, 2).Range.text = "Rozsah"
                wdTable.cell(1, 3).Range.text = "Zdroj"
                r = 2
                For Each tbl In ws.ListObjects
                    wdTable.cell(r, 1).Range.text = tbl.Name
                    wdTable.cell(r, 2).Range.text = tbl.Range.Address
                    wdTable.cell(r, 3).Range.text = tbl.SourceType
                    r = r + 1
                Next tbl
                wdTable.AutoFitBehavior wdAutoFitWindow
                .EndKey Unit:=6
                .TypeParagraph
            End If
            
            ' --- C) ANALÝZA KONTINGENČNÍCH TABULEK (PIVOT TABLES) ---
        If ws.PivotTables.count > 0 Then
            .Style = wdStyleHeading3
            .TypeText text:="Kontingenční tabulky (PivotTables)"
            .TypeParagraph

            Set wdTable = wdDoc.Tables.Add(.Range, ws.PivotTables.count + 1, 3)
            wdTable.Borders.Enable = True
            wdTable.Rows(1).Range.Font.Bold = True
            wdTable.Rows(1).Shading.BackgroundPatternColor = 15132390 ' Jemně šedá

            wdTable.cell(1, 1).Range.text = "Název Pivotu"
            wdTable.cell(1, 2).Range.text = "Rozsah dat (Source)"
            wdTable.cell(1, 3).Range.text = "Umístění"

            r = 2
            For Each pt In ws.PivotTables
                wdTable.cell(r, 1).Range.text = pt.Name

                ' Získání zdroje dat (může být text nebo odkaz)
                On Error Resume Next
                wdTable.cell(r, 2).Range.text = pt.sourceData
                If err.Number <> 0 Then wdTable.cell(r, 2).Range.text = "[Externí/Složité]"
                On Error GoTo 0

                wdTable.cell(r, 3).Range.text = pt.TableRange1.Address
                r = r + 1
            Next pt

            wdTable.AutoFitBehavior 2 ' wdAutoFitWindow
            .EndKey Unit:=6
            .TypeParagraph
        End If
            
            .InsertBreak Type:=wdPageBreak
        End With
    Next ws
    
    ' Uložení
    sFileName = Format(Now, "dd.mm.yyyy, hh,mm") & "_TechBlueprints_FULL.docx"
    On Error Resume Next
    If Dir(sPath & sFileName) <> "" Then Kill sPath & sFileName
    On Error GoTo 0
    wdDoc.SaveAs2 FileName:=sPath & sFileName
    
    MsgBox "Hotovo! Full Blueprint uložen: " & sFileName, vbInformation
    
    Set wdDoc = Nothing
    Set wdApp = Nothing
End Sub
