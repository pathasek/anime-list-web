Const COL_MELODIE As String = "Melodie"
Const COL_VIDEO As String = "Videoklip"

'————————————————————————————
' 2) Wrapper – Spustí všechna makra (CENTRALIZOVANÁ DATA)
'————————————————————————————
Public Sub AktualizovatOpEdGrafy()
    On Error GoTo ErrWrapper
    
    Dim wsData As Worksheet, wsCharts As Worksheet, wsAL As Worksheet
    Dim tbl As ListObject
    Dim arrData As Variant
    Dim dictMap As Object
    Dim dictSerieComments As Object
    Dim dataUDT() As AnimeRow
    Dim i As Long
    
    ' Optimalizace prostředí
    With Application
        .ScreenUpdating = False
        .Calculation = xlCalculationManual
        .EnableEvents = False
    End With
    
    Set wsData = ThisWorkbook.Sheets(WS_OPED)
    Set wsCharts = ThisWorkbook.Sheets(WS_CHARTS)
    Set wsAL = ThisWorkbook.Sheets(WS_AL)
    Set tbl = wsData.ListObjects(TBL_OPED)
    
    ' 1. Načtení dat tabulky do pole
    If tbl.DataBodyRange Is Nothing Then
        MsgBox "Tabulka OP/ED je prázdná.", vbExclamation
        GoTo CleanExit
    End If
    ' OPTIMALIZACE FÁZE 5: Value2 je rychlejší (nekonvertuje typy jako Currency/Date)
    arrData = tbl.DataBodyRange.Value2
    
    ' 2. Mapování sloupců (Název -> Index)
    Set dictMap = CreateObject("Scripting.Dictionary")
    dictMap.CompareMode = vbTextCompare
    For i = 1 To tbl.ListColumns.count
        dictMap(tbl.ListColumns(i).Name) = i
    Next i
    
    ' 3. PARSING DAT (Pre-processing)
    dataUDT = ParseDataToUDT(arrData, dictMap)
    
    ' 4. Načtení komentářů pro Série
    Set dictSerieComments = LoadSerieComments(wsAL)
    
    ' 5. Volání grafů (použití UDT)
    Call VytvorGrafHodnoceniBreakdown(dataUDT, wsCharts)
    Call VytvorGrafRadarKategorii(dataUDT, wsCharts)
    Call VytvorGrafTopAutori(dataUDT, wsCharts)
    Call VytvorGrafFrissonVliv(dataUDT, wsCharts)
    Call VytvorGrafTopSerie(dataUDT, wsCharts, dictSerieComments)
    Call VytvorGrafTopAutoriAvg(dataUDT, wsCharts)
    Call VytvorGrafPrumerOpVsEd(dataUDT, wsCharts)
    Call VytvorGrafJazykAnalyza(dataUDT, wsCharts)
    Call VytvorGrafSingAlong(dataUDT, wsCharts)
    Call VytvorGrafHypeBubble(dataUDT, wsCharts)
    
CleanExit:
    With Application
        .ScreenUpdating = True
        .Calculation = xlCalculationAutomatic
        .EnableEvents = True
    End With
    ' MsgBox "Grafy OP/ED byly úspěšně aktualizovány.", vbInformation
    Exit Sub
    
ErrWrapper:
    With Application
        .ScreenUpdating = True
        .Calculation = xlCalculationAutomatic
        .EnableEvents = True
    End With
    MsgBox "Kritická chyba: " & err.Description, vbCritical
End Sub

Private Function ParseDataToUDT(arr As Variant, map As Object) As AnimeRow()
    Dim rCount As Long, i As Long
    Dim res() As AnimeRow
    Dim idxAni As Long, idxTyp As Long, idxSong As Long, idxAut As Long
    Dim idxLyr As Long, idxEmo As Long, idxMel As Long, idxVid As Long, idxHlas As Long
    Dim idxFri As Long, idxRat As Long, idxJaz As Long, idxSing As Long
    
    rCount = UBound(arr, 1)
    ReDim res(1 To rCount)
    
    idxAni = map(COL_ANIME): idxTyp = map(COL_TYP): idxSong = map(COL_SONG)
    idxAut = map(COL_AUTOR): idxLyr = map(COL_LYRICS): idxEmo = map(COL_EMOCE)
    idxMel = map(COL_MELODIE): idxVid = map(COL_VIDEO): idxHlas = map(COL_HLAS)
    idxFri = map(COL_FRISSON): idxRat = map(COL_RATING_FINAL): idxJaz = map(COL_JAZYK)
    idxSing = map(COL_SINGALONG)
    
    Dim v As Variant
    
    For i = 1 To rCount
        ' OPTIMALIZACE FÁZE 5: Inline Check místo volání Nz() (ušetří 30k+ volání funkce)
        With res(i)
            .ID = i
            
            ' Name
            v = arr(i, idxAni)
            If IsError(v) Or IsEmpty(v) Then .Name = "" Else .Name = Trim(CStr(v))
            
            ' Typ
            v = arr(i, idxTyp)
            If IsError(v) Or IsEmpty(v) Then .typ = "" Else .typ = Trim(CStr(v))
            
            ' Song
            v = arr(i, idxSong)
            If IsError(v) Or IsEmpty(v) Then .Song = "" Else .Song = Trim(CStr(v))
            
            ' Author
            v = arr(i, idxAut)
            If IsError(v) Or IsEmpty(v) Then .Author = "" Else .Author = Trim(CStr(v))
            
            ' Language
            v = arr(i, idxJaz)
            If IsError(v) Or IsEmpty(v) Then .Language = "" Else .Language = Trim(CStr(v))
            
            ' Rating Final
            v = arr(i, idxRat)
            If IsNumeric(v) And Not IsEmpty(v) And Not IsError(v) Then
                .RatingFinal = CDbl(v)
                .HasRating = True
            Else
                .RatingFinal = 0
                .HasRating = False
            End If
            
            ' Frisson (Boolean)
            v = arr(i, idxFri)
            If IsError(v) Or IsEmpty(v) Then
                .Frisson = False
            Else
                .Frisson = (UCase(Trim(CStr(v))) = "ANO")
            End If
            
            ' SingAlong
            v = arr(i, idxSing)
            If IsNumeric(v) And Not IsEmpty(v) And Not IsError(v) Then
                .SingAlong = CDbl(v)
                .HasSingAlong = True
            Else
                .SingAlong = 0
                .HasSingAlong = False
            End If
            
            ' Radar Metrics (Lyrics, Emoce, Melodie, Video, Hlas)
            ' Logic: All must be valid numeric > 0
            v = arr(i, idxLyr)
            If IsNumeric(v) And Not IsEmpty(v) And Not IsError(v) Then .Lyrics = CDbl(v) Else .Lyrics = -1
            
            v = arr(i, idxEmo)
            If IsNumeric(v) And Not IsEmpty(v) And Not IsError(v) Then .Emotion = CDbl(v) Else .Emotion = -1
            
            v = arr(i, idxMel)
            If IsNumeric(v) And Not IsEmpty(v) And Not IsError(v) Then .Melody = CDbl(v) Else .Melody = -1
            
            v = arr(i, idxVid)
            If IsNumeric(v) And Not IsEmpty(v) And Not IsError(v) Then .Video = CDbl(v) Else .Video = -1
            
            v = arr(i, idxHlas)
            If IsNumeric(v) And Not IsEmpty(v) And Not IsError(v) Then .Voice = CDbl(v) Else .Voice = -1
            
            ' Radar Validity Check
            .IsRadarValid = (.Lyrics >= 0 And .Emotion >= 0 And .Melody >= 0 And .Video >= 0 And .Voice >= 0)
        End With
    Next i
    
    ParseDataToUDT = res
End Function

' Pomocná funkce pro načtení komentářů sérií (pouze ty s komentářem)
Private Function LoadSerieComments(ws As Worksheet) As Object
    Dim dict As Object
    Dim rngComments As Range, cell As Range
    Dim lastRow As Long
    
    Set dict = CreateObject("Scripting.Dictionary")
    dict.CompareMode = vbTextCompare
    
    lastRow = ws.Cells(ws.Rows.count, COL_AL_ANIME_LETTER).End(xlUp).row
    On Error Resume Next
    Set rngComments = ws.Range(COL_AL_ANIME_LETTER & "2:" & COL_AL_ANIME_LETTER & lastRow).SpecialCells(xlCellTypeComments)
    On Error GoTo 0
    
    If Not rngComments Is Nothing Then
        For Each cell In rngComments
            ' Ukládáme: Klíč = Název Anime, Hodnota = Text komentáře
            ' OPTIMALIZACE FÁZE 6: Value2
            dict(Trim(cell.Value2)) = cell.Comment.text
        Next cell
    End If
    
    Set LoadSerieComments = dict
End Function

' 3) Graf 1: Rozložení Finálního hodnocení

'————————————————————————————
' 3) Graf 1: Rozložení Finálního hodnocení (Array Optimized)
'————————————————————————————
Private Sub VytvorGrafHodnoceniBreakdown(data() As AnimeRow, wsC As Worksheet)
    On Error GoTo ErrH
    Dim cntTotal(1 To 10) As Long, cntOP(1 To 10) As Long, cntED(1 To 10) As Long
    Dim i As Long
    Dim rInt As Long
    Dim isNew As Boolean
    
    For i = LBound(data) To UBound(data)
        If data(i).HasRating And Len(data(i).typ) > 0 Then
            rInt = Int(data(i).RatingFinal)
            If rInt >= 1 And rInt <= 10 Then
                cntTotal(rInt) = cntTotal(rInt) + 1
                If data(i).typ = "OP" Then cntOP(rInt) = cntOP(rInt) + 1
                If data(i).typ = "ED" Then cntED(rInt) = cntED(rInt) + 1
            End If
        End If
    Next i
    
    ' Osa X jen pro existující hodnoty
    Dim catList As Object: Set catList = CreateObject("System.Collections.ArrayList")
    For i = 1 To 10
        If cntTotal(i) > 0 Then catList.Add i
    Next i
    If catList.count = 0 Then Exit Sub
    
    Dim cats() As String, vTot() As Long, vOP() As Long, vED() As Long
    Dim cnt As Long: cnt = catList.count
    ReDim cats(0 To cnt - 1): ReDim vTot(0 To cnt - 1)
    ReDim vOP(0 To cnt - 1): ReDim vED(0 To cnt - 1)
    
    Dim currIdx As Long
    For i = 0 To cnt - 1
        currIdx = catList(i)
        cats(i) = "Hodnocení " & currIdx
        vTot(i) = cntTotal(currIdx)
        vOP(i) = cntOP(currIdx)
        vED(i) = cntED(currIdx)
    Next i
    
    Dim chtObj As ChartObject, cht As Chart
    On Error Resume Next
    Set chtObj = wsC.ChartObjects("GrafHodnoceniOpEd")
    On Error GoTo ErrH
    
    If chtObj Is Nothing Then
        isNew = True
        Set chtObj = wsC.ChartObjects.Add(20, 50, 450, 350)
        chtObj.Name = "GrafHodnoceniOpEd"
        Set cht = chtObj.Chart
        With cht
            .ChartType = xlColumnClustered
            .HasTitle = True: .ChartTitle.text = "Rozložení hodnocení (Počet)"
            .ChartTitle.Font.Name = "Aptos Narrow": .ChartTitle.Font.size = 12: .ChartTitle.Font.Bold = True
            .HasLegend = True: .Legend.Position = xlLegendPositionBottom
            .Axes(xlValue).HasTitle = True: .Axes(xlValue).AxisTitle.text = "Počet hodnocení"
        End With
    Else
        isNew = False
        Set cht = chtObj.Chart
    End If
    
    cht.ChartType = xlColumnClustered
    
    Dim srsOP As Series, srsED As Series, srsCelkem As Series
    Set srsCelkem = GetSeriesByName(cht, "Počet (Celkem)")
    Set srsOP = GetSeriesByName(cht, "Počet (OP)")
    Set srsED = GetSeriesByName(cht, "Počet (ED)")
    
    Call ApplyDataWithLabelSave(srsCelkem, cats, vTot, isNew)
    Call ApplyDataWithLabelSave(srsOP, cats, vOP, isNew)
    Call ApplyDataWithLabelSave(srsED, cats, vED, isNew)
    
    On Error Resume Next
    srsOP.Format.Fill.ForeColor.RGB = 10498160
    srsED.Format.Fill.ForeColor.RGB = 15773696
    
    Exit Sub
ErrH:
    MsgBox "Chyba Graf 1: " & err.Description
End Sub

'————————————————————————————
' 4) Graf 3: Radar kategorií (OPRAVENO)
'————————————————————————————
'————————————————————————————
' 4) Graf 3: Radar kategorií (OPRAVENO)
'————————————————————————————
Private Sub VytvorGrafRadarKategorii(data() As AnimeRow, wsC As Worksheet)
    On Error GoTo ErrH
    Dim sums(0 To 4) As Double, avgs(0 To 4) As Double
    Dim i As Long, j As Long, validRows As Long
    
    ' Indicies mapping for result arrays: 0=Lyrics, 1=Emotion, 2=Melody, 3=Video, 4=Voice
    
    validRows = 0
    For i = LBound(data) To UBound(data)
        If data(i).IsRadarValid Then
            validRows = validRows + 1
            sums(0) = sums(0) + data(i).Lyrics
            sums(1) = sums(1) + data(i).Emotion
            sums(2) = sums(2) + data(i).Melody
            sums(3) = sums(3) + data(i).Video
            sums(4) = sums(4) + data(i).Voice
        End If
    Next i
    
    If validRows = 0 Then Exit Sub
    
    Dim catsRaw As Variant: catsRaw = Array(COL_LYRICS, COL_EMOCE, COL_MELODIE, COL_VIDEO, COL_HLAS)
    Dim catsFmt(0 To 4) As String
    For j = 0 To 4
        avgs(j) = sums(j) / validRows
        catsFmt(j) = catsRaw(j) & ": " & Replace(Format(avgs(j), "0.0"), ".", ",") & "/10"
    Next j
    
    Dim chtObj As ChartObject, cht As Chart
    On Error Resume Next
    Set chtObj = wsC.ChartObjects("GrafRadarKategorii")
    On Error GoTo ErrH
    
    If chtObj Is Nothing Then
        Set chtObj = wsC.ChartObjects.Add(20, 420, 450, 350)
        chtObj.Name = "GrafRadarKategorii"
        Set cht = chtObj.Chart
        With cht
            .ChartType = xlRadarFilled
            .HasTitle = True: .ChartTitle.text = "Průměrné hodnocení kategorií (FH)"
            .ChartTitle.Font.Name = "Aptos Narrow": .ChartTitle.Font.size = 12: .ChartTitle.Font.Bold = True
            .HasLegend = False
            .Axes(xlValue).MinimumScale = 0: .Axes(xlValue).MaximumScale = 10
        End With
    Else
        Set cht = chtObj.Chart
    End If
    
    ' ——— OPRAVA CHYBY ———
    On Error Resume Next
    With cht.PlotArea
        .top = 10: .left = 10: .Width = cht.ChartArea.Width - 20: .Height = cht.ChartArea.Height - 20
    End With
    On Error GoTo ErrH
    
    If cht.SeriesCollection.count = 0 Then cht.SeriesCollection.NewSeries
    Do While cht.SeriesCollection.count > 1: cht.SeriesCollection(2).Delete: Loop
    
    With cht.SeriesCollection(1)
        .XValues = catsFmt
        .values = avgs
        .Name = "Průměr"
    End With
    Exit Sub
ErrH:
    MsgBox "Chyba Graf Radar: " & err.Description
End Sub

' 5) Graf 4: Top autoři (Pie Chart)
'————————————————————————————
'————————————————————————————
' 5) Graf 4: Top autoři (Pie Chart)
'————————————————————————————
Private Sub VytvorGrafTopAutori(data() As AnimeRow, wsC As Worksheet)
    On Error GoTo ErrH
    Dim dictAut As Object, dictChart As Object
    Dim i As Long
    Dim authors As Variant, auth As Variant
    Dim isNew As Boolean
    
    Set dictAut = CreateObject("Scripting.Dictionary")
    
    For i = LBound(data) To UBound(data)
        If Len(data(i).Author) > 0 Then
            authors = Split(data(i).Author, ";")
            For Each auth In authors
                auth = Trim(auth)
                If Len(auth) > 0 Then dictAut(auth) = dictAut(auth) + 1
            Next auth
        End If
    Next i
    
    If dictAut.count = 0 Then Exit Sub
    
    Set dictChart = CreateObject("Scripting.Dictionary")
    Dim otherSum As Long: otherSum = 0
    Dim key As Variant
    
    For Each key In dictAut.keys
        If dictAut(key) < 3 Then
            otherSum = otherSum + dictAut(key)
        Else
            dictChart(key) = dictAut(key)
        End If
    Next key
    If otherSum > 0 Then dictChart("Ostatní autoři") = otherSum
    
    If dictChart.count = 0 Then Exit Sub
    
    Dim keysArr As Variant, valsArr As Variant
    Call SortDictionaryByValue(dictChart, keysArr, valsArr, False)
    
    Dim chtObj As ChartObject, cht As Chart
    On Error Resume Next
    Set chtObj = wsC.ChartObjects("GrafTopAutori")
    On Error GoTo ErrH
    
    If chtObj Is Nothing Then
        isNew = True
        Set chtObj = wsC.ChartObjects.Add(500, 420, 500, 350)
        chtObj.Name = "GrafTopAutori"
        Set cht = chtObj.Chart
        With cht
            .ChartType = xlPie
            .HasTitle = True: .ChartTitle.text = "Zastoupení autorů (Počet výskytů >= 3)"
            .ChartTitle.Font.Name = "Aptos Narrow": .ChartTitle.Font.size = 12: .ChartTitle.Font.Bold = True
            .HasLegend = False
        End With
    Else
        isNew = False
        Set cht = chtObj.Chart
        cht.ChartType = xlPie
    End If
    
    If cht.SeriesCollection.count = 0 Then cht.SeriesCollection.NewSeries
    Dim srs As Series: Set srs = cht.SeriesCollection(1)
    
    ' Vložení dat
    Call ApplyDataWithLabelSave(srs, keysArr, valsArr, isNew)
    
    ' Aktualizace popisků pro Pie Chart
    Dim total As Double: total = Application.Sum(valsArr)
    Dim lblTxt As String, pct As Double, ptIdx As Long
    
    On Error Resume Next
    For ptIdx = 1 To srs.Points.count
        pct = (valsArr(ptIdx - 1) / total) * 100
        lblTxt = keysArr(ptIdx - 1) & ": " & valsArr(ptIdx - 1) & " (" & Replace(Format(pct, "0.0"), ".", ",") & "%)"
        
        If srs.Points(ptIdx).DataLabel.text <> lblTxt Then
            srs.Points(ptIdx).DataLabel.text = lblTxt
        End If
        
        If isNew Then
            With srs.Points(ptIdx).DataLabel
                .Position = xlLabelPositionOutsideEnd
                .ShowCategoryName = False: .ShowValue = False: .ShowPercentage = False
                .Font.Name = "Aptos Narrow": .Font.size = 9: .Font.Bold = True
            End With
        End If
    Next ptIdx
    srs.HasLeaderLines = True
    
    Exit Sub
ErrH:
    MsgBox "Chyba Graf Autoři: " & err.Description
End Sub


'————————————————————————————
' 6) Graf 5: Vliv Frisson Feeling (POZOR: Cíl je WS_OPED)
'————————————————————————————
'————————————————————————————
' 6) Graf 5: Vliv Frisson Feeling (Array Optimized)
'————————————————————————————
Private Sub VytvorGrafFrissonVliv(data() As AnimeRow, wsC As Worksheet)
    On Error GoTo ErrH
    Dim sumAno As Double, cntAno As Long
    Dim sumNe As Double, cntNe As Long
    Dim i As Long
    
    For i = LBound(data) To UBound(data)
        If data(i).HasRating Then
            If data(i).Frisson Then
                sumAno = sumAno + data(i).RatingFinal
                cntAno = cntAno + 1
            Else
                sumNe = sumNe + data(i).RatingFinal
                cntNe = cntNe + 1
            End If
        End If
    Next i
    
    If cntAno = 0 And cntNe = 0 Then Exit Sub
    
    Dim catArr As Variant, cntArr() As Long, avgArr() As Variant
    Dim num As Long, idx As Long
    
    ' Dynamická alokace podle toho co existuje
    If cntAno > 0 And cntNe > 0 Then
        catArr = Array("ANO", "NE")
        num = 2
    ElseIf cntAno > 0 Then
        catArr = Array("ANO")
        num = 1
    Else
        catArr = Array("NE")
        num = 1
    End If
    
    ReDim cntArr(1 To num, 1 To 1): ReDim avgArr(1 To num, 1 To 1)
    
    For i = 0 To num - 1
        If catArr(i) = "ANO" Then
            cntArr(i + 1, 1) = cntAno
            avgArr(i + 1, 1) = sumAno / cntAno
        Else
            cntArr(i + 1, 1) = cntNe
            avgArr(i + 1, 1) = sumNe / cntNe
        End If
    Next i
    
    Dim chtObj As ChartObject, cht As Chart, isNew As Boolean
    On Error Resume Next
    Set chtObj = wsC.ChartObjects("GrafFrissonVliv")
    On Error GoTo ErrH
    
    If chtObj Is Nothing Then
        isNew = True
        Set chtObj = wsC.ChartObjects.Add(20, 100, 450, 350)
        chtObj.Name = "GrafFrissonVliv"
        Set cht = chtObj.Chart
        With cht
            .ChartType = xlColumnClustered
            .HasTitle = True: .ChartTitle.text = "Vliv 'Frisson Feeling' (Počet vs. Průměr)"
            .ChartTitle.Font.Name = "Aptos Narrow": .ChartTitle.Font.size = 12: .ChartTitle.Font.Bold = True
            .HasLegend = True: .Legend.Position = xlLegendPositionBottom
        End With
    Else
        isNew = False
        Set cht = chtObj.Chart
    End If
    
    ' Nastavení sérií
    Do While cht.SeriesCollection.count > 2: cht.SeriesCollection(3).Delete: Loop
    Do While cht.SeriesCollection.count < 2: cht.SeriesCollection.NewSeries: Loop
    
    ' 1. Série (Počet)
    With cht.SeriesCollection(1)
        .Name = "Počet": .ChartType = xlColumnClustered: .AxisGroup = xlPrimary
        .XValues = catArr: .values = cntArr: .HasDataLabels = True
        
        Dim tot As Long: tot = 0
        For i = 1 To UBound(cntArr): tot = tot + cntArr(i, 1): Next i
        
        For i = 1 To .Points.count
            Dim pc As Double: pc = (cntArr(i, 1) / tot) * 100
            .Points(i).DataLabel.text = "# " & cntArr(i, 1) & vbLf & "(" & Replace(Format(pc, "0.0"), ".", ",") & " %)"
            If isNew Then .Points(i).DataLabel.Font.size = 9
        Next i
    End With
    
    ' 2. Série (Průměr)

Private Sub VytvorGrafTopSerie(data() As AnimeRow, wsC As Worksheet, dictComm As Object)
    On Error GoTo ErrH
    Dim dictSum As Object, dictCnt As Object
    Dim dictAnimePerSerie As Object, dictOpCnt As Object, dictEdCnt As Object
    Dim i As Long
    Dim ani As String, serie As String, rate As Double
    Dim commentTxt As String
    
    Set dictSum = CreateObject("Scripting.Dictionary")
    Set dictCnt = CreateObject("Scripting.Dictionary")
    Set dictAnimePerSerie = CreateObject("Scripting.Dictionary")
    Set dictOpCnt = CreateObject("Scripting.Dictionary")
    Set dictEdCnt = CreateObject("Scripting.Dictionary")
    
    For i = LBound(data) To UBound(data)
        If data(i).HasRating Then
            rate = data(i).RatingFinal
            ani = data(i).Name
            
            ' Získání série z komentářů
            serie = ani
            If dictComm.Exists(ani) Then
                commentTxt = dictComm(ani)
                If InStr(commentTxt, "Název série:") > 0 Then
                    serie = Trim(Split(commentTxt, "Název série:")(1))
                    If InStr(serie, vbLf) > 0 Then serie = left(serie, InStr(serie, vbLf) - 1)
                End If
            End If
            
            dictSum(serie) = dictSum(serie) + rate
            dictCnt(serie) = dictCnt(serie) + 1
            
            If Not dictAnimePerSerie.Exists(serie) Then Set dictAnimePerSerie(serie) = CreateObject("Scripting.Dictionary")
            dictAnimePerSerie(serie)(ani) = 1

            If data(i).typ = "ED" Then dictEdCnt(serie) = dictEdCnt(serie) + 1

    Dim key As Variant
    For Each key In dictCnt.keys
        If dictCnt(key) > 0 And dictAnimePerSerie(key).count > 1 Then
            dictAvg(key) = dictSum(key) / dictCnt(key)
        End If
    Next key

            Case 2, 3, 4: dStr = dilu & " díly"
            Case Else: dStr = dilu & " dílů"
        End Select
        
        lbl = serie & vbLf & dStr
        If opC > 0 Then lbl = lbl & "; " & opC & " OP"
        If edC > 0 Then
            If opC > 0 Then lbl = lbl & ", " & edC & " ED" Else lbl = lbl & "; " & edC & " ED"
        End If
        
        kTop(i) = lbl
        vTop(i) = vSort(i)
    Next i

            tick.Characters(1, ln - 1).Font.Italic = True
            tick.Characters(ln, Len(tick.text)).Font.Italic = False
        Else
            tick.Font.Italic = True
        End If

' 9) Graf 8: Analýza Jazyka (Vážený průměr)
'————————————————————————————
Private Sub VytvorGrafJazykAnalyza(data() As AnimeRow, wsC As Worksheet)
    On Error GoTo ErrH
    Dim dictCnt As Object, dictSumW As Object, dictSumRateW As Object
    Dim i As Long
    Dim jStr As String, rate As Double
    Dim key As String, tempS As String, p1 As Long, p2 As Long, w As Double
    
    Set dictCnt = CreateObject("Scripting.Dictionary")
    Set dictSumW = CreateObject("Scripting.Dictionary")
    Set dictSumRateW = CreateObject("Scripting.Dictionary")
    dictCnt.CompareMode = vbTextCompare
    
    For i = LBound(data) To UBound(data)
        jStr = data(i).Language
        If Len(jStr) > 0 And data(i).HasRating Then
            rate = data(i).RatingFinal
            
            If InStr(jStr, "(") = 0 Then
                ' Jednoduchý jazyk (3 znaky)
                If Len(jStr) = 3 Then
                    dictCnt(jStr) = dictCnt(jStr) + 1
                    dictSumW(jStr) = dictSumW(jStr) + 1
                    dictSumRateW(jStr) = dictSumRateW(jStr) + rate
                End If
            Else
                ' Složený s %
                If InStr(jStr, "%") > 0 Then
                    tempS = jStr
                    Do While InStr(tempS, "(") > 0
                        p1 = InStr(tempS, "("): p2 = InStr(tempS, ")")
                        If p2 = 0 Then Exit Do
                        key = Trim(Mid(tempS, 1, p1 - 1))
                        If Len(key) = 3 Then
                            w = val(Mid(tempS, p1 + 1, p2 - p1 - 1)) / 100
                            If w > 0 Then
                                dictCnt(key) = dictCnt(key) + 1
                                dictSumW(key) = dictSumW(key) + w
                                dictSumRateW(key) = dictSumRateW(key) + (rate * w)
                            End If
                        End If
                        tempS = Trim(Mid(tempS, p2 + 1))
                    Loop
                End If
            End If
        End If
    Next i
    
    If dictCnt.count = 0 Then Exit Sub
    
    ' Výpočet průměru
    Dim dictAvg As Object: Set dictAvg = CreateObject("Scripting.Dictionary")
    Dim k As Variant
    For Each k In dictSumW.keys
        If dictSumW(k) > 0 Then dictAvg(k) = dictSumRateW(k) / dictSumW(k)
    Next k
    
    Dim kS As Variant, vS As Variant
    Call SortDictionaryByValue(dictCnt, kS, vS, False)
    
    Dim num As Long: num = UBound(kS) + 1
    Dim cA() As String, cCnt() As Long, cAvg() As Variant
    ReDim cA(0 To num - 1): ReDim cCnt(1 To num, 1 To 1): ReDim cAvg(1 To num, 1 To 1)
    
    For i = 0 To num - 1
        k = kS(i)
        cA(i) = k
        cCnt(i + 1, 1) = vS(i)
        cAvg(i + 1, 1) = dictAvg(k)
    Next i
    
    Dim chtObj As ChartObject, cht As Chart, isNew As Boolean
    On Error Resume Next
    Set chtObj = wsC.ChartObjects("GrafJazykAnalyza")
    On Error GoTo ErrH
    
    If chtObj Is Nothing Then
        isNew = True
        Set chtObj = wsC.ChartObjects.Add(20, 790, 450, 350)
        chtObj.Name = "GrafJazykAnalyza"
        Set cht = chtObj.Chart
        With cht
            .ChartType = xlColumnClustered
            .HasTitle = True: .ChartTitle.text = "Analýza jazyka (počet vs. vážený průměr)"
            .ChartTitle.Font.Name = "Aptos Narrow": .ChartTitle.Font.size = 12: .ChartTitle.Font.Bold = True
            .HasLegend = True: .Legend.Position = xlLegendPositionBottom
        End With
    Else
        isNew = False
        Set cht = chtObj.Chart
    End If
    
    Do While cht.SeriesCollection.count > 2: cht.SeriesCollection(3).Delete: Loop
    Do While cht.SeriesCollection.count < 2: cht.SeriesCollection.NewSeries: Loop
    
    ' Série 1: Počet
    With cht.SeriesCollection(1)
        .Name = "Počet": .ChartType = xlColumnClustered: .AxisGroup = xlPrimary
        .XValues = cA: .values = cCnt: .HasDataLabels = True
        
        Dim tot As Long: tot = 0
        For i = 1 To UBound(cCnt): tot = tot + cCnt(i, 1): Next i
        
        For i = 1 To .Points.count
            Dim pc As Double: pc = (cCnt(i, 1) / tot) * 100
            .Points(i).DataLabel.text = "# " & cCnt(i, 1) & vbLf & "(" & Replace(Format(pc, "0.0"), ".", ",") & " %)"
            If isNew Then .Points(i).DataLabel.Font.size = 9
        Next i
    End With
    
    ' Série 2: Průměr
    With cht.SeriesCollection(2)
        .Name = "Vážený průměr": .ChartType = xlLineMarkers: .AxisGroup = xlSecondary
        .XValues = cA: .values = cAvg: .HasDataLabels = True
        If isNew Then
            .DataLabels.Position = xlLabelPositionAbove
            .DataLabels.Font.Bold = True
            .Format.line.ForeColor.RGB = RGB(237, 125, 49)
        End If
        .DataLabels.NumberFormat = "0,00"
    End With
    
    cht.Axes(xlValue, xlSecondary).MaximumScale = 10
    Exit Sub
ErrH:
    MsgBox "Chyba Jazyk: " & err.Description
End Sub

        Set cht = chtObj.Chart
        cht.ChartType = xlBubble
        With cht
            .HasTitle = True: .ChartTitle.text = "Kvalita (tech vs. hloubka)"
            .ChartTitle.Font.Name = "Aptos Narrow": .ChartTitle.Font.size = 12: .ChartTitle.Font.Bold = True
        End With
    Else
        isNew = False
        Set cht = chtObj.Chart
        With cht.Axes(xlCategory): xMin = .MinimumScale: xMax = .MaximumScale: End With
        With cht.Axes(xlValue): yMin = .MinimumScale: yMax = .MaximumScale: End With
    End If
    
    ' Smazání sérií (rebuild)
    Do While cht.SeriesCollection.count > 0: cht.SeriesCollection(1).Delete: Loop
    
    Dim k As Long, ser As Series
    For k = 1 To valid
        Set ser = cht.SeriesCollection.NewSeries
        With ser
            .Name = names(k)
            .XValues = Array(xArr(k))
            .values = Array(yArr(k))
            .BubbleSizes = Array(szArr(k))
            
            If isFri(k) Then
                .Format.Fill.ForeColor.RGB = RGB(23, 169, 192)
                .Format.line.ForeColor.RGB = RGB(10, 80, 90)
            Else
                .Format.Fill.ForeColor.RGB = RGB(191, 135, 38)
                .Format.line.ForeColor.RGB = RGB(100, 70, 20)
            End If
            .Format.Fill.Transparency = 0
            .Format.line.weight = 0.5
        End With
    Next k
    
    cht.HasLegend = False
    With cht.Axes(xlValue)
        .HasTitle = True: .AxisTitle.text = "Hloubka (emoce + text)"
        If Not isNew Then .MinimumScale = yMin: .MaximumScale = yMax
    End With
    With cht.Axes(xlCategory)
        .HasTitle = True: .AxisTitle.text = "Technická kvalita (video + hlas)"
        If Not isNew Then .MinimumScale = xMin: .MaximumScale = xMax
    End With
    Exit Sub
ErrH:
    MsgBox "Chyba Bubble: " & err.Description
End Sub

Private Sub ApplyDataWithLabelSave(srs As Series, xVals As Variant, yVals As Variant, isNewChart As Boolean)
    On Error Resume Next
    
    ' ——— PHASE 4: INTELLIGENT UPDATE CHECK ———
    If Not isNewChart Then
        Dim curX As Variant, curY As Variant
        curX = srs.XValues
        curY = srs.values
        If ArraysMatch(curX, xVals) And ArraysMatch(curY, yVals) Then Exit Sub
    End If
    ' ——— END PHASE 4 ———

    ' OPTIMALIZACE FÁZE 5: ODSTRANĚNÍ ZBYTEČNÉHO UKLÁDÁNÍ/OBNOVY FORMÁTU
    ' Dříve zde byl cyklus, který ukládal formát každého bodu. To je pomalé a u dynamických dat (Top X)
    ' často chybné, protože se mění pořadí. Excel si formát drží, pokud neměníme strukturu.
    ' Pokud je graf nový, nastavíme výchozí formát.

    ' 1. Data Update
    srs.XValues = xVals
    srs.values = yVals
    srs.HasDataLabels = True
    
    ' 2. Basic Formatting (Only if needed/new)
    If isNewChart Then
        Dim dl As DataLabel, i As Long
        For i = 1 To srs.Points.count
            Set dl = srs.Points(i).DataLabel
            With dl
                .Position = xlLabelPositionOutsideEnd
                .Font.size = 9: .Font.Bold = False: .Font.Color = RGB(0, 0, 0)
            End With
        Next i
    End If
End Sub

' ——— HELPER: POROVNÁNÍ POLÍ (PHASE 4) ———
Private Function ArraysMatch(arr1 As Variant, arr2 As Variant) As Boolean
    ' Rychlé porovnání dvou polí. Vrací True, pokud jsou shodná.
    On Error GoTo NoMatch
    
    If IsEmpty(arr1) Or IsEmpty(arr2) Then GoTo NoMatch
    If Not IsArray(arr1) Or Not IsArray(arr2) Then GoTo CHECK_SCALAR
    
    Dim lb1 As Long, ub1 As Long, lb2 As Long, ub2 As Long
    lb1 = LBound(arr1): ub1 = UBound(arr1)
    lb2 = LBound(arr2): ub2 = UBound(arr2)
    
    ' Kontrola dimenzí (zjednodušeně 1D)
    If (ub1 - lb1) <> (ub2 - lb2) Then GoTo NoMatch
    
    Dim i As Long, v1 As Variant, v2 As Variant
    For i = 0 To (ub1 - lb1)
        v1 = arr1(lb1 + i)
        v2 = arr2(lb2 + i)
        
        If VarType(v1) <> VarType(v2) Then
             ' Tolerance pro Double/Integer rozdíly? Zkusíme přímé porovnání hodnot
             If IsNumeric(v1) And IsNumeric(v2) Then
                If Abs(CDbl(v1) - CDbl(v2)) > 0.000001 Then GoTo NoMatch
             Else
                If CStr(v1) <> CStr(v2) Then GoTo NoMatch
             End If
        Else
            If v1 <> v2 Then GoTo NoMatch
        End If
    Next i
    
    ArraysMatch = True
    Exit Function

CHECK_SCALAR:
    If arr1 = arr2 Then ArraysMatch = True: Exit Function
    
NoMatch:
    ArraysMatch = False
End Function

Private Sub SortDictionaryByValue(ByVal dict As Object, ByRef keysArr As Variant, ByRef valsArr As Variant, ByVal ascending As Boolean)
    If dict.count = 0 Then Exit Sub
    keysArr = dict.keys
    valsArr = dict.items
    
    Call QuickSortLists(keysArr, valsArr, LBound(valsArr), UBound(valsArr), ascending)
End Sub

Private Sub QuickSortLists(ByRef kArr As Variant, ByRef vArr As Variant, ByVal inLow As Long, ByVal inHi As Long, ByVal ascending As Boolean)
    Dim pivot   As Variant
    Dim tmpSwap As Variant
    Dim tmpLow  As Long
    Dim tmpHi   As Long
    
    tmpLow = inLow
    tmpHi = inHi
    
    If inLow >= inHi Then Exit Sub
    
    pivot = vArr((inLow + inHi) \ 2)
    
    While (tmpLow <= tmpHi)
        If ascending Then
            While (vArr(tmpLow) < pivot And tmpLow < inHi)
                tmpLow = tmpLow + 1
            Wend
            While (pivot < vArr(tmpHi) And tmpHi > inLow)
                tmpHi = tmpHi - 1
            Wend
        Else
            While (vArr(tmpLow) > pivot And tmpLow < inHi)
                tmpLow = tmpLow + 1
            Wend
            While (pivot > vArr(tmpHi) And tmpHi > inLow)
                tmpHi = tmpHi - 1
            Wend
        End If
        
        If (tmpLow <= tmpHi) Then
            tmpSwap = vArr(tmpLow)
            vArr(tmpLow) = vArr(tmpHi)
            vArr(tmpHi) = tmpSwap
            
            tmpSwap = kArr(tmpLow)
            kArr(tmpLow) = kArr(tmpHi)
            kArr(tmpHi) = tmpSwap
            
            tmpLow = tmpLow + 1
            tmpHi = tmpHi - 1
        End If
    Wend
    
    If (inLow < tmpHi) Then QuickSortLists kArr, vArr, inLow, tmpHi, ascending
    If (tmpLow < inHi) Then QuickSortLists kArr, vArr, tmpLow, inHi, ascending
End Sub
