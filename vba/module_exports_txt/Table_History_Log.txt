Private Sub VytvorNeboAktualizujTlacitko(ws As Worksheet, ByVal btnName As String, rng As Range, text As String, macroName As String, colorValue As Long)
    Dim btn As Button
    
    ' Pokusí se najít existující tlačítko podle jeho programového jména
    On Error Resume Next
    Set btn = ws.Buttons(btnName)
    On Error GoTo 0
    
    If btn Is Nothing Then
        ' Tlačítko neexistuje, vytvoříme ho na výchozí pozici
        Set btn = ws.Buttons.Add(rng.left, rng.top, rng.Width, rng.Height)
        btn.Name = btnName ' Přiřadíme unikátní jméno pro budoucí identifikaci
    End If
    
    ' Aplikujeme vlastnosti a formátování (pro nové i pro existující tlačítko)
    With btn
        .Caption = text
        .OnAction = macroName
        
        ' Nastavení písma
        With .Font
            .Name = "Aptos Narrow"
            .size = 11
            .Bold = True
        End With
        
        ' Pokročilé formátování přes objekt ShapeRange
        With .ShapeRange
            ' Nastavení barvy výplně
            With .Fill
                .Visible = msoTrue
                .ForeColor.RGB = colorValue
                .Solid
            End With
            
            ' Nastavení ohraničení
            With .line
                .Visible = msoTrue
                .ForeColor.RGB = RGB(0, 0, 0) ' Černá barva
                .weight = 1.5 ' Tloušťka 1.5 bodu
            End With
        End With
    End With
End Sub

Public Sub SearchAndFilterLog()
    Dim searchTerm As String
    searchTerm = InputBox("Zadejte název Anime nebo série, kterou chcete vyhledat:", "Vyhledávání")
    
    ' Pokud uživatel klikne na Storno nebo nechá pole prázdné

End Sub

Public Sub SpustitVytvoreniHistoryLog()
    Dim wsAnime As Worksheet, wsLog As Worksheet
    Dim currentHash As String
    Dim oldCalc As Long
    Dim dailyTotals As Scripting.Dictionary
    
    On Error GoTo ErrHandler
    
    ' --- MAXIMÁLNÍ VYPNUTÍ BRZD EXCELU (OPT-10) ---
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    oldCalc = Application.Calculation
    Application.Calculation = xlCalculationManual
    
    Set wsAnime = ThisWorkbook.Worksheets("ANIME LIST")
    Set wsLog = ThisWorkbook.Worksheets("HISTORY LOG")
    
    If g_SortBy = "" And g_SearchFilter = "" Then g_SortBy = "Date"
    
    ' 1. Výpočet Hash
    currentHash = CreateStateHash(wsAnime) & g_SearchFilter

    ' 2. Příprava Dat
    If Not (currentHash = g_CachedHash And IsArrayInitialized(g_CachedData)) Then
        g_CachedData = PripravitDataProLog(wsAnime)
        g_CachedHash = currentHash
    End If

    ' 3. Zápis Logu
    Set g_DateRowMap = New Scripting.Dictionary
    ZapsatLog wsLog, g_CachedData
    
    ' 4. Společná agregace dat pro Graf a Heatmapu (OPT-6)
    Set dailyTotals = ZiskatDataProHeatmapu_Core(g_CachedData)
    
    ' 5. Generování Grafu (OPT-6)
    Call GenerovatGrafZhlédnutýchEpizod_FinalniVerze_AtytoJeUzOstre(dailyTotals, g_CachedData)
    
    ' 6. Vytvoření Heatmapy (OPT-6)
    Call VytvoritHeatmapu(dailyTotals, "HISTORY LOG", "E38", g_DateRowMap)
    
    ' <<< NOVÁ ČÁST PRO ÚPRAVU NADPISU V A1 >>>
    If g_SearchFilter <> "" Then
        wsLog.Range("A1").Value = "Historie sledování '" & g_SearchFilter & "'"
    Else
        wsLog.Range("A1").Value = "Historie sledování od počátku věků (ANIME)"
    End If
    ' <<< KONEC NOVÉ ČÁSTI >>>

    ' --- OBNOVENÍ NASTAVENÍ EXCELU (OPT-10) ---
CleanExit:
    Application.ScreenUpdating = True
    Application.EnableEvents = True
    Application.Calculation = oldCalc
    
    Exit Sub

ErrHandler:
    MsgBox "Nastala kritická chyba při vytváření History Logu: " & err.Description, vbCritical
    Resume CleanExit
End Sub

Private Function CreateStateHash(ws As Worksheet) As String
    ' OPTIMALIZOVANÁ VERZE 2.0: Bleskový hash bez iterace objektů
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.count, "C").End(xlUp).row
    
    If lastRow < 2 Then
        CreateStateHash = "empty"
        Exit Function
    End If
    
    Dim hashParts(1 To 5) As String
    
    ' 1. Počet řádků
    hashParts(1) = CStr(lastRow)
    
    ' 2. Součet sloupce O (celkový čas)
    hashParts(2) = CStr(Application.WorksheetFunction.Sum(ws.Range("O2:O" & lastRow)))
    
    ' 3. Součet sloupce I (počet epizod)
    hashParts(3) = CStr(Application.WorksheetFunction.Sum(ws.Range("I2:I" & lastRow)))
    
    ' 4. Počet komentářů (OPTIMALIZOVÁNO - SpecialCells)
    ' Místo For Each, které trvá dlouho, se zeptáme Excelu přímo na počet buněk s komentáři.
    ' SpecialCells vyhodí chybu, pokud nic nenajde, proto On Error Resume Next.
    Dim countCmt As Long
    countCmt = 0
    
    On Error Resume Next
    Dim rngComments As Range
    ' Zajímá nás jen sloupec I a N (epizody a rewatche)
    Set rngComments = ws.Range("I2:I" & lastRow & ",N2:N" & lastRow).SpecialCells(xlCellTypeComments)
    If Not rngComments Is Nothing Then
        countCmt = rngComments.count ' Okamžitý počet buněk s komentářem
    End If
    On Error GoTo 0
    
    hashParts(4) = CStr(countCmt)
    
    ' 5. Poslední název anime (pro detekci smazání posledního řádku)
    hashParts(5) = CStr(ws.Cells(lastRow, "C").Value)
    
    CreateStateHash = Join(hashParts, "|")
End Function


Private Sub ProcessWatchPeriod(ByRef dictLog As Scripting.Dictionary, ByVal animeName As String, ByVal startDate As Date, ByVal endDate As Date, ByVal totalEpisodes As Long, ByVal timeForPeriod As Double, ByVal sourceRow As Long, Optional ByVal epOverrides As Scripting.Dictionary = Nothing, Optional ByVal defaultDuration As Double = 0)
    If totalEpisodes <= 0 Or timeForPeriod <= 0 Or startDate > endDate Then Exit Sub
    
    Dim numDays As Long
    numDays = (endDate - startDate) + 1 ' OPTIMALIZACE: Nativní matematika místo DateDiff
    
    Dim epsPerDay As Long: epsPerDay = totalEpisodes \ numDays
    Dim remainderEps As Long: remainderEps = totalEpisodes Mod numDays
    
    Dim epCounter As Long: epCounter = 1
    Dim k As Long
    Dim currentDate As Date
    Dim epsForThisDay As Long
    Dim logKey As String
    
    For k = 0 To numDays - 1
        currentDate = startDate + k ' OPTIMALIZACE: Nativní matematika místo DateAdd
        epsForThisDay = epsPerDay
        If remainderEps > 0 Then
            epsForThisDay = epsForThisDay + 1
            remainderEps = remainderEps - 1
        End If
        
        If epsForThisDay > 0 Then
            ' OPTIMALIZACE: Rychlejší klíč (CLng)
            logKey = CLng(currentDate) & "|" & animeName
            
            If Not dictLog.Exists(logKey) Then
                ' Flat Array layout (OPT-3): 0=Date, 1=Name, 2=Minutes, 3=EpisodesDict, 4=SourceRow
                Dim logEntry(0 To 4) As Variant
                logEntry(0) = currentDate
                logEntry(1) = animeName
                logEntry(2) = 0#
                Set logEntry(3) = New Scripting.Dictionary
                logEntry(4) = sourceRow
                dictLog.Add logKey, logEntry
            End If
            
            ' VBA re-assignment rule for arrays in dictionary (retrieve, modify, re-assign)
            Dim tempEntry As Variant
            tempEntry = dictLog(logKey)
            
            If totalEpisodes > 0 Then
                tempEntry(2) = tempEntry(2) + CalcEpMinutes(epCounter, epCounter + epsForThisDay - 1, epOverrides, defaultDuration, totalEpisodes, timeForPeriod)
            End If

            Dim j As Long
            For j = 1 To epsForThisDay
                If epCounter <= totalEpisodes Then
                    If Not tempEntry(3).Exists(epCounter) Then
                        tempEntry(3).Add epCounter, Nothing
                    End If
                    epCounter = epCounter + 1
                End If
            Next j
            
            dictLog(logKey) = tempEntry
        End If
    Next k
End Sub

' <<< NOVÁ POMOCNÁ FUNKCE PRO ZÍSKÁNÍ SHODUJÍCÍCH SE ŘÁDKŮ (OPTIMALIZOVANÁ) >>>
Private Function GetMatchingAnimeRows(ws As Worksheet, searchTerm As String) As Collection
    Set GetMatchingAnimeRows = New Collection
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.count, "C").End(xlUp).row
    If lastRow < 2 Then Exit Function

    ' 1. Načtení dat do pole (Name)
    Dim dataName As Variant
    dataName = ws.Range("C2:C" & lastRow).Value

    ' 2. Načtení komentářů do paměti (BLESKOVÉ)
    Dim cmtData() As String
    ReDim cmtData(1 To lastRow)
    Dim rngCmt As Range, cell As Range
    
    On Error Resume Next
    Set rngCmt = ws.Range("C2:C" & lastRow).SpecialCells(xlCellTypeComments)
    On Error GoTo 0
    
    If Not rngCmt Is Nothing Then
        For Each cell In rngCmt
            cmtData(cell.row) = cell.Comment.text
        Next cell
    End If

    Dim i As Long
    Dim animeName As String
    Dim lSearch As String
    Dim seriesNamePattern As String
    
    lSearch = LCase(searchTerm)
    seriesNamePattern = "Název série: " & searchTerm
    
    ' 3. Rychlá smyčka v paměti
    For i = 2 To lastRow
        ' A. Kontrola jména
        animeName = CStr(dataName(i - 1, 1))
        If InStr(1, LCase(animeName), lSearch) > 0 Then
            GetMatchingAnimeRows.Add i
        Else
            ' B. Kontrola komentáře (pokud existuje)
            If Len(cmtData(i)) > 0 Then
                If InStr(1, cmtData(i), seriesNamePattern, vbTextCompare) > 0 Then
                    GetMatchingAnimeRows.Add i
                End If
            End If
        End If
    Next i
End Function

Private Function PripravitDataProLog(wsAnime As Worksheet) As Variant
    ' --- Hlavní deklarace (OPT-4) ---
    Dim dictLog As Scripting.Dictionary, lastRow As Long, i As Long
    Dim rowsToProcess As Collection, loopVar As Variant
    Dim dataArray As Variant
    Dim regEp As Object, regN As Object
    Dim useCommentI As Boolean
    
    ' <<< NOVÉ PROMĚNNÉ PRO CACHE KOMENTÁŘŮ >>>
    Dim CommentsI() As String
    Dim CommentsN() As String
    Dim CommentsJ() As String
    Dim rngCmt As Range
    Dim cell As Range

    ' --- Deklarace pro hlavní smyčku ---
    Dim animeName As String, epDuration As Double, totalEpisodes As Long
    Dim watchTime As Double, rewatchCount As Long, timePerSession As Double

    ' --- Deklarace pro vnitřní smyčky a parsování ---
    Dim linesEp As Variant, lineEp As Variant, matchesEp As Object
    Dim firstEp As Long, lastEp As Long, epDate As Date, j As Long, k As Long, logKey As String
    Dim linesN As Variant, lineN As Variant, matchesN As Object, rwStart As Date, rwEnd As Date
    Dim rewatchAnimeName As String

    ' --- Deklarace specificky pro novou logiku EPs: ---
    Dim parts() As String, datePart As String, epCountsPart As String, epCounts() As String
    Dim numDays As Long, currentDate As Date, epsForDay As Long
    Dim timePerEp As Double, epCounterForRewatch As Long
    
    ' --- Tělo funkce ---
    Set dictLog = New Scripting.Dictionary
    lastRow = wsAnime.Cells(wsAnime.Rows.count, "L").End(xlUp).row
    If lastRow < 2 Then
        PripravitDataProLog = Array()
        Exit Function
    End If
    
    If g_SearchFilter <> "" Then
        Set rowsToProcess = GetMatchingAnimeRows(wsAnime, g_SearchFilter)
        If rowsToProcess.count = 0 Then
             PripravitDataProLog = Array()
             Exit Function
        End If
    Else
        Set rowsToProcess = New Collection
        For i = 2 To lastRow
            rowsToProcess.Add i
        Next i
    End If
    
    dataArray = wsAnime.Range("A1:O" & lastRow).Value

    ' <<< NOVÝ BLOK: CACHING KOMENTÁŘŮ >>>
    ReDim CommentsI(1 To lastRow)
    ReDim CommentsN(1 To lastRow)
    ReDim CommentsJ(1 To lastRow)
    
    On Error Resume Next
    Set rngCmt = wsAnime.Range("I2:I" & lastRow).SpecialCells(xlCellTypeComments)
    If Not rngCmt Is Nothing Then
        For Each cell In rngCmt
            CommentsI(cell.row) = cell.Comment.text
        Next cell
    End If
    
    Set rngCmt = Nothing
    Set rngCmt = wsAnime.Range("N2:N" & lastRow).SpecialCells(xlCellTypeComments)
    If Not rngCmt Is Nothing Then
        For Each cell In rngCmt
            CommentsN(cell.row) = cell.Comment.text
        Next cell
    End If
    
    Set rngCmt = Nothing
    Set rngCmt = wsAnime.Range("J2:J" & lastRow).SpecialCells(xlCellTypeComments)
    If Not rngCmt Is Nothing Then
        For Each cell In rngCmt
            CommentsJ(cell.row) = cell.Comment.text
        Next cell
    End If
    On Error GoTo 0
    ' <<< KONEC NOVÉHO BLOKU >>>

    Set regEp = CreateObject("VBScript.RegExp")
    regEp.Pattern = "EP:\s*(\d+)\s*(?:-\s*(\d+))?;\s*(\d{1,2}\.\d{1,2}\.\d{4})"
    regEp.Global = True
    Set regN = CreateObject("VBScript.RegExp")
    regN.Pattern = "\d{1,2}\.\d{1,2}\.\d{4}"
    regN.Global = True
    
    For Each loopVar In rowsToProcess
        i = CLng(loopVar)
        useCommentI = False
        
        animeName = Nz(dataArray(i, 3), "[Neznámé Anime]")
        If IsNumeric(Nz(dataArray(i, 10))) Then epDuration = CDbl(Nz(dataArray(i, 10))) Else epDuration = 0
        If IsNumeric(Nz(dataArray(i, 9))) Then totalEpisodes = CLng(Nz(dataArray(i, 9))) Else totalEpisodes = 0
        
        Dim epOverrides As Scripting.Dictionary: Set epOverrides = Nothing
        Dim timeColO As Double
        If IsNumeric(Nz(dataArray(i, 15))) Then timeColO = CDbl(Nz(dataArray(i, 15))) Else timeColO = totalEpisodes * epDuration
        If Len(CommentsJ(i)) > 0 Then Set epOverrides = ParseEpDurationsFromComment(CommentsJ(i))

        ' NOVÉ (VLOŽIT):
        If Len(CommentsI(i)) > 0 And epDuration > 0 Then
            linesEp = Split(CommentsI(i), vbLf)
        
            For Each lineEp In linesEp
                Set matchesEp = regEp.Execute(lineEp)
                If matchesEp.count > 0 Then
                    firstEp = CLng(Nz(matchesEp(0).SubMatches(0)))
                    If Nz(matchesEp(0).SubMatches(1)) <> "" Then lastEp = CLng(Nz(matchesEp(0).SubMatches(1))) Else lastEp = firstEp
                    epDate = CDate(Nz(matchesEp(0).SubMatches(2)))
                    ' OPTIMALIZACE: Rychlejší klíč
                    logKey = CLng(epDate) & "|" & animeName
                    If Not dictLog.Exists(logKey) Then
                        ' Flat Array layout (OPT-3): 0=Date, 1=Name, 2=Minutes, 3=EpisodesDict, 4=SourceRow
                        Dim logEntry(0 To 4) As Variant
                        logEntry(0) = epDate
                        logEntry(1) = animeName
                        logEntry(2) = 0#
                        Set logEntry(3) = New Scripting.Dictionary
                        logEntry(4) = i
                        dictLog.Add logKey, logEntry
                    End If
                    
                    ' VBA re-assignment rule for arrays in dictionary (retrieve, modify, re-assign)
                    Dim tempEntry As Variant
                    tempEntry = dictLog(logKey)
                    tempEntry(2) = tempEntry(2) + CalcEpMinutes(firstEp, lastEp, epOverrides, epDuration, totalEpisodes, totalEpisodes * epDuration)
                    For j = firstEp To lastEp
                        If Not tempEntry(3).Exists(j) Then tempEntry(3).Add j, Nothing
                    Next j
                    dictLog(logKey) = tempEntry
                    
                    useCommentI = True
                End If
            Next lineEp
        End If
        
        If IsDate(Nz(dataArray(i, 12))) And IsNumeric(Nz(dataArray(i, 15))) Then
            watchTime = CDbl(Nz(dataArray(i, 15)))
            rewatchCount = val(Nz(dataArray(i, 14)))
            
            If watchTime > 0 Then
                If rewatchCount > 0 Then timePerSession = watchTime / (rewatchCount + 1) Else timePerSession = watchTime
                
                ' NOVÉ (VLOŽIT):
                If Len(CommentsN(i)) > 0 Then
                    linesN = Split(CommentsN(i), vbLf)
                
                    For Each lineN In linesN
                        rewatchAnimeName = animeName
                        If InStr(lineN, ".") > 0 Then
                            Dim rewatchNumberPart As String
                            rewatchNumberPart = Trim(Split(lineN, ".")(0))
                            If IsNumeric(rewatchNumberPart) And CLng(rewatchNumberPart) > 0 Then
                                rewatchAnimeName = animeName & "|RW|" & rewatchNumberPart
                            End If
                        End If

                        '--- NOVÁ LOGIKA: Prioritní zpracování formátu "; EPs:" ---
                        If InStr(1, lineN, "; EPs:", vbTextCompare) > 0 Then
                            parts = Split(lineN, "; EPs:")
                            datePart = Trim(Mid(parts(0), InStr(parts(0), ".") + 1))
                            epCountsPart = Trim(parts(1))
                            epCounts = Split(epCountsPart, ",")
                            
                            If InStr(datePart, "-") > 0 Then
                                rwStart = CDate(Trim(Split(datePart, "-")(0)))
                                rwEnd = CDate(Trim(Split(datePart, "-")(1)))
                            Else
                                rwStart = CDate(datePart)
                                rwEnd = rwStart
                            End If
                            
                            numDays = (rwEnd - rwStart) + 1 ' OPTIMALIZACE
                            
                            If UBound(epCounts) + 1 = numDays Then
                                If totalEpisodes > 0 Then timePerEp = timePerSession / totalEpisodes Else timePerEp = 0
                                
                                epCounterForRewatch = 1
                                
                                For k = 0 To numDays - 1
                                    currentDate = rwStart + k ' OPTIMALIZACE
                                    epsForDay = CLng(Trim(epCounts(k)))
                                    
                                    If epsForDay > 0 Then
                                        ' OPTIMALIZACE: Rychlejší klíč
                                        logKey = CLng(currentDate) & "|" & rewatchAnimeName
                                        If Not dictLog.Exists(logKey) Then
                                            ' Flat Array layout (OPT-3)
                                            Dim logEntryRW(0 To 4) As Variant
                                            logEntryRW(0) = currentDate
                                            logEntryRW(1) = rewatchAnimeName
                                            logEntryRW(2) = 0#
                                            Set logEntryRW(3) = New Scripting.Dictionary
                                            logEntryRW(4) = i
                                            dictLog.Add logKey, logEntryRW
                                        End If
                                        
                                        Dim tempEntryRW As Variant
                                        tempEntryRW = dictLog(logKey)
                                        tempEntryRW(2) = tempEntryRW(2) + CalcEpMinutes(epCounterForRewatch, epCounterForRewatch + epsForDay - 1, epOverrides, epDuration, totalEpisodes, timePerSession)
                                        
                                        For j = 1 To epsForDay
                                            If Not tempEntryRW(3).Exists(epCounterForRewatch) Then
                                                tempEntryRW(3).Add epCounterForRewatch, Nothing
                                            End If
                                            epCounterForRewatch = epCounterForRewatch + 1
                                        Next j
                                        dictLog(logKey) = tempEntryRW
                                    End If
                                Next k
                            End If
                        '--- PŮVODNÍ LOGIKA pro ostatní formáty ---
                        Else
                            Set matchesN = regN.Execute(lineN)
                            If matchesN.count > 0 Then
                                rwStart = CDate(matchesN(0).Value)
                                If matchesN.count > 1 Then rwEnd = CDate(matchesN(1).Value) Else rwEnd = rwStart
                                ProcessWatchPeriod dictLog, rewatchAnimeName, rwStart, rwEnd, totalEpisodes, timePerSession, i, epOverrides, epDuration
                            End If
                        End If
                    Next lineN
                End If
                
                If Not useCommentI Then
                    Dim originalStartDate As Date, originalEndDate As Date
                    originalStartDate = CDate(Nz(dataArray(i, 12)))
                    If IsDate(Nz(dataArray(i, 13))) Then originalEndDate = CDate(Nz(dataArray(i, 13))) Else originalEndDate = originalStartDate
                    ProcessWatchPeriod dictLog, animeName, originalStartDate, originalEndDate, totalEpisodes, timePerSession, i, epOverrides, epDuration
                End If
            End If
        End If
    Next loopVar
    
    Dim outputData() As Variant
    If dictLog.count = 0 Then
        PripravitDataProLog = Array()
        Exit Function
    End If
    ' <<< ZMĚNA STRUKTURY: Nyní 6 sloupců (6. je raw Episode Count) >>>
    ReDim outputData(1 To dictLog.count, 1 To 6)
    Dim keys As Variant: keys = dictLog.keys
    For i = 0 To dictLog.count - 1
        Dim currentEntry As Variant
        currentEntry = dictLog(keys(i))
        outputData(i + 1, 1) = currentEntry(0) ' Date
        outputData(i + 1, 2) = currentEntry(1) ' Name
        outputData(i + 1, 3) = FormatEpisodes(currentEntry(3)) ' Episodes dict
        outputData(i + 1, 4) = currentEntry(2) ' Minutes
        outputData(i + 1, 5) = currentEntry(4) ' SourceRow
        outputData(i + 1, 6) = currentEntry(3).count ' Předpočítaný počet epizod (Long)
    Next i
    PripravitDataProLog = outputData
End Function

' === VYKRESLENÍ TABULKY A FORMÁTOVÁNÍ (OPTIMALIZOVANÁ VERZE - BATCH)
' =========================================================================================
Private Sub ZapsatLog(wsLog As Worksheet, ByRef outputData() As Variant)
    
    ' 1. Příprava filtru
    If wsLog.AutoFilterMode Then wsLog.AutoFilterMode = False
    
    ' 2. Vyčištění starých dat (Hromadně)
    Dim lastRow As Long
    On Error Resume Next
    lastRow = wsLog.Cells.Find(What:="*", SearchOrder:=xlByRows, SearchDirection:=xlPrevious).row
    If err.Number <> 0 Then lastRow = 3
    On Error GoTo 0
    If lastRow < 3 Then lastRow = 3

With wsLog.Range("A3:D" & lastRow) ' Změna na E pro jistotu, i když E je skryté
        .UnMerge ' <<< TOTO JE TA OPRAVA: Rozdělí případné sloučené buňky
        .ClearContents
        .ClearComments
        .Interior.ColorIndex = xlNone
        .Borders.LineStyle = xlNone
        .Font.Bold = False
        .Font.Italic = False
        .RowHeight = 15
        .WrapText = False
        .VerticalAlignment = xlBottom
        .Hyperlinks.Delete
    End With
    
    ' 3. Kontrola dat
    If Not IsArrayInitialized(outputData) Then
        wsLog.Range("A3").Value = "Pro zadaný filtr nebyly nalezeny žádné záznamy."
        wsLog.Range("A3").Font.Italic = True
        Exit Sub
    End If
    
    ' 4. Příprava datových struktur
    Dim dayDetails As Scripting.Dictionary: Set dayDetails = New Scripting.Dictionary
    Dim i As Long
    Dim minDate As Date, maxDate As Date
    minDate = CDate(outputData(1, 1))
    maxDate = CDate(outputData(1, 1))

    ' Agregace dat (stejné jako dříve)
    For i = 1 To UBound(outputData, 1)
        Dim entryDate As Date: entryDate = CDate(outputData(i, 1))
        If Not dayDetails.Exists(entryDate) Then
            dayDetails.Add entryDate, New Collection
        End If
        ' Ukládáme: Name, Episodes, Minutes, SourceRow, [EpisodeCount]
        Dim epCountRaw As Long
        If UBound(outputData, 2) >= 6 Then
            epCountRaw = CLng(outputData(i, 6))
        Else
            epCountRaw = ParseEpisodeCount(outputData(i, 3))
        End If
        dayDetails(entryDate).Add Array(outputData(i, 2), outputData(i, 3), outputData(i, 4), outputData(i, 5), epCountRaw)
        
        If entryDate < minDate Then minDate = entryDate
        If entryDate > maxDate Then maxDate = entryDate
    Next i
    
    Dim yesterday As Date: yesterday = Date - 1
    If g_SearchFilter = "" Then
        If maxDate < yesterday Then maxDate = yesterday
    End If
    
    If dayDetails.count = 0 Then
        Exit Sub
    End If

    ' 5. Třídění dnů (stejné jako dříve)
    Dim daysArray() As Variant
    Dim sortedDates As Collection: Set sortedDates = New Collection ' Pro uložení pořadí
    
    If g_SortBy <> "Date" Then
        ' ... logika pro řazení dle statistik ...
        ReDim daysArray(1 To dayDetails.count, 1 To 6)
        Dim dayKey As Variant, idx As Long: idx = 1
        Dim detailsCollection As Collection, detail As Variant
        Dim daySumMinutes As Double, daySumEpisodes As Long
        
        For Each dayKey In dayDetails.keys
            Set detailsCollection = dayDetails(dayKey)
            daySumMinutes = 0: daySumEpisodes = 0
            For Each detail In detailsCollection
                daySumEpisodes = daySumEpisodes + CLng(detail(4)) ' OPTIMALIZACE: Použití předpočítaného
                daySumMinutes = daySumMinutes + detail(2)
            Next detail
            daysArray(idx, 1) = dayKey
            daysArray(idx, 2) = daySumEpisodes
            daysArray(idx, 3) = daySumMinutes
            daysArray(idx, 4) = detailsCollection(1)(0)
            Set daysArray(idx, 5) = detailsCollection
            daysArray(idx, 6) = detailsCollection.count
            idx = idx + 1
        Next dayKey

' NOVÉ:
Select Case g_SortBy
    Case "Episodes": QuickSort2DArray daysArray, 2, 1, UBound(daysArray, 1), False
    Case "Time": QuickSort2DArray daysArray, 3, 1, UBound(daysArray, 1), False
    Case "AnimeCount": QuickSort2DArray daysArray, 6, 1, UBound(daysArray, 1), False
End Select

        
        ' Převedeme pole zpět na kolekci pro jednotný průchod
        Dim sortedItems As Collection: Set sortedItems = New Collection
        For i = 1 To UBound(daysArray, 1)
            sortedItems.Add daysArray(i, 5) ' Collection detailů
            ' Uložíme si metadata dne do "hlavičky" v kolekci? Ne, uděláme to jednodušeji.
            ' Pro tento účel si uděláme pomocnou strukturu, abychom mohli iterovat lineárně.
        Next i
    End If
    
    ' ==================================================================
    ' === HLAVNÍ FÁZE: PŘÍPRAVA VELKÉHO POLE PRO ZÁPIS (BATCH) ===
    ' ==================================================================
    
    ' Odhadneme počet řádků (Počet záznamů + Počet dní + Rezerva pro prázdné dny)
    Dim estimatedRows As Long
    estimatedRows = UBound(outputData, 1) + dayDetails.count + DateDiff("d", minDate, maxDate) + 10
    
    Dim finalData() As Variant
    ReDim finalData(1 To estimatedRows, 1 To 4)
    Dim writeIdx As Long: writeIdx = 0
    
    ' Kolekce pro odložené formátování (Range objekty - nyní pole stringů pro Buffer)
    ' Změna: Místo Unionu budeme používat Buffer
    Dim strGreen As String, strBlue As String
    Dim countGreen As Long, countBlue As Long
    
    Dim rngBoldText As Range, rngItalicText As Range
    Dim richData As Collection: Set richData = New Collection ' Pro hyperlinky a komentáře
    
    Dim currentDate As Date
    Dim emptyStartDate As Date, emptyEndDate As Date: emptyStartDate = 0
    
    ' Pomocná proměnná pro cyklus
    Dim loopDays As Variant
    
    ' Logika průchodu (Dvě větve: Seřazené vs Datum)
    If g_SortBy <> "Date" Then
        ' --- Větev A: Řazení dle statistik ---
        For i = 1 To UBound(daysArray, 1)
            writeIdx = writeIdx + 1
            ' Hlavička dne
            finalData(writeIdx, 1) = "(" & daysArray(i, 6) & "x)"
            finalData(writeIdx, 2) = "(" & daysArray(i, 2) & "x)"
            finalData(writeIdx, 3) = FormatWatchTime(daysArray(i, 3), True)
            finalData(writeIdx, 4) = daysArray(i, 1) ' Datum
            
            ' OPT-8: Sestavení DateRowMap z paměti
            If Not g_DateRowMap.Exists(daysArray(i, 1)) Then
                g_DateRowMap.Add daysArray(i, 1), writeIdx + 2
            End If
            
            ' Přidat do Green Bufferu
            AddRowToBuffer strGreen, countGreen, writeIdx + 2
            
            Set detailsCollection = daysArray(i, 5)
            For Each detail In detailsCollection
                writeIdx = writeIdx + 1
                ProcessDetailRow finalData, writeIdx, detail, richData
            Next detail
        Next i
        
    Else
        ' --- Větev B: Řazení dle Data (včetně prázdných dní) ---
        For currentDate = maxDate To minDate Step -1
            If dayDetails.Exists(currentDate) Then
                ' 1. Vyřešit předchozí prázdné dny
                If emptyStartDate <> 0 Then
                    emptyEndDate = currentDate + 1
                    writeIdx = writeIdx + 1
                    
                    finalData(writeIdx, 1) = "Nic nebylo zhlédnuto"
                    finalData(writeIdx, 2) = "(0x)"
                    finalData(writeIdx, 3) = ":-("
                    
                    If emptyStartDate = emptyEndDate Then
                         finalData(writeIdx, 4) = emptyStartDate
                    Else
                         finalData(writeIdx, 4) = FormatDateRange(emptyEndDate, emptyStartDate)
                    End If
                    
                    ' Přidat do Blue Bufferu a Italic
                    AddRowToBuffer strBlue, countBlue, writeIdx + 2
                    AddToRange rngItalicText, wsLog.Cells(writeIdx + 2, 1)
                    
                    emptyStartDate = 0
                End If
                
                ' 2. Zapsat aktuální den
                Set detailsCollection = dayDetails(currentDate)
                daySumMinutes = 0: daySumEpisodes = 0
                For Each detail In detailsCollection
                    daySumEpisodes = daySumEpisodes + CLng(detail(4)) ' OPTIMALIZACE
                    daySumMinutes = daySumMinutes + detail(2)
                Next detail
                
                writeIdx = writeIdx + 1
                finalData(writeIdx, 1) = "(" & detailsCollection.count & "x)"
                finalData(writeIdx, 2) = "(" & daySumEpisodes & "x)"
                finalData(writeIdx, 3) = FormatWatchTime(daySumMinutes, True)
                finalData(writeIdx, 4) = currentDate
                
                ' OPT-8: Sestavení DateRowMap z paměti
                If Not g_DateRowMap.Exists(currentDate) Then
                    g_DateRowMap.Add currentDate, writeIdx + 2
                End If
                
                ' Přidat do Green Bufferu
                AddRowToBuffer strGreen, countGreen, writeIdx + 2
                
                ' 3. Zapsat detaily
                For Each detail In detailsCollection
                    writeIdx = writeIdx + 1
                    ProcessDetailRow finalData, writeIdx, detail, richData
                Next detail
            Else
                If emptyStartDate = 0 Then emptyStartDate = currentDate
            End If
        Next currentDate
        
        ' Dojezd prázdných dní na konci
        If emptyStartDate <> 0 Then
            emptyEndDate = minDate
            writeIdx = writeIdx + 1
            finalData(writeIdx, 1) = "Nic nebylo zhlédnuto"
            finalData(writeIdx, 2) = "(0x)"
            finalData(writeIdx, 3) = ":-("
            If emptyStartDate = emptyEndDate Then
                 finalData(writeIdx, 4) = emptyStartDate
            Else
                 finalData(writeIdx, 4) = FormatDateRange(emptyEndDate, emptyStartDate)
            End If
            AddRowToBuffer strBlue, countBlue, writeIdx + 2
            AddToRange rngItalicText, wsLog.Cells(writeIdx + 2, 1)
        End If
    End If
    
    ' ==================================================================
    ' === FÁZE ZÁPISU A FORMÁTOVÁNÍ (BATCH) ===
    ' ==================================================================
    
    ' 1. Hromadný zápis dat (BLESKOVÉ)
    ' OPT-5: Pro absolutní bezpečnost dat a správné typy zapíšeme celou tabulku přes .Value,
    ' a následně přepíšeme pouze sloupec A přes .Formula, aby se správně vyhodnotily vzorce HYPERLINK.
    If writeIdx > 0 Then
        ' Zápis všech surových hodnot (bezpečné pro data a čísla)
        wsLog.Range("A3").Resize(writeIdx, 4).Value = finalData
        
        ' Extrakce sloupce A se vzorci a jeho zápis jako Formula (Excel vzorce zanalyzuje a přeloží)
        Dim colA() As Variant
        ReDim colA(1 To writeIdx, 1 To 1)
        Dim idxA As Long
        For idxA = 1 To writeIdx
            colA(idxA, 1) = finalData(idxA, 1)
        Next idxA
        wsLog.Range("A3").Resize(writeIdx, 1).Formula = colA
        
        ' OPT-5: Jednorázový formát písma pro sloupec A (barva automatická, bez podtržení)
        ' Nahrazuje stovky per-cell .Font.ColorIndex a .Font.Underline volání
        With wsLog.Range("A3").Resize(writeIdx, 1)
            .Font.ColorIndex = xlAutomatic
            .Font.Underline = xlUnderlineStyleNone
        End With
        
        ' Formátování sloupce s datem (D) - základní nátěr
        With wsLog.Range("D3").Resize(writeIdx, 1)
            .NumberFormat = "dd\.mm\.yyyy"
            .HorizontalAlignment = xlRight
            .VerticalAlignment = xlCenter
            .WrapText = False
        End With
    End If
    
    ' DateRowMap se nyní plní přímo v paměti během generování finalData, není nutná žádná dodatečná inicializace ani čtení z listu.

    
' =========================================================================================
    ' === NOVÁ SEKVENCE BARVENÍ A OHRANIČENÍ (ŠEDÝ DESIGN) ===
    ' =========================================================================================
    
    Dim tableRange As Range
    
    ' 1. Aplikace základního designu na celou tabulku (pokud existují data)
    ' Toto vytvoří "šedý podklad" (#BFBFBF) a šedé linky (#808080)
    If writeIdx > 0 Then
        Set tableRange = wsLog.Range("A3:D" & (writeIdx + 2))
        
        With tableRange
            ' Výplň: Světle šedá #BFBFBF (RGB 191, 191, 191)
            .Interior.Color = RGB(191, 191, 191)
            
            ' Vnitřní horizontální čáry: Tmavší šedá #808080 (RGB 128, 128, 128)
            With .Borders(xlInsideHorizontal)
                .LineStyle = xlContinuous
                .weight = xlThin
                .Color = RGB(128, 128, 128)
            End With
            
            ' Vnitřní vertikální čáry: Černé (RGB 0, 0, 0)
            With .Borders(xlInsideVertical)
                .LineStyle = xlContinuous
                .weight = xlThin
                .Color = RGB(0, 0, 0)
            End With
            
            ' Vnější ohraničení celé tabulky: Černé a středně tlusté
            .BorderAround LineStyle:=xlContinuous, weight:=xlMedium, Color:=RGB(0, 0, 0)
        End With
    End If

    ' 2. Aplikace formátu z Bufferů (Zelené a Modré oblasti) - OPTIMALIZOVANÁ BATCH VERZE (OPRAVENÁ)
    ' Toto se provádí AŽ PO základním nátěru, aby se šedá barva přepsala
    
    Dim rngUnionG As Range, cntG As Long
    Dim rngUnionB As Range, cntB As Long
    
    ' Aplikace Zelené (Headers)
    If strGreen <> "" Then
        Dim partsG() As String
        partsG = Split(strGreen, ",")
        Dim pG As Variant
        Dim rngG As Range
        
        cntG = 0
        Set rngUnionG = Nothing
        
        For Each pG In partsG
            If pG <> "" Then
                Set rngG = wsLog.Range(pG)
                
                ' 1. Přidání do Unionu
                If rngUnionG Is Nothing Then
                    Set rngUnionG = rngG
                Else
                    Set rngUnionG = Union(rngUnionG, rngG)
                End If
                cntG = cntG + 1
                
                ' 2. FLUSH po 40 položkách (aby se aplikovaly formáty a uvolnil Union)
                If cntG >= 40 Then
                    With rngUnionG
                        .Font.Bold = True
                        .Interior.Color = RGB(142, 217, 115)
                        .Font.Color = vbBlack ' !!! Důležité: Text musí být vidět (černý)
                        
                        ' Aplikace hranic pro celou oblast Unionu (Excel to udělá chytře pro každou Area)
                        ' DateRowMap se již plní z paměti, není nutné sahat na buňky listu.
                        With .Borders(xlEdgeTop)
                            .LineStyle = xlContinuous: .weight = xlMedium: .Color = vbBlack
                        End With
                        With .Borders(xlEdgeBottom)
                            .LineStyle = xlContinuous: .weight = xlThin: .Color = vbBlack
                        End With
                        .Borders(xlInsideHorizontal).LineStyle = xlNone
                    End With
                    Set rngUnionG = Nothing
                    cntG = 0
                End If
                
                ' OPT-8: Mapu dat plníme přímo v paměti z finalData
            End If
        Next pG
        
        ' 4. Final FLUSH zbytku
        If Not rngUnionG Is Nothing Then
            With rngUnionG
                .Font.Bold = True
                .Interior.Color = RGB(142, 217, 115)
                .Font.Color = vbBlack
                
                With .Borders(xlEdgeTop)
                    .LineStyle = xlContinuous: .weight = xlMedium: .Color = vbBlack
                End With
                With .Borders(xlEdgeBottom)
                    .LineStyle = xlContinuous: .weight = xlThin: .Color = vbBlack
                End With
                .Borders(xlInsideHorizontal).LineStyle = xlNone
            End With
            Set rngUnionG = Nothing
        End If
    End If
    
    ' Aplikace Modré (Empty)
    If strBlue <> "" Then
        Dim partsB() As String
        partsB = Split(strBlue, ",")
        Dim pB As Variant
        Dim rngB As Range
        
        cntB = 0
        Set rngUnionB = Nothing
        
        For Each pB In partsB
            If pB <> "" Then
                Set rngB = wsLog.Range(pB)
                
                ' 1. Přidání do Unionu
                If rngUnionB Is Nothing Then
                    Set rngUnionB = rngB
                Else
                    Set rngUnionB = Union(rngUnionB, rngB)
                End If
                cntB = cntB + 1
                
                ' 2. Zarovnání Textu (Musíme řešit per-row, protože to závisí na obsahu)
                ' Toto nelze efektivně batchovat v jednom kroku, pokud se liší.
                ' Ale je to rychlá vlastnost.
                If InStr(rngB.Cells(1, 4).text, "-") > 0 Then
                     rngB.Cells(1, 4).WrapText = True
                     rngB.HorizontalAlignment = xlLeft
                Else
                     rngB.Range("A1:C1").HorizontalAlignment = xlLeft
                     rngB.Cells(1, 4).HorizontalAlignment = xlRight
                End If
                
                ' 3. FLUSH po 40 položkách
                If cntB >= 40 Then
                    With rngUnionB
                        .Font.Bold = True
                        .Interior.Color = RGB(97, 203, 243)
                        .Font.Color = vbBlack ' !!! Text viditelný
                        .VerticalAlignment = xlCenter
                        .RowHeight = 30
                        
                        With .Borders(xlEdgeTop)
                            .LineStyle = xlContinuous: .weight = xlMedium: .Color = vbBlack
                        End With
                        With .Borders(xlEdgeBottom)
                            .LineStyle = xlContinuous: .weight = xlMedium: .Color = vbBlack
                        End With
                        .Borders(xlInsideHorizontal).LineStyle = xlNone
                    End With
                    Set rngUnionB = Nothing
                    cntB = 0
                End If
            End If
        Next pB
        
        ' 4. Final FLUSH Modré
        If Not rngUnionB Is Nothing Then
            With rngUnionB
                .Font.Bold = True
                .Interior.Color = RGB(97, 203, 243)
                .Font.Color = vbBlack
                .VerticalAlignment = xlCenter
                .RowHeight = 30
                With .Borders(xlEdgeTop)
                    .LineStyle = xlContinuous: .weight = xlMedium: .Color = vbBlack
                End With
                With .Borders(xlEdgeBottom)
                    .LineStyle = xlContinuous: .weight = xlMedium: .Color = vbBlack
                End With
                .Borders(xlInsideHorizontal).LineStyle = xlNone
            End With
            Set rngUnionB = Nothing
        End If
    End If
    
    If Not rngItalicText Is Nothing Then
        rngItalicText.Font.Italic = True
    End If

    ' (Původní sekce 3. "Ohraničení celé tabulky" je nyní integrována v bodě 1, takže ji mažeme)

    ' =========================================================================================
    ' === KONEC NOVÉ SEKVENCE ===
    ' =========================================================================================

    ' 4. Aplikace "Bohatých" dat (Komentáře, Bold Search) - OPT-5: Hyperlinky řešeny vzorcem
    ' Tento cyklus se nyní spouští JEN pro řádky s rewatchem nebo hledáním
    Dim item As Variant
    Dim rNum As Long
    Dim fName As String, rewNum As String
    Dim cCell As Range
    
    For Each item In richData
        rNum = item(0) ' Row Index (absolutní v listu)
        fName = item(1) ' Display Name
        rewNum = item(2) ' Rewatch Text
        
        Set cCell = wsLog.Cells(rNum, 1)
        
        ' Komentář (OPT-2)
        If rewNum <> "" Then
            cCell.AddComment rewNum & ". Rewatch"
            ' Použijeme dokonale vybalancované rozměry (předchází zalamování a odseknutí textu)
            With cCell.Comment.Shape
                .Width = 68
                .Height = 15
                .TextFrame.Characters.Font.Bold = True
            End With
        End If
    Next item

    ' OPT-7: Sloučení průchodů Blue bufferu - formátování a zarovnání sloupce D proběhlo koordinovaně v jednom průchodu

    ' Obnovení nastavení
    wsLog.Calculate
End Sub

' === POMOCNÁ PROCEDURA PRO BATCH BUFFER (OPTIMALIZACE UNION) ===
Private Sub AddRowToBuffer(ByRef buffer As String, ByRef count As Long, ByVal rowIdx As Long)
    ' Přidá adresu do bufferu. Pokud buffer překročí limit, zpracování se dělá v hlavní smyčce (zde jen hromadíme string)
    ' Ale pro jednoduchost v VBA stringy snesou hodně.
    ' Limit address stringu pro Range() je 255 znaků.
    ' Proto používáme trik: Oddělíme čárkou a v hlavní smyčce pak Splitneme a projdeme.
    ' Tím se vyhneme Union objektu, ale stále musíme volat Range().
    ' Pro 3000 položek bude string obrovský, to nevadí? VBA String má 2GB.
    ' Ale Split trvá. Je to rychlejší než Union? ANO, řádově. Union musí přepočítat strom.
    
    Dim newAddr As String
    newAddr = "A" & rowIdx & ":D" & rowIdx
    
    If buffer = "" Then
        buffer = newAddr
    Else
        buffer = buffer & "," & newAddr
    End If
    count = count + 1
End Sub

' === POMOCNÁ PROCEDURA PRO UNION (Zachováno pro Italics) ===
Private Sub AddToRange(ByRef rng As Range, ByVal newRng As Range)
    If rng Is Nothing Then
        Set rng = newRng
    Else
        Set rng = Union(rng, newRng)
    End If
End Sub

' === POMOCNÁ PROCEDURA PRO PŘÍPRAVU ŘÁDKU ===
Private Sub ProcessDetailRow(ByRef finalData() As Variant, ByVal idx As Long, ByVal detail As Variant, ByRef richData As Collection)
    Dim fullAnimeName As String: fullAnimeName = detail(0)
    Dim displayName As String
    Dim rewatchNum As String
    
    If InStr(fullAnimeName, "|RW|") > 0 Then
        Dim parts() As String
        parts = Split(fullAnimeName, "|RW|")
        displayName = parts(0)
        rewatchNum = parts(1)
    Else
        displayName = fullAnimeName
        rewatchNum = ""
    End If
    
    ' OPT-5: Zápis HYPERLINK vzorce přímo do pole (místo pozdějšího Hyperlinks.Add)
    Dim sourceRow As Long: sourceRow = CLng(detail(3))
    If sourceRow > 0 Then
        ' Escapování uvozovek v názvu anime pro vzorec
        Dim safeName As String: safeName = Replace(displayName, Chr(34), Chr(34) & Chr(34))
        finalData(idx, 1) = "=HYPERLINK(""#'ANIME LIST'!C" & sourceRow & """,""" & safeName & """)"
    Else
        finalData(idx, 1) = displayName
    End If
    finalData(idx, 2) = detail(1) ' Episodes
    finalData(idx, 3) = FormatWatchTime(detail(2), False) ' Time
    ' Datum (sloupec 4) necháváme prázdné pro detaily
    
    ' Uložení metadat pro pozdější formátování (Hyperlink již řešen vzorcem)
    ' Array Structure: [RowIndex, DisplayName, RewatchNum]
    ' RowIndex v listu je idx + 2 (protože začínáme na A3)
    ' OPT: Plníme pouze tehdy, když řádek obsahuje rewatch (pro komentář), čímž dramaticky šetříme čas
    If rewatchNum <> "" Then
        richData.Add Array(idx + 2, displayName, rewatchNum)
    End If
End Sub

' =========================================================================================
' === POMOCNÉ FUNKCE A PROCEDURY (HISTORY LOG)
' =========================================================================================
Function IsArrayInitialized(arr As Variant) As Boolean
    On Error Resume Next
    IsArrayInitialized = (UBound(arr, 1) >= LBound(arr, 1))
    If err.Number <> 0 Then IsArrayInitialized = False
    On Error GoTo 0
End Function

Function FormatWatchTime(ByVal totalMinutes As Double, ByVal isSummary As Boolean) As String
    If totalMinutes <= 0 Then
        FormatWatchTime = "0 min"
        Exit Function
    End If
    
    Dim roundedMinutes As Long
    roundedMinutes = Round(totalMinutes, 0)
    
    Dim hoursDecimal As Double
    hoursDecimal = totalMinutes / 60
    
    Dim hoursString As String
    hoursString = Replace(Format(hoursDecimal, "0.0"), ".", ",")

    If isSummary Then
        FormatWatchTime = roundedMinutes & " min (" & hoursString & " hod)"
    Else
        If totalMinutes < 60 Then
            FormatWatchTime = roundedMinutes & " min"
        Else
            FormatWatchTime = roundedMinutes & " min (" & hoursString & " hod)"
        End If
    End If
End Function

Function FormatEpisodes(ByVal epDict As Object) As String
    On Error GoTo CleanExit
    If epDict.count = 0 Then
        FormatEpisodes = "(0x)"
        Exit Function
    End If
    Dim keys As Variant
    keys = epDict.keys
    If Not IsNumeric(keys(0)) Then
        FormatEpisodes = keys(0)
        Exit Function
    End If
    Dim totalCount As Long: totalCount = epDict.count
    Dim nums() As Long, i As Long
    ReDim nums(0 To epDict.count - 1)
    For i = 0 To UBound(keys)
        nums(i) = CLng(keys(i))
    Next i
    QuickSort1DArray nums, 0, UBound(nums)
    Dim episodeRangeString As String, startRange As Long
    startRange = nums(0)
    For i = 1 To UBound(nums)
        If nums(i) <> nums(i - 1) + 1 Then
            If startRange = nums(i - 1) Then
                episodeRangeString = episodeRangeString & startRange & ", "
            Else
                episodeRangeString = episodeRangeString & startRange & "-" & nums(i - 1) & ", "
            End If
            startRange = nums(i)
        End If
    Next i
    If startRange = nums(UBound(nums)) Then
        episodeRangeString = episodeRangeString & startRange
    Else
        episodeRangeString = episodeRangeString & startRange & "-" & nums(UBound(nums))
    End If
    If Right(episodeRangeString, 2) = ", " Then
        episodeRangeString = left(episodeRangeString, Len(episodeRangeString) - 2)
    End If
    FormatEpisodes = "(" & totalCount & "x) EP " & episodeRangeString
CleanExit:
    If err.Number <> 0 Then FormatEpisodes = "Chyba formátování"
End Function

Function ParseEpisodeCount(ByVal epString As String) As Long
    On Error Resume Next
    Dim startPos As Long, endPos As Long
    startPos = InStr(1, epString, "(")
    endPos = InStr(startPos, epString, "x)")
    If startPos > 0 And endPos > 0 Then
        ParseEpisodeCount = CLng(Mid(epString, startPos + 1, endPos - startPos - 1))
    Else
        ParseEpisodeCount = 0
    End If
End Function



' === QUICKSORT PRO 1D POLE (NÁHRADA ZA BUBBLESORT) ===
Private Sub QuickSort1DArray(ByRef arr() As Long, ByVal First As Long, ByVal Last As Long)
    Dim pivot As Long, Temp As Long
    Dim i As Long, j As Long
    
    If First >= Last Then Exit Sub
    
    pivot = arr((First + Last) \ 2)
    i = First
    j = Last
    
    Do
        Do While arr(i) < pivot: i = i + 1: Loop
        Do While arr(j) > pivot: j = j - 1: Loop
        
        If i <= j Then
            Temp = arr(i)
            arr(i) = arr(j)
            arr(j) = Temp
            i = i + 1
            j = j - 1
        End If
    Loop Until i > j
    
    If First < j Then QuickSort1DArray arr, First, j
    If i < Last Then QuickSort1DArray arr, i, Last
End Sub


Function Nz(Expression As Variant, Optional ValueIfNull As Variant) As Variant
    If IsMissing(ValueIfNull) Then ValueIfNull = ""
    If IsNull(Expression) Or IsEmpty(Expression) Then
        Nz = ValueIfNull
    Else
        Nz = Expression
    End If
End Function

Private Function FormatDateRange(startDate As Date, endDate As Date) As String
    ' Formátuje rozsah datumů. VŽDY používá plný formát dd.mm.yyyy.
    If startDate = endDate Then
        FormatDateRange = Format(startDate, "dd.mm.yyyy")
    Else
        FormatDateRange = Format(startDate, "dd.mm.yyyy") & " - " & Format(endDate, "dd.mm.yyyy")
    End If
End Function

' === QUICKSORT PRO 2D POLE (OPRAVENÁ VERZE) ===
Private Sub QuickSort2DArray(ByRef arr As Variant, ByVal sortCol As Long, ByVal First As Long, ByVal Last As Long, Optional ascending As Boolean = True)
    Dim pivot As Variant, Temp As Variant
    Dim i As Long, j As Long, k As Long
    Dim numCols As Long
    
    If First >= Last Then Exit Sub
    
    numCols = UBound(arr, 2)
    pivot = arr((First + Last) \ 2, sortCol)
    i = First
    j = Last
    
    Do
        If ascending Then
            Do While arr(i, sortCol) < pivot: i = i + 1: Loop
            Do While arr(j, sortCol) > pivot: j = j - 1: Loop
        Else
            Do While arr(i, sortCol) > pivot: i = i + 1: Loop
            Do While arr(j, sortCol) < pivot: j = j - 1: Loop
        End If
        
        If i <= j Then
            ' OPRAVENÁ ČÁST: Bezpečné prohození celých řádků (včetně objektů)
            For k = LBound(arr, 2) To numCols
                
                ' 1. Uložit arr(i, k) do temp
                If IsObject(arr(i, k)) Then
                    Set Temp = arr(i, k)
                Else
                    Temp = arr(i, k)
                End If
                
                ' 2. Přesunout arr(j, k) do arr(i, k)
                If IsObject(arr(j, k)) Then
                    Set arr(i, k) = arr(j, k)
                Else
                    arr(i, k) = arr(j, k)
                End If
                
                ' 3. Přesunout temp do arr(j, k)
                If IsObject(Temp) Then
                    Set arr(j, k) = Temp
                Else
                    arr(j, k) = Temp
                End If
                
            Next k
            
            i = i + 1
            j = j - 1
        End If
    Loop Until i > j
    
    If First < j Then QuickSort2DArray arr, sortCol, First, j, ascending
    If i < Last Then QuickSort2DArray arr, sortCol, i, Last, ascending
End Sub

Public Sub VytvoritHeatmapu(ByRef dailyTotals As Scripting.Dictionary, ByVal targetSheetName As String, ByVal targetCellAddress As String, Optional ByRef dateRowMap As Scripting.Dictionary = Nothing)
    On Error GoTo Chyba
    
    Dim wsLog As Worksheet
    Set wsLog = ThisWorkbook.Worksheets(targetSheetName)

    Const HEATMAP_WEEKS As Integer = 53

    Const HEATMAP_COLOR_LEVEL_2 As Integer = 6
    Const HEATMAP_COLOR_LEVEL_3 As Integer = 13

    If dailyTotals Is Nothing Then
        ' Vyčištění při prázdných datech

Private Sub GenerovatHeatmapu_Core(ByRef dailyTotals As Scripting.Dictionary, ByRef targetRange As Range, ByRef wsLog As Worksheet, ByVal HEATMAP_WEEKS As Integer, ByVal STATS_EPISODES_SHAPE As String, ByVal STATS_HOURS_SHAPE As String, ByVal HEATMAP_COLOR_LEVEL_1 As Integer, ByVal HEATMAP_COLOR_LEVEL_2 As Integer, ByVal HEATMAP_COLOR_LEVEL_3 As Integer, ByVal HEATMAP_COLOR_LEVEL_4 As Integer, Optional ByRef dateRowMap As Scripting.Dictionary = Nothing)
    
    If dailyTotals Is Nothing Or dailyTotals.count = 0 Then Exit Sub

    ' Zjištění rozsahu dat
    Dim minDate As Date, maxDate As Date
    Dim keys As Variant: keys = dailyTotals.keys
    minDate = CDate(keys(0)): maxDate = CDate(keys(0))
    Dim dKey As Variant
    For Each dKey In keys
        If CDate(dKey) > maxDate Then maxDate = CDate(dKey)
    Next dKey
    
    ' Výpočet startovního data (vždy pondělí před X týdny)
    Dim startDate As Date
    startDate = maxDate - (HEATMAP_WEEKS * 7) + 1
    startDate = startDate - (Weekday(startDate, vbMonday) - 1)
    
    ' Rychlý součet statistik pro Shapes
    Dim totalEpisodesInPeriod As Long: totalEpisodesInPeriod = 0
    Dim totalMinutesInPeriod As Double: totalMinutesInPeriod = 0
    Dim tempDate As Date
    For tempDate = startDate To maxDate
        If dailyTotals.Exists(tempDate) Then
            Dim tempDay As Variant
            tempDay = dailyTotals(tempDate)
            totalEpisodesInPeriod = totalEpisodesInPeriod + tempDay(0)
            totalMinutesInPeriod = totalMinutesInPeriod + tempDay(1)
        End If
    Next tempDate
    
    ' --- 2. HROMADNÁ PŘÍPRAVA OBLASTI (Batch Processing) ---
    Dim fullHeatmapArea As Range
    ' Oblast pro dny (7 řádků x 53 sloupců) - posunuto o 2 dolů a 1 doprava
    Set fullHeatmapArea = targetRange.Offset(1, 1).Resize(7, HEATMAP_WEEKS)
    
    ' Vyčištění a základní formát NAJEDNOU
    ' KROK 1: Resize na 11 řádků (E38 až E48). Řádek 49 zůstane nedotčen.
    With targetRange.Resize(10, HEATMAP_WEEKS + 1)
        .Clear
        .Font.Name = "Aptos Narrow" ' Nebo tvůj font
        
        .Interior.Color = RGB(191, 191, 191) ' Vyplnit pozadi #BFBFBF
    End With
    
    ' Nastavení popisků dnů
    targetRange.Offset(1, 0).Value = "Po"
    targetRange.Offset(3, 0).Value = "St"
    targetRange.Offset(5, 0).Value = "Pá"
    
    With targetRange.Offset(1, 0).Resize(5)
        .Font.size = 8
        .Font.Bold = True
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
    End With

    ' Aplikace mřížky a DEFAULTNÍ BARVY (Level 0) na celou oblast naráz
    With fullHeatmapArea
        .Interior.Color = RGB(235, 237, 240) ' Šedá (Level 0) - vnitřek mřížky
        .Font.Color = RGB(235, 237, 240)     ' Skryje text "XX" pro hyperlinky
        With .Borders
            .LineStyle = xlContinuous
            .Color = vbWhite
            .weight = xlThin
        End With
    End With
    
    ' Specifické ohraničení pravé strany mřížky (vnitřní)
    With fullHeatmapArea.Columns(HEATMAP_WEEKS).Borders(xlEdgeRight)
        .LineStyle = xlContinuous
        .weight = xlMedium
        .ColorIndex = xlAutomatic
    End With

    ' Aktualizace Shapes
    Call VytvorNeboAktualizujStatistiky_Core(wsLog, totalEpisodesInPeriod, totalMinutesInPeriod, startDate, maxDate, STATS_EPISODES_SHAPE, STATS_HOURS_SHAPE)

    ' --- 3. HLAVNÍ SMYČKA ---
    ' Použijeme pole pro měsíce, abychom nezapisovali do buněk v každém cyklu
    Dim arrMonths() As Variant
    ReDim arrMonths(1 To 1, 1 To HEATMAP_WEEKS)
    
    ' BATCH 3.0: Object-Based Batching (Union s limitem)
    ' Toto je kompromis mezi rychlostí a stabilitou. Union se resetuje každých 50 buněk,
    ' což zabrání exponenciálnímu zpomalení, ale vyhne se problémům s délkou stringu u Range().
    Dim rngLevel1 As Range, cntLevel1 As Long
    Dim rngLevel2 As Range, cntLevel2 As Long
    Dim rngLevel3 As Range, cntLevel3 As Long
    Dim rngLevel4 As Range, cntLevel4 As Long
    Dim rngLevel5 As Range, cntLevel5 As Long
    
    Dim C As Long, r As Long
    Dim currentMonth As Integer: currentMonth = 0
    Dim currentDate As Date, mondayDate As Date
    Dim cell As Range
    Dim episodeCount As Long
    Dim minutesForDay As Double
    Dim cmtText As String
    
    ' Pro zrychlení komentářů a hyperlinků
    Dim hasActivity As Boolean
    
    For C = 1 To HEATMAP_WEEKS
        mondayDate = startDate + ((C - 1) * 7)
        
        ' Logika měsíců (do pole)
        If Month(mondayDate) <> currentMonth Then
            currentMonth = Month(mondayDate)
            arrMonths(1, C) = Format(mondayDate, "mmm")
        End If

        For r = 1 To 7
            currentDate = mondayDate + (r - 1)
            If currentDate > maxDate Then GoTo LoopFinished
            
            ' Získání dat
            If dailyTotals.Exists(currentDate) Then
                Dim tempDay2 As Variant
                tempDay2 = dailyTotals(currentDate)
                episodeCount = tempDay2(0)
                minutesForDay = tempDay2(1)
                hasActivity = (episodeCount > 0)
            Else
                episodeCount = 0
                minutesForDay = 0
                hasActivity = False
            End If
            
            Set cell = fullHeatmapArea.Cells(r, C)
            
            ' A. Barvení (Object Batch) a Hyperlinky
            If hasActivity Then
                ' 1. Hyperlink (Musí být PRVNÍ, aby Batch Color přepsal modrou barvu odkazu)
                If Not dateRowMap Is Nothing Then
                    If dateRowMap.Exists(currentDate) Then
                          wsLog.Hyperlinks.Add Anchor:=cell, Address:="", subAddress:="'" & wsLog.Name & "'!D" & dateRowMap(currentDate), _
                                                           ScreenTip:="Přejít na záznamy", TextToDisplay:="XX"
                    End If
                End If
                
                ' 2. Barvení (Object Batch)
                Select Case episodeCount
                    Case 1 To HEATMAP_COLOR_LEVEL_1:      AddCellToBatch rngLevel1, cntLevel1, cell, RGB(10, 50, 20)
                    Case Is <= HEATMAP_COLOR_LEVEL_2:     AddCellToBatch rngLevel2, cntLevel2, cell, RGB(25, 97, 39)
                    Case Is <= HEATMAP_COLOR_LEVEL_3:     AddCellToBatch rngLevel3, cntLevel3, cell, RGB(35, 154, 59)
                    Case Is <= HEATMAP_COLOR_LEVEL_4:     AddCellToBatch rngLevel4, cntLevel4, cell, RGB(123, 201, 111)
                    Case Else:                            AddCellToBatch rngLevel5, cntLevel5, cell, RGB(198, 238, 172)
                End Select
            End If
            
            ' C. Komentáře
            If hasActivity Then
                Dim episodeText As String
                Select Case episodeCount
                    Case 1: episodeText = " epizoda"
                    Case 2 To 4: episodeText = " epizody"
                    Case Else: episodeText = " epizod"
                End Select
                cmtText = Format(currentDate, "d. mmmm yyyy") & vbCrLf & _
                          episodeCount & episodeText & " (" & Replace(Format(minutesForDay / 60, "0.0"), ".", ",") & " h)"
            Else
                cmtText = Format(currentDate, "d. mmmm yyyy")
            End If
            
            cell.AddComment cmtText
            ' Použijeme dokonale přizpůsobené rozměry pro minimalizaci prázdného místa (OPT-2)
            With cell.Comment.Shape
                If hasActivity Then
                    .Width = 90
                    .Height = 26
                Else
                    .Width = 80
                    .Height = 14
                End If
            End With
            
        Next r
    Next C

LoopFinished:
    
    ' --- 4. FINÁLNÍ ZÁPIS A DO-BARVENÍ (Flush zbytek) ---
    
    ' Zápis měsíců naráz
    With targetRange.Offset(0, 1).Resize(1, HEATMAP_WEEKS)
        .Value = arrMonths
        .Font.size = 9
        .Font.Bold = True
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
    End With
    
    ' Flush zbývajících cellů v Union objektech
    FlushBatch rngLevel1, RGB(10, 50, 20)
    FlushBatch rngLevel2, RGB(25, 97, 39)
    FlushBatch rngLevel3, RGB(35, 154, 59)
    FlushBatch rngLevel4, RGB(123, 201, 111)

Private Sub AddCellToBatch(ByRef rng As Range, ByRef cnt As Long, ByVal cell As Range, ByVal colorRGB As Long)
    If rng Is Nothing Then
        Set rng = cell
    Else
        Set rng = Union(rng, cell)
    End If
    cnt = cnt + 1
    
    ' Flush každých 40 položek (Prevence "Method Union Failed" nebo zpomalení)
    If cnt >= 40 Then
        rng.Interior.Color = colorRGB
        rng.Font.Color = colorRGB
        Set rng = Nothing
        cnt = 0
    End If
End Sub

Private Sub FlushBatch(ByRef rng As Range, ByVal colorRGB As Long)
    If Not rng Is Nothing Then
        rng.Interior.Color = colorRGB
        rng.Font.Color = colorRGB
        Set rng = Nothing
    End If
End Sub

Private Function ZiskatDataProHeatmapu_Core(ByRef data() As Variant) As Scripting.Dictionary
    Dim dictTotals As Scripting.Dictionary
    Set dictTotals = New Scripting.Dictionary
    
    If Not IsArrayInitialized(data) Then
        Set ZiskatDataProHeatmapu_Core = dictTotals
        Exit Function
    End If
    
    Dim i As Long
    For i = 1 To UBound(data, 1)
        Dim currentDate As Date: currentDate = CDate(data(i, 1))
        Dim episodeCount As Long
        Dim minutesForEntry As Double: minutesForEntry = CDbl(data(i, 4))
        
        ' OPTIMALIZACE: Heatmapa čte přímo ze sloupce 6
        If UBound(data, 2) >= 6 Then
            episodeCount = CLng(data(i, 6))
        Else
            Dim episodeString As String: episodeString = data(i, 3)
            episodeCount = ParseEpisodeCount(episodeString)
        End If
        
        If dictTotals.Exists(currentDate) Then
            Dim tempDay As Variant
            tempDay = dictTotals(currentDate)
            tempDay(0) = tempDay(0) + episodeCount
            tempDay(1) = tempDay(1) + minutesForEntry
            dictTotals(currentDate) = tempDay
        Else
            Dim dayData(0 To 1) As Variant
            dayData(0) = episodeCount
            dayData(1) = minutesForEntry
            dictTotals.Add currentDate, dayData
        End If
    Next i
    
    Set ZiskatDataProHeatmapu_Core = dictTotals
End Function

Private Function DefinovatBarvu_Core(pocetEpizod As Long, ByVal L1 As Integer, ByVal L2 As Integer, ByVal l3 As Integer, ByVal l4 As Integer) As Long
    Select Case pocetEpizod
        Case 0:
            DefinovatBarvu_Core = RGB(235, 237, 240) ' Šedá
        Case 1 To L1:
            DefinovatBarvu_Core = RGB(10, 50, 20) ' Level 1
        Case (L1 + 1) To L2:
            DefinovatBarvu_Core = RGB(25, 97, 39)   ' Level 2
        Case (L2 + 1) To l3:
            DefinovatBarvu_Core = RGB(35, 154, 59)   ' Level 3
        Case (l3 + 1) To l4:
            DefinovatBarvu_Core = RGB(123, 201, 111)   ' Level 4 (původní nejtmavší)
        Case Is > l4:
            DefinovatBarvu_Core = RGB(198, 238, 172)    ' Level 5 (nová nejtmavší)
        Case Else:
            DefinovatBarvu_Core = RGB(235, 237, 240)
    End Select
End Function



' =========================================================================================
' === PARSER PRO P?ESN? D?LKY EPIZOD (Sloupec J)
' =========================================================================================
Private Function ParseEpDurationsFromComment(ByVal cmtText As String) As Scripting.Dictionary
    Dim dict As Scripting.Dictionary: Set dict = New Scripting.Dictionary
    If Len(cmtText) = 0 Then
        Set ParseEpDurationsFromComment = Nothing
        Exit Function
    End If
    
    Static reg As Object
    If reg Is Nothing Then
        Set reg = CreateObject("VBScript.RegExp")
        reg.Pattern = "(\d+)[\.\s]*EP:\s*(\d+):(\d+)"
        reg.Global = True
        reg.IgnoreCase = True
    End If
    
    Dim lines() As String: lines = Split(cmtText, vbLf)
    Dim vLine As Variant, matches As Object
    Dim epNum As Long, mins As Double, secs As Double
    
    For Each vLine In lines
        Set matches = reg.Execute(vLine)
        If matches.count > 0 Then
            epNum = CLng(matches(0).SubMatches(0))
            mins = CDbl(matches(0).SubMatches(1))
            secs = CDbl(matches(0).SubMatches(2))
            If Not dict.Exists(epNum) Then
                dict.Add epNum, mins + (secs / 60)
            End If
        End If
    Next vLine
    
    If dict.count > 0 Then Set ParseEpDurationsFromComment = dict Else Set ParseEpDurationsFromComment = Nothing
End Function

Private Function CalcEpMinutes(ByVal firstEp As Long, ByVal lastEp As Long, _
                               ByVal epOverrides As Scripting.Dictionary, _
                               ByVal defaultDuration As Double, _
                               ByVal totalEpisodes As Long, _
                               ByVal totalTime As Double) As Double
    If epOverrides Is Nothing Then
        CalcEpMinutes = (lastEp - firstEp + 1) * defaultDuration
        Exit Function
    End If

    Dim sumOverrides As Double: sumOverrides = 0
    Dim countOverrides As Long: countOverrides = epOverrides.count
    Dim key As Variant
    For Each key In epOverrides.keys
        sumOverrides = sumOverrides + epOverrides(key)
    Next key

    Dim avgRemaining As Double
    If totalEpisodes > countOverrides Then
        avgRemaining = (totalTime - sumOverrides) / (totalEpisodes - countOverrides)
        If avgRemaining < 0 Then avgRemaining = defaultDuration
    Else
        avgRemaining = defaultDuration
    End If

    Dim totalMins As Double: totalMins = 0
    Dim j As Long
    For j = firstEp To lastEp
        If epOverrides.Exists(j) Then
            totalMins = totalMins + epOverrides(j)
        Else
            totalMins = totalMins + avgRemaining
        End If
    Next j

    CalcEpMinutes = totalMins
End Function
