Sub CopyAnimeWatchPlanPrompt()
    ' ==========================================================================================
    ' POPIS: Načte Anime data, přidá info o délce epizod a vygeneruje PROMPT pro AI.
    ' ==========================================================================================
    
    Dim ws As Worksheet
    Dim lastRow As Long, i As Long
    Dim animeData As String
    Dim jsonItem As String
    Dim finalPrompt As String
    
    ' --- NASTAVENÍ SLOUPCŮ (Konstanty pro snadnou úpravu) ---
    Const COL_NAZEV As String = "G"
    Const COL_TYP As String = "H"
    Const COL_EP_POCET As String = "I"
    Const COL_EP_DELKA As String = "J"  ' Nový sloupec pro délku epizody
    Const COL_CAS_TOTAL As String = "K"
    Const COL_POZN As String = "L"
    Const COL_STATUS As String = "M"
    
    ' Nastavení listu
    Set ws = ThisWorkbook.Sheets("ANIME PLAN TO WATCH + FUTURES")
    lastRow = ws.Cells(ws.Rows.count, COL_NAZEV).End(xlUp).row
    
    ' Hlavička JSON
    animeData = "["
    
    ' --- ITERACE DATA ---
    For i = 2 To lastRow
        ' Kontrola, zda má Anime název
        If Len(ws.Range(COL_NAZEV & i).Value) > 0 Then
            
            ' 1. Ošetření celkového času (pro případ chyb #HODNOTA!)
            Dim timeTotal As String
            If IsError(ws.Range(COL_CAS_TOTAL & i).Value) Then
                timeTotal = "Neznámá délka"
            Else
                timeTotal = ws.Range(COL_CAS_TOTAL & i).Value & " min"
            End If
            
            ' 2. Načtení délky epizody (Sloupec J)
            Dim timeEp As String
            If IsNumeric(ws.Range(COL_EP_DELKA & i).Value) Then
                timeEp = ws.Range(COL_EP_DELKA & i).Value & " min"
            Else
                timeEp = "N/A"
            End If
            
            ' 3. Čištění poznámky (escape uvozovek, lomítek a odřádkování pro validní JSON)
            Dim reasonClean As String
            reasonClean = ws.Range(COL_POZN & i).Value
            reasonClean = Replace(reasonClean, "\", "\\")   ' Escape zpětných lomítek
            reasonClean = Replace(reasonClean, """", "'")    ' Nahrazení uvozovek apostrofy
            reasonClean = Replace(reasonClean, vbNewLine, " ")
            reasonClean = Replace(reasonClean, Chr(10), " ")
            reasonClean = Replace(reasonClean, Chr(13), " ")
            
            ' 4. Sestavení JSON záznamu
            jsonItem = "{" & _
                """Nazev"": """ & ws.Range(COL_NAZEV & i).Value & """, " & _
                """Typ"": """ & ws.Range(COL_TYP & i).Value & """, " & _
                """Pocet_EP"": """ & ws.Range(COL_EP_POCET & i).Value & """, " & _
                """Delka_EP"": """ & timeEp & """, " & _
                """Celkovy_Cas"": """ & timeTotal & """, " & _
                """Duvod_Poznamka"": """ & reasonClean & """, " & _
                """Status"": """ & ws.Range(COL_STATUS & i).Value & """" & _
                "}"
            
            ' Přidání čárky před další položku
            If Len(animeData) > 1 Then animeData = animeData & "," & vbNewLine
            animeData = animeData & jsonItem
        End If
    Next i
    
    animeData = animeData & "]"
    
    ' --- SESTAVENÍ PROMPTU ---
    finalPrompt = "Jsi můj osobní Anime kurátor a expert na time-management." & vbNewLine & vbNewLine & _
    "Mám zde svůj seznam 'Plan to Watch' v JSON formátu. Tvým úkolem je analyzovat data (včetně počtu epizod a délky) a poznámky, " & _
    "a vytvořit pro mě **3 různé Anime Watch Plány**." & vbNewLine & vbNewLine & _
    "**DATASET (Seznam Anime):**" & vbNewLine & _
    animeData & vbNewLine & vbNewLine & _
    "**PRAVIDLA PRO TVORBU PLÁNŮ:**" & vbNewLine & _
    "1. **Franchise Grouping:** Pokud je v seznamu více záznamů patřících do stejné série, drž je u sebe." & vbNewLine & _
    "2. **Respektuj Status:** Pokud je status 'Vydáno', mohu to bingewatchovat. Pokud 'Vysíláno', ber v potaz týdenní vydávání." & vbNewLine & _
    "3. **Tři různé varianty:** Vytvoř tři odlišné plány (např. 'Temný & Psychologický', 'Action & Hype', 'Chill' apod.)." & vbNewLine & vbNewLine & _
    "**POŽADOVANÝ VÝSTUP PRO KAŽDÝ PLÁN:**" & vbNewLine & _
    "- **Název Plánu:** (Vystihující téma)" & vbNewLine & _
    "- **Celkový čas:** (Kolik hodin zabere zhlédnutí celého plánu)" & vbNewLine & _
    "- **Celkový počet epizod:** (Sečti epizody všech anime v tomto plánu)" & vbNewLine & _
    "- **Seznam Anime:** Seřazený v doporučeném pořadí. Formát: [Název] | [Počet EP] x [Délka EP]" & vbNewLine & _
    "- **Důvod výběru:** Proč jsi vybral tato Anime k sobě?" & vbNewLine & _
    "- **Spojující prvek:** Co tato Anime spojuje?"

    ' --- KOPÍROVÁNÍ DO SCHRÁNKY (Robustní metoda přes HTML) ---
    Dim clipboardCheck As Boolean
    clipboardCheck = CopyTextToClipboard(finalPrompt)
    
    If clipboardCheck Then
        'MsgBox "Prompt zkopírován! Nyní vlož do AI (Ctrl+V).", vbInformation, "Anime Plan Generator"
    Else
        MsgBox "Chyba při kopírování do schránky.", vbCritical
    End If

End Sub

' Pomocná funkce pro spolehlivé kopírování na Win 10/11
Function CopyTextToClipboard(text As String) As Boolean
    On Error GoTo ErrHandler
    Dim htmlObj As Object
    Set htmlObj = CreateObject("htmlfile")
    htmlObj.parentWindow.clipboardData.setData "text", text
    CopyTextToClipboard = True
    Exit Function
ErrHandler:
    CopyTextToClipboard = False
End Function
