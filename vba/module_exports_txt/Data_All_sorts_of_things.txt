Option Explicit

' ********************************************************************************
' KONFIGURACE A KONSTANTY
' ********************************************************************************
' První rok, který sledujeme (Sloupec G)
Private Const ROK_START As Integer = 2024

' Poslední rok, který sledujeme (Změň na 2027, 2028 atd.)
Private Const ROK_KONEC As Integer = 2026

' Sloupec, kde začíná ROK_START (G = 7)
Private Const COL_START_INDEX As Integer = 7

' Datum, od kterého se počítají detailní statistiky (starší sledování se ignoruje pro roční stats)
Private Const DATUM_POCATKU_MERENI As Date = #1/1/2018#

' Mapování sloupců v poli dat
Private Const COL_IDX_NAME As Long = 1      ' C (Název)
Private Const COL_IDX_EPS As Long = 7       ' I (Počet epizod)
Private Const COL_IDX_MINS As Long = 8      ' J (Délka epizody)
Private Const COL_IDX_START As Long = 10    ' L (Datum začátku)
Private Const COL_IDX_REWATCH As Long = 12  ' N (Počet rewatchů)
Private Const COL_IDX_TOTAL As Long = 13    ' O (Celkový čas)

' ********************************************************************************
' HLAVNÍ SPOUŠTĚCÍ PROCEDURA
' ********************************************************************************
Public Sub AktualizovatDataRewatchu()
    On Error GoTo ErrorHandler

    Dim wsAnimeList As Worksheet
    Dim wsInfo As Worksheet
    Dim lastRow As Long
    Dim arrData As Variant
    Dim dictComments As Object
    
    Set wsAnimeList = ThisWorkbook.Sheets("ANIME LIST")
    Set wsInfo = ThisWorkbook.Sheets("OBECNÉ INFORMACE")

    ' ——— OPTIMALIZACE PROSTŘEDÍ ———
    With Application
        .ScreenUpdating = False
        .Calculation = xlCalculationManual
        .EnableEvents = False
    End With

    ' 1. Načtení dat
    lastRow = wsAnimeList.Cells(wsAnimeList.Rows.count, "C").End(xlUp).row
    If lastRow < 2 Then GoTo CleanExit
    
    arrData = wsAnimeList.Range("C2:O" & lastRow).Value
    Set dictComments = GetCommentsDictionary(wsAnimeList, "N")
    
    Dim dictCommentsI As Object
    Set dictCommentsI = GetCommentsDictionary(wsAnimeList, "I")
    
    Dim dictCommentsJ As Object
    Set dictCommentsJ = GetCommentsDictionary(wsAnimeList, "J")

    ' 2. Spuštění NOVÉ hlavní výpočetní procedury (Single Pass)
    Call ZpracovatKompletniData(arrData, dictComments, dictCommentsI, dictCommentsJ, wsInfo)

CleanExit:
    With Application
        .ScreenUpdating = True
        .Calculation = xlCalculationAutomatic
        .EnableEvents = True
    End With
    Exit Sub

ErrorHandler:
    MsgBox "Chyba: " & err.Description, vbCritical
    Resume CleanExit
End Sub

' ********************************************************************************
' NOVÁ HLAVNÍ PROCEDURA: JEDNOTNÉ ZPRACOVÁNÍ (SINGLE PASS OPTIMIZATION)
' ********************************************************************************
' Tato procedura nahrazuje původní "ZpracovatSeznamyRewatchu" a "ZpracovatCasSledovani".
' Prochází data pouze jednou a počítá vše najednou.
Private Sub ZpracovatKompletniData(dataArr As Variant, commentsDict As Object, commentsDictI As Object, commentsDictJ As Object, wsOut As Worksheet)
    
    ' --- POLE A PROMĚNNÉ PRO ČAS A EPIZODY ---
    Dim totalMinCelkem As Double, totalEpCelkem As Long
    Dim rwMinCelkem As Double, rwEpCelkem As Long
    Dim totalMin() As Double, totalEp() As Long
    Dim rwMin() As Double, rwEp() As Long
    
    ReDim totalMin(ROK_START To ROK_KONEC)
    ReDim totalEp(ROK_START To ROK_KONEC)
    ReDim rwMin(ROK_START To ROK_KONEC)
    ReDim rwEp(ROK_START To ROK_KONEC)
    
    ' --- POLE A PROMĚNNÉ PRO SEZNAMY ---
    Dim rwCountCelkem As Long
    Dim rwListCelkem As Object ' ArrayList
    Dim rwCounts() As Long
    Dim rwLists() As Object ' Pole ArrayListů
    
    ReDim rwCounts(ROK_START To ROK_KONEC)
    ReDim rwLists(ROK_START To ROK_KONEC)
    Set rwListCelkem = CreateObject("System.Collections.ArrayList")
    rwCountCelkem = 0
    
    Dim y As Long
    For y = ROK_START To ROK_KONEC
        Set rwLists(y) = CreateObject("System.Collections.ArrayList")
        rwCounts(y) = 0
        totalMin(y) = 0: totalEp(y) = 0: rwMin(y) = 0: rwEp(y) = 0
    Next y

    ' --- PROMĚNNÉ PRO SMYČKU ---
    Dim i As Long
    Dim animeName As String, commentText As String
    Dim epsPerWatch As Long, singleWatchMins As Double
    Dim firstWatchDate As Date
    Dim yr As Integer
    
    ' Regex pro parsování komentářů sloupce I (EP: X-Y; dd.mm.yyyy)
    Dim regEpI As Object
    Set regEpI = CreateObject("VBScript.RegExp")
    regEpI.Pattern = "EP:\s*(\d+)\s*(?:-\s*(\d+))?;\s*(\d{1,2}\.\d{1,2}\.\d{4})"
    regEpI.Global = True
    Dim usedCommentI As Boolean
    Dim epLinesI As Variant, eLineI As Variant
    Dim eMatchesI As Object
    Dim epSI As Long, epEI As Long, epDI As Date, epMinsI As Double

    ' ==============================================================================
    ' HLAVNÍ SMYČKA - JEDEN PRŮCHOD VŠEMI DATY
    ' ==============================================================================
    For i = 1 To UBound(dataArr, 1)
        animeName = dataArr(i, COL_IDX_NAME)
        
        Dim epOverridesA As Object: Set epOverridesA = Nothing
        If commentsDictJ.Exists(i + 1) Then Set epOverridesA = ParseEpDurationsFromComment(commentsDictJ(i + 1))
        
        Dim epBaseMins As Double: epBaseMins = 0
        If IsNumeric(dataArr(i, COL_IDX_MINS)) Then epBaseMins = CDbl(dataArr(i, COL_IDX_MINS))
        
        ' A) Základní validace epizod a času
        If IsNumeric(dataArr(i, COL_IDX_EPS)) And epBaseMins > 0 Then
            epsPerWatch = dataArr(i, COL_IDX_EPS)
            singleWatchMins = CalcEpMinutes(1, epsPerWatch, epOverridesA, epBaseMins, epsPerWatch, epsPerWatch * epBaseMins)
        Else
            epsPerWatch = 0: singleWatchMins = 0
        End If
        
        ' B) Zpracování PRVNÍHO ZHLÉDNUTÍ (First Watch)
        If IsDate(dataArr(i, COL_IDX_START)) Then
            firstWatchDate = dataArr(i, COL_IDX_START)
            
            If firstWatchDate >= DATUM_POCATKU_MERENI Then
                ' Celkové součty
                If IsNumeric(dataArr(i, COL_IDX_TOTAL)) Then
                    totalMinCelkem = totalMinCelkem + dataArr(i, COL_IDX_TOTAL)
                Else
                    totalMinCelkem = totalMinCelkem + singleWatchMins
                End If
                
                If IsNumeric(dataArr(i, COL_IDX_REWATCH)) Then
                    totalEpCelkem = totalEpCelkem + (1 + dataArr(i, COL_IDX_REWATCH)) * epsPerWatch
                Else
                    totalEpCelkem = totalEpCelkem + epsPerWatch
                End If
                
                ' Roční součty (First watch) — přes komentáře sloupce I pokud existují
                usedCommentI = False
                If commentsDictI.Exists(i + 1) And epsPerWatch > 0 Then
                    epLinesI = Split(commentsDictI(i + 1), Chr$(10))
                    For Each eLineI In epLinesI
                        Set eMatchesI = regEpI.Execute(CStr(eLineI))
                        If eMatchesI.count > 0 Then
                            epSI = CLng(eMatchesI(0).SubMatches(0))
                            If eMatchesI(0).SubMatches(1) <> "" Then epEI = CLng(eMatchesI(0).SubMatches(1)) Else epEI = epSI
                            On Error Resume Next
                            epDI = CDate(eMatchesI(0).SubMatches(2))
                            On Error GoTo 0
                            
                            epMinsI = CalcEpMinutes(epSI, epEI, epOverridesA, epBaseMins, epsPerWatch, epsPerWatch * epBaseMins)
                            yr = year(epDI)
                            If yr >= ROK_START And yr <= ROK_KONEC Then
                                totalMin(yr) = totalMin(yr) + epMinsI
                                totalEp(yr) = totalEp(yr) + (epEI - epSI + 1)
                            End If
                            usedCommentI = True
                        End If
                    Next eLineI
                End If
                
                ' Fallback: bez komentářů › celý čas do roku data začátku
                If Not usedCommentI Then
                    yr = year(firstWatchDate)
                    If yr >= ROK_START And yr <= ROK_KONEC Then
                        totalMin(yr) = totalMin(yr) + singleWatchMins
                        totalEp(yr) = totalEp(yr) + epsPerWatch
                    End If
                End If
            End If
        End If
        
        ' C) Zpracování REWATCHŮ (Pokud existují)
        If IsNumeric(dataArr(i, COL_IDX_REWATCH)) And dataArr(i, COL_IDX_REWATCH) > 0 Then
            If commentsDict.Exists(i + 1) Then
                commentText = commentsDict(i + 1)
                
                ' >>> VOLÁNÍ NOVÉHO MASTER PARSERU <<<
                ' Předáváme mu veškerá pole odkazem (ByRef), aby je naplnil najednou.
                Call MasterParsujKomentar(commentText, animeName, epsPerWatch, singleWatchMins, epBaseMins, epOverridesA, _
                                          rwLists, rwCounts, rwListCelkem, rwCountCelkem, _
                                          totalMin, totalEp, rwMin, rwEp, rwMinCelkem, rwEpCelkem)
            End If
        End If
    Next i
    
    ' ==============================================================================
    ' ZÁPIS VÝSLEDKŮ (Kombinace zápisů z obou původních procedur)
    ' ==============================================================================
    
    ' 1. Zápis CELKOVÝCH dat (Sloupec F = 6)
    With wsOut
        ' Čas a Epizody
        .Cells(3, 6).Value = totalMinCelkem / 1440
        .Cells(4, 6).Value = FormatujCasCesky(totalMinCelkem / 1440)
        Call AddRewatchTimeComment(.Cells(3, 6), rwMinCelkem, totalMinCelkem)
        
        .Cells(5, 6).Value = totalEpCelkem
        Call AddRewatchEpisodesComment(.Cells(5, 6), rwEpCelkem, totalEpCelkem)
        
        If totalEpCelkem > 0 Then .Cells(6, 6).Value = totalMinCelkem / totalEpCelkem Else .Cells(6, 6).Value = 0
        
        ' Seznamy rewatchů
        .Cells(7, 6).Value = rwCountCelkem
        Call CreateAndAddRewatchComment(.Cells(7, 6), rwListCelkem)
    End With
    
    ' 2. Zápis ROČNÍCH dat (Dynamicky G, H, I...)
    Dim targetCol As Integer
    For y = ROK_START To ROK_KONEC
        targetCol = COL_START_INDEX + (y - ROK_START)
        
        With wsOut
            ' Čas a Epizody
            .Cells(3, targetCol).Value = totalMin(y) / 1440
            .Cells(4, targetCol).Value = FormatujCasCesky(totalMin(y) / 1440)
            Call AddRewatchTimeComment(.Cells(3, targetCol), rwMin(y), totalMin(y))
            
            .Cells(5, targetCol).Value = totalEp(y)
            Call AddRewatchEpisodesComment(.Cells(5, targetCol), rwEp(y), totalEp(y))
            
            If totalEp(y) > 0 Then .Cells(6, targetCol).Value = totalMin(y) / totalEp(y) Else .Cells(6, targetCol).Value = 0
            
            ' Seznamy rewatchů
            .Cells(7, targetCol).Value = rwCounts(y)
            Call CreateAndAddRewatchComment(.Cells(7, targetCol), rwLists(y))
        End With
    Next y
    
    ' 3. Finální formátování
    Dim endCol As Integer: endCol = COL_START_INDEX + (ROK_KONEC - ROK_START)
    wsOut.Range(wsOut.Cells(3, 6), wsOut.Cells(3, endCol)).NumberFormat = "[h]:mm"
    wsOut.Range(wsOut.Cells(5, 6), wsOut.Cells(5, endCol)).NumberFormat = "0"
    wsOut.Range(wsOut.Cells(6, 6), wsOut.Cells(6, endCol)).NumberFormat = "0.0"
    wsOut.Range(wsOut.Cells(7, 6), wsOut.Cells(7, endCol)).NumberFormat = "0"
End Sub

' ********************************************************************************
' NOVÝ MASTER PARSER (Srdce optimalizace)
' ********************************************************************************
' Tato procedura zpracuje text komentáře jen jednou a vytáhne z něj data
' jak pro seznamy (kdy a co), tak pro statistiky (kolik minut a epizod).
Private Sub MasterParsujKomentar(txt As String, aName As String, epsPerWatch As Long, singleMins As Double, epBaseMins As Double, epOverridesA As Object, _
                                 ByRef rwLists() As Object, ByRef rwCounts() As Long, _
                                 ByRef listAll As Object, ByRef cntAll As Long, _
                                 ByRef totMin() As Double, ByRef totEp() As Long, _
                                 ByRef rwMin() As Double, ByRef rwEp() As Long, _
                                 ByRef rwMinAll As Double, ByRef rwEpAll As Long)

    Dim lines() As String
    Dim rwNum As String, dateStrForList As String
    Dim rwDate As Date
    Dim yr As Integer
    Dim iLine As Long, entry As String
    Dim posDot As Long, posEps As Long
    
    lines = Split(txt, Chr$(10))
    
    For iLine = 0 To UBound(lines)
        entry = lines(iLine)
        rwDate = #12:00:00 AM#
        rwNum = ""
        dateStrForList = ""
        
        On Error Resume Next ' Bezpečnost pro parsování
        
        ' --- TYP 1: ČÁSTEČNÝ REWATCH (Partial: "; EPs:") ---
        posEps = InStr(1, entry, "; EPs:", vbTextCompare)
        If posEps > 0 Then
            ' A) Získání dat pro SEZNAM
            posDot = InStr(1, entry, ".")
            If posDot > 0 Then
                rwNum = Trim$(left$(entry, posDot - 1))
                dateStrForList = Trim$(Mid$(entry, posDot + 1))
            Else
                rwNum = "999"
                dateStrForList = Trim$(entry)
            End If
            
            Dim posEpsInner As Long: posEpsInner = InStr(1, dateStrForList, "; EPs:", vbTextCompare)
            If posEpsInner > 0 Then dateStrForList = Trim$(left$(dateStrForList, posEpsInner - 1))
            rwDate = ParseDateString(dateStrForList) ' Zkusíme parsovat datum pro určení roku
            
            Dim formattedDate As String: formattedDate = FormatRewatchDate(dateStrForList)
            
            ' Zápis do globálního seznamu
            cntAll = cntAll + 1
            listAll.Add Array(CInt(rwNum), aName, formattedDate)
            
            ' Zápis do ročního seznamu (pokud máme datum)
            If rwDate <> #12:00:00 AM# And rwDate >= DATUM_POCATKU_MERENI Then
                yr = year(rwDate)
                If yr >= ROK_START And yr <= ROK_KONEC Then
                    rwCounts(yr) = rwCounts(yr) + 1
                    rwLists(yr).Add Array(CInt(rwNum), aName, formattedDate)
                End If
            End If
            
            ' B) Výpočet ČASU (Detailní rozpad po dnech)
            Dim epArr() As String
            Dim rwStart As Date, rwEnd As Date, curDate As Date
            Dim numDays As Long, k As Long
            Dim epsDay As Long, minDay As Double
            Dim datePartRaw As String, epPartRaw As String
            Dim posDash As Long
            
            datePartRaw = Trim$(Mid$(entry, posDot + 1, posEps - posDot - 1))
            epPartRaw = Trim$(Mid$(entry, posEps + 6))
            
            epArr = Split(epPartRaw, ",")
            
            posDash = InStr(1, datePartRaw, "-")
            If posDash > 0 Then
                rwStart = CDate(Trim$(left$(datePartRaw, posDash - 1)))
                rwEnd = CDate(Trim$(Mid$(datePartRaw, posDash + 1)))
            Else
                rwStart = CDate(datePartRaw): rwEnd = rwStart
            End If
            
            numDays = CLng(rwEnd) - CLng(rwStart) + 1
            
            If UBound(epArr) + 1 = numDays Then
                Dim rwEpCtr As Long: rwEpCtr = 1
                For k = 0 To numDays - 1
                    curDate = rwStart + k
                    yr = year(curDate)
                    epsDay = CLng(Trim$(epArr(k)))
                    minDay = CalcEpMinutes(rwEpCtr, rwEpCtr + epsDay - 1, epOverridesA, epBaseMins, epsPerWatch, epsPerWatch * epBaseMins)
                    rwEpCtr = rwEpCtr + epsDay
                    
                    If minDay > 0 Then
                        rwMinAll = rwMinAll + minDay: rwEpAll = rwEpAll + epsDay
                        If yr >= ROK_START And yr <= ROK_KONEC Then
                            totMin(yr) = totMin(yr) + minDay: totEp(yr) = totEp(yr) + epsDay
                            rwMin(yr) = rwMin(yr) + minDay: rwEp(yr) = rwEp(yr) + epsDay
                        End If
                    End If
                Next k
            End If
            
        ' --- TYP 2: NEURČITO ("Netuším přesně") ---
        ElseIf InStr(1, entry, "Netuším přesně", vbTextCompare) > 0 Then
            ' A) SEZNAM
            posDot = InStr(1, entry, ".")
            If posDot > 0 Then
                If IsNumeric(Trim$(left$(entry, posDot - 1))) Then
                    rwNum = Trim$(left$(entry, posDot - 1))
                Else
                    rwNum = "999"
                End If
            Else
                rwNum = "999"
            End If
            
            cntAll = cntAll + 1
            listAll.Add Array(CLng(rwNum), aName, "(netuším přesně)")
            
            ' B) ČAS (Jen celkový, ne roční)
            rwMinAll = rwMinAll + singleMins: rwEpAll = rwEpAll + epsPerWatch

        ' --- TYP 3: PLNÝ REWATCH S DATEM ---
        ElseIf InStr(1, entry, ".") > 0 Then
            posDot = InStr(1, entry, ".")
            ' A) Získání dat pro SEZNAM a ROK
            rwNum = Trim$(left$(entry, posDot - 1))
            dateStrForList = Trim$(Mid$(entry, posDot + 1))
            rwDate = ParseDateString(dateStrForList)
            
            Dim fDate3 As String: fDate3 = FormatRewatchDate(dateStrForList)
            
            If rwDate <> #12:00:00 AM# And rwDate >= DATUM_POCATKU_MERENI Then
                yr = year(rwDate)
                
                ' Zápis do seznamů
                cntAll = cntAll + 1
                listAll.Add Array(CInt(rwNum), aName, fDate3)
                If yr >= ROK_START And yr <= ROK_KONEC Then
                    rwCounts(yr) = rwCounts(yr) + 1
                    rwLists(yr).Add Array(CInt(rwNum), aName, fDate3)
                End If
                
                ' B) Zápis ČASU
                rwMinAll = rwMinAll + singleMins: rwEpAll = rwEpAll + epsPerWatch
                If yr >= ROK_START And yr <= ROK_KONEC Then
                    totMin(yr) = totMin(yr) + singleMins: totEp(yr) = totEp(yr) + epsPerWatch
                    rwMin(yr) = rwMin(yr) + singleMins: rwEp(yr) = rwEp(yr) + epsPerWatch
                End If
            End If
        End If
        On Error GoTo 0
    Next iLine
End Sub


' ********************************************************************************
' POMOCNÉ FUNKCE (ZŮSTÁVAJÍ BEZ ZMĚNY)
' ********************************************************************************
Function ParseDateString(dateStr As String) As Date
    On Error GoTo ErrDate
    Dim cleanStr As String: cleanStr = Trim$(dateStr)
    
    Dim pDash As Long: pDash = InStr(1, cleanStr, "-")
    If pDash > 0 Then cleanStr = Trim$(left$(cleanStr, pDash - 1))
    
    Dim pSpace As Long: pSpace = InStr(1, cleanStr, " ")
    
    If pSpace > 0 Then
        Dim sPart0 As String: sPart0 = Trim$(left$(cleanStr, pSpace - 1))
        Dim sPart1 As String: sPart1 = Trim$(Mid$(cleanStr, pSpace + 1))
        
        If InStr(1, sPart0, ".") > 0 Then
            ParseDateString = CDate(cleanStr)
        Else
            Dim mNum As Integer: mNum = ConvertCzechMonthToNumber(sPart0)
            If mNum > 0 And IsNumeric(sPart1) Then ParseDateString = DateSerial(CInt(sPart1), mNum, 1)
        End If
    Else
        ParseDateString = CDate(cleanStr)
    End If
    Exit Function
ErrDate:
    ParseDateString = #12:00:00 AM#
End Function

Function GetCommentsDictionary(ws As Worksheet, colLetter As String) As Object
    Dim dict As Object: Set dict = CreateObject("Scripting.Dictionary")
    Dim rngComments As Range
    On Error Resume Next
    Set rngComments = ws.Columns(colLetter).SpecialCells(xlCellTypeComments)
    On Error GoTo 0
    If Not rngComments Is Nothing Then
        Dim cell As Range
        For Each cell In rngComments
            dict.Add cell.row, cell.Comment.text
        Next cell
    End If
    Set GetCommentsDictionary = dict
End Function

Function ConvertCzechMonthToNumber(monthName As String) As Integer
    Select Case LCase(monthName)
        Case "leden": ConvertCzechMonthToNumber = 1: Case "únor": ConvertCzechMonthToNumber = 2
        Case "březen": ConvertCzechMonthToNumber = 3: Case "duben": ConvertCzechMonthToNumber = 4
        Case "květen": ConvertCzechMonthToNumber = 5: Case "červen": ConvertCzechMonthToNumber = 6
        Case "červenec": ConvertCzechMonthToNumber = 7: Case "srpen": ConvertCzechMonthToNumber = 8
        Case "září": ConvertCzechMonthToNumber = 9: Case "říjen": ConvertCzechMonthToNumber = 10
        Case "listopad": ConvertCzechMonthToNumber = 11: Case "prosinec": ConvertCzechMonthToNumber = 12
        Case Else: ConvertCzechMonthToNumber = 0
    End Select
End Function

Private Function HHmmFromMinutes(ByVal minutesVal As Double) As String
    Dim m As Long: m = CLng(Application.Max(0, Round(minutesVal, 0)))
    HHmmFromMinutes = (m \ 60) & ":" & Format(m Mod 60, "00")
End Function

Sub AddRewatchTimeComment(targetCell As Range, rewatchMinutes As Double, Optional totalMinutes As Double = -1)
    If Not targetCell.Comment Is Nothing Then targetCell.Comment.Delete
    If totalMinutes < 0 Then totalMinutes = CDbl(targetCell.Value) * 1440#
    Dim s1prefix As String: s1prefix = "Z toho rewatchnuto: "
    Dim s2prefix As String: s2prefix = "Čas sledování bez rewatchů: "
    Dim s1val As String: s1val = HHmmFromMinutes(rewatchMinutes)
    Dim s2val As String: s2val = HHmmFromMinutes(Application.Max(0, totalMinutes - rewatchMinutes))
    Dim line1 As String: line1 = s1prefix & s1val & " (hh:mm)"
    Dim line2 As String: line2 = s2prefix & s2val & " (hh:mm)"
    targetCell.AddComment line1 & Chr(10) & line2
    With targetCell.Comment.Shape.TextFrame
        .Characters(Start:=Len(s1prefix) + 1, length:=Len(s1val) + 1).Font.Bold = True
        .Characters(Start:=Len(line1) + 1 + Len(s2prefix), length:=Len(s2val) + 1).Font.Bold = True
        .AutoSize = True
    End With
    targetCell.Comment.Visible = False
End Sub

Sub AddRewatchEpisodesComment(targetCell As Range, rewatchEpisodes As Long, Optional totalEpisodes As Long = -1)
    If Not targetCell.Comment Is Nothing Then targetCell.Comment.Delete
    If totalEpisodes < 0 Then totalEpisodes = 0
    Dim s1prefix As String: s1prefix = "Z toho rewatchnuto: "
    Dim s2prefix As String: s2prefix = "Počet zhlédnutých originálních epizod: "
    Dim s1val As String: s1val = CStr(Application.Max(0, rewatchEpisodes)) & " epizod"
    Dim s2val As String: s2val = CStr(Application.Max(0, totalEpisodes - rewatchEpisodes)) & " epizod"
    Dim line1 As String: line1 = s1prefix & s1val
    Dim line2 As String: line2 = s2prefix & s2val
    targetCell.AddComment line1 & Chr(10) & line2
    With targetCell.Comment.Shape.TextFrame
        .Characters(Start:=Len(s1prefix) + 1, length:=Len(s1val) + 1).Font.Bold = True
        .Characters(Start:=Len(line1) + 1 + Len(s2prefix), length:=Len(s2val) + 1).Font.Bold = True
        .AutoSize = True
    End With
    targetCell.Comment.Visible = False
End Sub

Sub CreateAndAddRewatchComment(targetCell As Range, dataList As Object)
    If Not targetCell.Comment Is Nothing Then targetCell.Comment.Delete
    If dataList Is Nothing Or dataList.count = 0 Then Exit Sub
    
    Dim listArray() As Variant: listArray = dataList.ToArray()
    QuickSort listArray, LBound(listArray), UBound(listArray)
    
    Dim i As Long
    Dim count As Long: count = UBound(listArray) - LBound(listArray) + 1
    ReDim linesArray(0 To count - 1) As String
    ReDim dictBoldPos(0 To count - 1) As Long
    ReDim dictBoldLen(0 To count - 1) As Long
    
    Dim currentPos As Long: currentPos = 1
    Dim dateLen As Long, fullLine As String
    Dim dtStr As String
    
    For i = 0 To count - 1
        dtStr = CStr(listArray(i + LBound(listArray))(2))
        fullLine = listArray(i + LBound(listArray))(0) & ". Rewatch; " & listArray(i + LBound(listArray))(1) & " " & dtStr
        linesArray(i) = fullLine
        
        dateLen = Len(dtStr)
        dictBoldPos(i) = currentPos + Len(fullLine) - dateLen
        dictBoldLen(i) = dateLen
        
        currentPos = currentPos + Len(fullLine) + 1 ' +1 for Chr(10)
    Next i
    
    targetCell.AddComment Join(linesArray, Chr(10))
    targetCell.Comment.Shape.TextFrame.AutoSize = True
    
    For i = 0 To count - 1
        targetCell.Comment.Shape.TextFrame.Characters(dictBoldPos(i), dictBoldLen(i)).Font.Bold = True
    Next i
    
    targetCell.Comment.Visible = False
End Sub

Function FormatRewatchDate(rawDateString As String) As String
    On Error GoTo DateParseError
    Dim trimmedStr As String
    Dim pEps As Long
    
    ' 1. Očištění o "EPs" část (pokud existuje)
    pEps = InStr(1, rawDateString, "; EPs:", vbTextCompare)
    If pEps > 0 Then
        trimmedStr = Trim$(left$(rawDateString, pEps - 1))
    Else
        trimmedStr = Trim$(rawDateString)
    End If

    ' 2. Zpracování rozsahů (pokud obsahuje pomlčku)
    Dim pDash As Long: pDash = InStr(1, trimmedStr, "-")
    If pDash > 0 Then
        Dim sStart As String: sStart = Trim$(left$(trimmedStr, pDash - 1))
        Dim sEnd As String: sEnd = Trim$(Mid$(trimmedStr, pDash + 1))
        
        ' Konec je obvykle plné datum, to je náš kotevní bod
        Dim dEnd As Date
        dEnd = CDate(sEnd)
        
        Dim dStart As Date
        ' A) Start je plné datum (např. 31.12.2025)
        If IsDate(sStart) Then
            dStart = CDate(sStart)
        Else
            ' B) Start je jen číslo dne (např. "22" nebo "22.")
            Dim cleanStart As String: cleanStart = Replace$(sStart, ".", "")
            If IsNumeric(cleanStart) Then
                ' Dopočítáme měsíc a rok podle koncového data
                dStart = DateSerial(year(dEnd), Month(dEnd), CInt(cleanStart))
            Else
                ' C) Start je den a měsíc bez roku (např. "31.12.")
                ' Zkusíme to složit s rokem koncového data, nebo nechat Excel rozhodnout
                On Error Resume Next
                dStart = CDate(sStart & year(dEnd))
                If err.Number <> 0 Then dStart = CDate(sStart)
                On Error GoTo DateParseError
            End If
        End If
        
        ' 3. Porovnání a formátování (Možnost B)
        If year(dStart) = year(dEnd) And Month(dStart) = Month(dEnd) Then
            ' Stejný měsíc a rok -> zkrácený formát: "22. - 23.03.2025"
            FormatRewatchDate = "(" & Format$(dStart, "dd") & ". - " & Format$(dEnd, "dd.mm.yyyy") & ")"
        Else
            ' Různý měsíc nebo rok -> plný formát: "31.12.2025 - 01.01.2026"
            FormatRewatchDate = "(" & Format$(dStart, "dd.mm.yyyy") & " - " & Format$(dEnd, "dd.mm.yyyy") & ")"
        End If

    ' 4. Ostatní případy (bez pomlčky)
    ElseIf InStr(1, trimmedStr, "Netuším", vbTextCompare) > 0 Then
        FormatRewatchDate = "(netuším přesně)"
    Else
        ' Jednoduché datum
        FormatRewatchDate = "(" & Format$(CDate(trimmedStr), "dd.mm.yyyy") & ")"
    End If
    
    Exit Function

DateParseError:
    ' Záchranná brzda: vrátí původní text v závorkách, pokud selže parsování
    FormatRewatchDate = "(" & trimmedStr & ")"
End Function

Private Sub QuickSort(ByRef arr As Variant, ByVal First As Long, ByVal Last As Long)
    Dim pivot As Variant, Temp As Variant, i As Long, j As Long
    If First >= Last Then Exit Sub
    i = First: j = Last: pivot = arr((First + Last) \ 2)
    
    Dim p0 As Long: p0 = pivot(0)
    Dim p1U As String: p1U = UCase(pivot(1))
    
    Do
        Do While (arr(i)(0) < p0) Or (arr(i)(0) = p0 And UCase(arr(i)(1)) < p1U): i = i + 1: Loop
        Do While (arr(j)(0) > p0) Or (arr(j)(0) = p0 And UCase(arr(j)(1)) > p1U): j = j - 1: Loop
        If i <= j Then
            Temp = arr(i): arr(i) = arr(j): arr(j) = Temp
            i = i + 1: j = j - 1
        End If
    Loop While i <= j
    
    If First < j Then QuickSort arr, First, j
    If i < Last Then QuickSort arr, i, Last
End Sub

Function FormatujCasCesky(ExcelTimeValue As Double) As String
    Dim totalMinutes As Long, days As Long, hours As Long, result As String
    totalMinutes = Round(ExcelTimeValue * 1440, 0)
    days = Int(totalMinutes / 1440)
    hours = Int((totalMinutes Mod 1440) / 60)
    Select Case days
        Case 0: result = ""
        Case 1: result = "1 den"
        Case 2 To 4: result = days & " dny"
        Case Else: result = days & " dní"
    End Select
    If days > 0 And hours > 0 Then result = result & " "
    Select Case hours
        Case 0: If days = 0 Then result = "0 hodin"
        Case 1: result = result & "1 hodina"
        Case 2 To 4: result = result & hours & " hodiny"
        Case Else: result = result & hours & " hodin"
    End Select
    FormatujCasCesky = result
End Function

Private Function ParseEpDurationsFromComment(ByVal cmtText As String) As Object
    Dim dict As Object: Set dict = CreateObject("Scripting.Dictionary")
    If Len(cmtText) = 0 Then
        Set ParseEpDurationsFromComment = Nothing
        Exit Function
    End If
    
    Dim reg As Object: Set reg = CreateObject("VBScript.RegExp")
    reg.Pattern = "(\d+)[\.\s]*EP:\s*(\d+):(\d+)"
    reg.Global = True
    reg.IgnoreCase = True
    
    Dim lines() As String: lines = Split(cmtText, vbLf)
    Dim vLine As Variant, matches As Object
    Dim epNum As Long, mins As Double, secs As Double
    
    For Each vLine In lines
        Set matches = reg.Execute(vLine)
        If matches.count > 0 Then
            epNum = CLng(matches(0).SubMatches(0))
            mins = CDbl(matches(0).SubMatches(1))
            secs = CDbl(matches(0).SubMatches(2))
            dict.Add epNum, mins + (secs / 60)
        End If
    Next vLine
    
    If dict.count > 0 Then Set ParseEpDurationsFromComment = dict Else Set ParseEpDurationsFromComment = Nothing
End Function

Private Function CalcEpMinutes(ByVal firstEp As Long, ByVal lastEp As Long, _
                               ByVal epOverrides As Object, _
                               ByVal defaultDuration As Double, _
                               ByVal totalEpisodes As Long, _
                               ByVal totalTime As Double) As Double
    If epOverrides Is Nothing Then
        CalcEpMinutes = (lastEp - firstEp + 1) * defaultDuration
        Exit Function
    End If

    Dim sumOverrides As Double: sumOverrides = 0
    Dim countOverrides As Long: countOverrides = epOverrides.count
    Dim key As Variant
    For Each key In epOverrides.keys
        sumOverrides = sumOverrides + epOverrides(key)
    Next key

    Dim avgRemaining As Double
    If totalEpisodes > countOverrides Then
        avgRemaining = (totalTime - sumOverrides) / (totalEpisodes - countOverrides)
        If avgRemaining < 0 Then avgRemaining = defaultDuration
    Else
        avgRemaining = defaultDuration
    End If

    Dim totalMins As Double: totalMins = 0
    Dim j As Long
    For j = firstEp To lastEp
        If epOverrides.Exists(j) Then
            totalMins = totalMins + epOverrides(j)
        Else
            totalMins = totalMins + avgRemaining
        End If
    Next j

    CalcEpMinutes = totalMins
End Function
