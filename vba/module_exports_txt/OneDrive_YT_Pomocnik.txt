Option Explicit

'====================================================================================================
' MODUL: Poloautomatický pomocník pro hledání odkazů (YouTube + OneDrive Smart)
' Popis: Makro identifikuje písničky bez odkazu, seřadí je podle názvu Anime (sloupec I)
'        a umožňuje vložit buď přímý OneDrive odkaz (který automaticky upraví),
'        nebo vyhledat video na YouTube.
' Autor: Gemini
' Verze: 1.5 (Integrace Smart OneDrive logiky + oprava sloupců po posunu)
'====================================================================================================

Public Sub ZpracujYouTubeFrontu()
    Dim ws As Worksheet
    Dim radek As Long, i As Long, j As Long
    Dim posledniRadek As Long
    ' Pole ukládá: 1=Název Anime (I), 2=Název Písničky (K), 3=Číslo řádku
    Dim frontaArr() As Variant
    Dim pocetChybicich As Long
    Dim bunka As Range, bunkaTypu As Range
    
    Dim animeNazev As String, typVidea As String, nazevPisnicky As String, nazevAutora As String
    Dim dotaz As String, youtubeURL As String, vlozenyOdkaz As String
    Dim formulaText As String
    Dim odpoved As VbMsgBoxResult

    ' --- Krok 1: Nastavení a příprava ---
    On Error GoTo ChybaProcedury
    Application.ScreenUpdating = False
    
    Set ws = ThisWorkbook.Sheets("ANIME FAV OP + ED + OST")
    
    ' --- Krok 2: Sestavení fronty ---
    ' Hledáme poslední řádek ve sloupci K (Písničky)
    posledniRadek = ws.Cells(ws.Rows.count, "K").End(xlUp).row
    pocetChybicich = 0
    
    ' Spočítáme chybějící
    For radek = 2 To posledniRadek
        If Not ws.Cells(radek, "K").Formula Like "=*HYPERLINK(*" And Not ws.Cells(radek, "K").Formula Like "=*HYPERTEXTOVÝ.ODKAZ(*" Then
            If ws.Cells(radek, "K").Value <> "" Then
                pocetChybicich = pocetChybicich + 1
            End If
        End If
    Next radek
    
    If pocetChybicich = 0 Then
        MsgBox "Všechny písničky již mají přiřazený odkaz. Není co zpracovávat.", vbInformation, "Hotovo"
        Exit Sub
    End If
    
    ReDim frontaArr(1 To pocetChybicich, 1 To 3)
    
    j = 1
    For radek = 2 To posledniRadek
        If Not ws.Cells(radek, "K").Formula Like "=*HYPERLINK(*" And Not ws.Cells(radek, "K").Formula Like "=*HYPERTEXTOVÝ.ODKAZ(*" Then
            If ws.Cells(radek, "K").Value <> "" Then
                ' Sloupec I = Název Anime (pro třídění)
                frontaArr(j, 1) = ws.Cells(radek, "I").Value
                ' Sloupec K = Název Písničky
                frontaArr(j, 2) = ws.Cells(radek, "K").Value
                ' Číslo řádku
                frontaArr(j, 3) = radek
                j = j + 1
            End If
        End If
    Next radek

    ' --- Třídění pole podle Názvu Anime (Bubble Sort) ---
    Dim tempAnime As String, tempSong As String, tempRow As Long
    For i = 1 To pocetChybicich - 1
        For j = i + 1 To pocetChybicich
            If UCase(frontaArr(i, 1)) > UCase(frontaArr(j, 1)) Then
                tempAnime = frontaArr(j, 1): frontaArr(j, 1) = frontaArr(i, 1): frontaArr(i, 1) = tempAnime
                tempSong = frontaArr(j, 2): frontaArr(j, 2) = frontaArr(i, 2): frontaArr(i, 2) = tempSong
                tempRow = frontaArr(j, 3): frontaArr(j, 3) = frontaArr(i, 3): frontaArr(i, 3) = tempRow
            End If
        Next j
    Next i

    ' --- Krok 3: Úvodní dialog ---
    odpoved = MsgBox("Nalezeno " & pocetChybicich & " OP/ED bez odkazu." & vbCrLf & vbCrLf & _
                     "Fronta je seřazena podle NÁZVU ANIME (A-Z)." & vbCrLf & _
                     "Chcete pokračovat?", vbYesNo + vbQuestion, "Start")
    If odpoved = vbNo Then Exit Sub
    
    ' --- Krok 4: Hlavní smyčka ---
    For i = 1 To pocetChybicich
        radek = frontaArr(i, 3)
        Set bunka = ws.Cells(radek, "K") ' Písnička je v K
        
        animeNazev = ws.Cells(radek, "I").Value   ' Anime je v I
        nazevPisnicky = bunka.Value
        nazevAutora = ws.Cells(radek, "L").Value  ' Autor je v L
        
        ' --- Čtení TYPU videa (Předpoklad: Sloupec J, mezi I a K) ---
        ' Původně C, posun +7 -> J. Pokud to nesedí, změň "J" níže.
        Set bunkaTypu = ws.Cells(radek, "J")
        
        If Not bunkaTypu.Comment Is Nothing Then
            Dim commentText As String
            commentText = bunkaTypu.Comment.text
            Dim poziceZalomeni As Long
            poziceZalomeni = InStr(commentText, vbLf)
            If poziceZalomeni > 0 Then
                typVidea = Trim(Mid(commentText, poziceZalomeni + 1))
            Else
                typVidea = commentText
            End If
        Else
            typVidea = bunkaTypu.Value
        End If
        
        ' 1. KROK: DOTAZ NA ONEDRIVE
        vlozenyOdkaz = InputBox("Zpracovává se: " & animeNazev & vbCrLf & _
                                "Skladba: " & nazevPisnicky & " (" & typVidea & ")" & vbCrLf & vbCrLf & _
                                "Máš přímý odkaz na OneDrive/SharePoint?" & vbCrLf & _
                                ">>> Pokud NE, nechte prázdné a stiskněte ENTER (otevře se YouTube).", _
                                "Zdroj odkazu (" & (pocetChybicich - i + 1) & " zbývá)")

        ' 2. KROK: LOGIKA ODKAZU
        If vlozenyOdkaz <> "" Then
            ' --- A) Zpracování OneDrive odkazu (Smart Logic) ---
            Dim rawUrl As String
            Dim parts() As String
            Dim basePart As String
            Dim eToken As String
            Dim paramArr() As String
            Dim pIter As Variant
            
            rawUrl = vlozenyOdkaz
            eToken = ""
            
            ' Rozdělení na základ a parametry
            parts = Split(rawUrl, "?")
            basePart = parts(0)
            
            ' Hledání tokenu "e=" v parametrech
            If UBound(parts) > 0 Then
                paramArr = Split(parts(1), "&")
                For Each pIter In paramArr
                    If left(pIter, 2) = "e=" Then
                        eToken = pIter
                        Exit For
                    End If
                Next pIter
            End If
            
            ' Sestavení finálního linku
            If eToken <> "" Then
                ' Pouze základní URL + token (bez download=1)
                vlozenyOdkaz = basePart & "?" & eToken
            Else
                ' Pouze základní URL (bez download=1)
                vlozenyOdkaz = basePart
            End If
            
        Else
            ' --- B) YouTube vyhledávání (pokud byl InputBox prázdný) ---
            dotaz = Replace(animeNazev & " " & typVidea & " " & nazevPisnicky & " " & nazevAutora, " ", "+")
            youtubeURL = "https://www.youtube.com/results?search_query=" & dotaz & "&sp=EgIYAToAMAE%3D"
            ThisWorkbook.FollowHyperlink Address:=youtubeURL
            
            vlozenyOdkaz = InputBox("YouTube hledání pro: " & nazevPisnicky & vbCrLf & vbCrLf & _
                                    "1. Zkopírujte URL videa." & vbCrLf & _
                                    "2. Vložte sem:", "Vložit YouTube odkaz")
        End If

        ' 3. KROK: ZÁPIS DO BUŇKY
        If vlozenyOdkaz <> "" Then
            formulaText = "=HYPERLINK(""" & vlozenyOdkaz & """,""" & Replace(nazevPisnicky, """", "''") & """)"
            bunka.Formula = formulaText
            
            With bunka
                .Font.Bold = True: .Font.Color = vbBlack: .Font.size = 11: .Font.Italic = True
                .Font.Underline = xlUnderlineStyleNone
                .HorizontalAlignment = xlCenter: .VerticalAlignment = xlCenter
            End With
        End If
        
        If (pocetChybicich - i) > 0 Then
            odpoved = MsgBox("Zbývá " & (pocetChybicich - i) & " odkazů." & vbCrLf & vbCrLf & "Pokračovat?", _
                             vbYesNo + vbQuestion, "Další?")
            If odpoved = vbNo Then Exit For
        End If
    Next i
    
    MsgBox "Hotovo.", vbInformation, "Konec"

Konec:
    Application.ScreenUpdating = True
    Exit Sub
ChybaProcedury:
    MsgBox "Chyba: " & vbCrLf & err.Description, vbCritical
    Resume Konec
End Sub
