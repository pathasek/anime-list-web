Function Nz(Expression As Variant, Optional ValueIfNull As Variant) As Variant
    If IsMissing(ValueIfNull) Then ValueIfNull = ""
    If IsNull(Expression) Or IsEmpty(Expression) Then
        Nz = ValueIfNull
    Else
        Nz = Expression
    End If
End Function

Function GetMonthNumber(monthName As String) As Integer
    Select Case LCase(Trim(monthName))
        Case "leden": GetMonthNumber = 1
        Case "únor": GetMonthNumber = 2
        Case "březen": GetMonthNumber = 3
        Case "duben": GetMonthNumber = 4
        Case "květen": GetMonthNumber = 5
        Case "červen": GetMonthNumber = 6
        Case "červenec": GetMonthNumber = 7
        Case "srpen": GetMonthNumber = 8
        Case "září": GetMonthNumber = 9
        Case "říjen": GetMonthNumber = 10
        Case "listopad": GetMonthNumber = 11
        Case "prosinec": GetMonthNumber = 12
        Case Else: GetMonthNumber = 0
    End Select
End Function

Sub QuickSortMonthlyData(ByRef names As Variant, ByRef values As Variant, ByRef dates As Variant, ByVal First As Long, ByVal Last As Long)
    Dim pivotDate As Date, tempName As String, tempValue As Double, tempDate As Date
    Dim i As Long, j As Long
    If First < Last Then
        pivotDate = dates((First + Last) \ 2)
        i = First: j = Last
        Do
            Do While dates(i) < pivotDate: i = i + 1: Loop
            Do While dates(j) > pivotDate: j = j - 1: Loop
            If i <= j Then
                tempName = names(i): names(i) = names(j): names(j) = tempName
                tempValue = values(i): values(i) = values(j): values(j) = tempValue
                tempDate = dates(i): dates(i) = dates(j): dates(j) = tempDate
                i = i + 1: j = j - 1
            End If
        Loop Until i > j
        If First < j Then QuickSortMonthlyData names, values, dates, First, j
        If i < Last Then QuickSortMonthlyData names, values, dates, i, Last
    End If
End Sub

Sub QuickSortArray(arr, ByVal First As Long, ByVal Last As Long, ByVal col As Long)
    Dim pivot As Variant, Temp As Variant, i As Long, j As Long
    If First < Last Then
        pivot = arr((First + Last) \ 2, col)
        i = First: j = Last
        Do
            Do While arr(i, col) < pivot: i = i + 1: Loop
            Do While arr(j, col) > pivot: j = j - 1: Loop
            If i <= j Then
                Temp = arr(i, 1): arr(i, 1) = arr(j, 1): arr(j, 1) = Temp
                Temp = arr(i, 2): arr(i, 2) = arr(j, 2): arr(j, 2) = Temp
                i = i + 1: j = j - 1
            End If
        Loop Until i > j
        If First < j Then QuickSortArray arr, First, j, col
        If i < Last Then QuickSortArray arr, i, Last, col
    End If
End Sub

Function SkloňujDen(pocet As Long) As String
    Select Case pocet
        Case 1: SkloňujDen = "den"
        Case 2, 3, 4: SkloňujDen = "dny"
        Case Else: SkloňujDen = "dní"
    End Select
End Function

Private Sub SortDates(ByRef arr As Variant, ByVal l As Long, ByVal r As Long)
    Dim i As Long, j As Long, x As Variant, y As Variant
    i = l: j = r
    x = arr((l + r) \ 2)
    Do While i <= j
        Do While arr(i) < x: i = i + 1: Loop
        Do While arr(j) > x: j = j - 1: Loop
        If i <= j Then
            y = arr(i): arr(i) = arr(j): arr(j) = y
            i = i + 1: j = j - 1
        End If
    Loop
    If l < j Then SortDates arr, l, j
    If i < r Then SortDates arr, i, r
End Sub

Sub VykresliMesicniGraf(ByRef dictM As Object, ByVal filterEnd As Date)
    Dim wsObecne As Worksheet: Set wsObecne = ThisWorkbook.Worksheets("OBECNÉ INFORMACE")
    Dim cht As ChartObject
    
    ' Filtrace klíčů, které jsou mimo rozsah (jen pojistka)
    Dim finalDict As Object: Set finalDict = CreateObject("Scripting.Dictionary")
    Dim key As Variant
    For Each key In dictM.keys
        If CDate("1 " & key) <= filterEnd Then finalDict.Add key, dictM(key)
    Next key
    
    If finalDict.count = 0 Then Exit Sub
    
    ' Příprava polí pro graf
    Dim MNames() As String, MVals() As Double, MDates() As Date
    ReDim MNames(1 To finalDict.count), MVals(1 To finalDict.count), MDates(1 To finalDict.count)
    
    Dim i As Long: i = 1
    For Each key In finalDict.keys
        MNames(i) = LCase(key)
        MVals(i) = Round(finalDict(key) / 60, 1) ' Převod na hodiny
        MDates(i) = CDate("1 " & key)
        i = i + 1
    Next key
    
    Call QuickSortMonthlyData(MNames, MVals, MDates, 1, finalDict.count)
    
    ' Vykreslení
    On Error Resume Next
    Set cht = wsObecne.ChartObjects("AnimeMonthlyWatchChart")
    On Error GoTo 0
    
    If cht Is Nothing Then
        Set cht = wsObecne.ChartObjects.Add(left:=550, Width:=600, top:=650, Height:=350)
        cht.Name = "AnimeMonthlyWatchChart"
        cht.Chart.ChartType = xlColumnClustered
    End If
    
    With cht.Chart
        .HasTitle = True
        .ChartTitle.text = "Sledování Anime v roce 2026 (dny, hodiny a minuty)"
        If .SeriesCollection.count = 0 Then .SeriesCollection.NewSeries
        .SeriesCollection(1).values = MVals
        .SeriesCollection(1).XValues = MNames
        .SeriesCollection(1).ApplyDataLabels
    End With
    
    ' Formátování popisků (včetně tvé gramatiky)
    Dim totalHours As Double: totalHours = Application.WorksheetFunction.Sum(MVals)
    
    For i = 1 To UBound(MVals)
        Dim hVal As Double: hVal = MVals(i)
        
        ' 1. Čas (HH:MM)
        Dim tMin As Long: tMin = Round(hVal * 60, 0)
        Dim txtTime As String: txtTime = Int(tMin / 60) & ":" & Format(tMin Mod 60, "00")
        
        ' 2. Dny (OPRAVENO: val -> hVal)
        Dim valDays As Double: valDays = hVal / 24
        
        ' 3. Skloňování (OPRAVENO: sjednocení názvu proměnné na sSuffix)
        Dim sSuffix As String
        If valDays = 1# Then
            sSuffix = "den"
        ElseIf valDays > 0# And valDays < 1# Then
            sSuffix = "dne"
        ElseIf (valDays >= 2# And valDays < 5# And (valDays - Fix(valDays) = 0)) Then
            sSuffix = "dny"
        Else
            sSuffix = "dní"
        End If
        
        Dim txtDays As String: txtDays = Replace(Format(valDays, "0.0"), ".", ",") & " " & sSuffix
        
        ' 4. Procenta
        Dim txtPerc As String: txtPerc = "0%"
        If totalHours > 0 Then txtPerc = Replace(Format(hVal / totalHours, "0.0%"), ".", ",")
        
        ' 5. Zápis do grafu
        With cht.Chart.SeriesCollection(1).Points(i).DataLabel
            .text = txtTime & vbLf & txtDays & vbLf & "(" & txtPerc & ")"
            .Font.Name = "Aptos Narrow"
            .Font.size = 9
            .Font.Bold = True
        End With
    Next i
End Sub

Sub VykresliDenniGraf(ByRef dictD As Object, ByVal fStart As Date, ByVal fEnd As Date)
    Dim wsObecne As Worksheet: Set wsObecne = ThisWorkbook.Worksheets("OBECNÉ INFORMACE")
    Dim chartObj As ChartObject
    
    ' --- OPRAVA: Oříznutí budoucnosti ---
    Dim effectiveEndDate As Date
    If fEnd > Date Then effectiveEndDate = Date Else effectiveEndDate = fEnd
    ' -------------------------------------
    
    ' Doplnění nulových dnů (pouze do effectiveEndDate)
    Dim d As Date
    For d = fStart To effectiveEndDate
        If Not dictD.Exists(d) Then dictD.Add d, 0
    Next d
    
    ' Filtrace dictionary (aby neobsahoval dny nad effectiveEndDate, pokud je analýza tam přidala)
    Dim finalDict As Object: Set finalDict = CreateObject("Scripting.Dictionary")
    Dim key As Variant
    For Each key In dictD.keys
        If CDate(key) >= fStart And CDate(key) <= effectiveEndDate Then
            finalDict.Add key, dictD(key)
        End If
    Next key
    
    If finalDict.count = 0 Then Exit Sub
    
    ' Převod dictionary na pole pro QuickSort
    Dim arrData As Variant
    ReDim arrData(1 To finalDict.count, 1 To 2)
    Dim i As Long: i = 1
    
    For Each key In finalDict.keys
        arrData(i, 1) = key
        arrData(i, 2) = Round(finalDict(key), 1)
        i = i + 1
    Next key
    
    Call QuickSortArray(arrData, 1, finalDict.count, 1)
    
    ' Vykreslení
    On Error Resume Next
    Set chartObj = wsObecne.ChartObjects("AnimeWatchChart")
    On Error GoTo 0
    
    If chartObj Is Nothing Then
        Set chartObj = wsObecne.ChartObjects.Add(left:=10, Width:=800, top:=100, Height:=400)
        chartObj.Name = "AnimeWatchChart"
        chartObj.Chart.ChartType = xlLine
    End If
    
    With chartObj.Chart
        .HasTitle = True
        .ChartTitle.text = "Sledování Anime v roce 2026 (minuty denně)"
        If .SeriesCollection.count = 0 Then .SeriesCollection.NewSeries
        .SeriesCollection(1).values = Application.Index(arrData, 0, 2)
        .SeriesCollection(1).XValues = Application.Index(arrData, 0, 1)
    End With
    
    ' Výpis průměru
    Dim sumTime As Double: sumTime = Application.WorksheetFunction.Sum(Application.Index(arrData, 0, 2))
    Dim daysCount As Long: daysCount = DateDiff("d", fStart, effectiveEndDate) + 1
    Dim avgM As Double: If daysCount > 0 Then avgM = sumTime / daysCount
    
    Dim txt As String
    txt = "Průměrný čas sledování za den v období: " & Format(avgM / 1440, "h:mm") & " (h:mm)"
    wsObecne.Range("FG2").Value = txt
    
    ' --- ROLLING AVERAGES PRO POZNÁMKU FG2 ---
    Dim sum7 As Double, sum30 As Double, sum180 As Double
    Dim cnt7 As Long, cnt30 As Long, cnt180 As Long
    Dim dLoop As Date, dayVal As Double, daysBack As Long
    Dim sixMonthStart As Date: sixMonthStart = DateAdd("m", -6, effectiveEndDate)
    If sixMonthStart < fStart Then sixMonthStart = fStart
    
    For dLoop = fStart To effectiveEndDate
        dayVal = 0
        If dictD.Exists(dLoop) Then dayVal = dictD(dLoop)
        daysBack = DateDiff("d", dLoop, effectiveEndDate)
        
        If daysBack < 7 Then sum7 = sum7 + dayVal: cnt7 = cnt7 + 1
        If daysBack < 30 Then sum30 = sum30 + dayVal: cnt30 = cnt30 + 1
        If dLoop >= sixMonthStart Then sum180 = sum180 + dayVal: cnt180 = cnt180 + 1
    Next dLoop
    
    Dim avg7 As Double, avg30 As Double, avg180 As Double
    If cnt7 > 0 Then avg7 = sum7 / cnt7
    If cnt30 > 0 Then avg30 = sum30 / cnt30
    If cnt180 > 0 Then avg180 = sum180 / cnt180
    
    ' Šipky a rozdíly – všechny vs. celkový průměr období (avgM z buňky)
    Dim diff7 As Double: diff7 = avg7 - avgM
    Dim diff30 As Double: diff30 = avg30 - avgM
    Dim diff180 As Double: diff180 = avg180 - avgM
    
    Dim arrow7 As String, arrow30 As String, arrow180 As String
    If diff7 > 1 Then
        arrow7 = ChrW(9650)
    ElseIf diff7 < -1 Then
        arrow7 = ChrW(9660)
    Else
        arrow7 = ChrW(9654)
    End If
    If diff30 > 1 Then
        arrow30 = ChrW(9650)
    ElseIf diff30 < -1 Then
        arrow30 = ChrW(9660)
    Else
        arrow30 = ChrW(9654)
    End If
    If diff180 > 1 Then
        arrow180 = ChrW(9650)
    ElseIf diff180 < -1 Then
        arrow180 = ChrW(9660)
    Else
        arrow180 = ChrW(9654)
    End If
    
    Dim diffTxt7 As String, diffTxt30 As String, diffTxt180 As String
    If diff7 >= 0 Then diffTxt7 = "+" & Format(Abs(diff7) / 1440, "h:mm") Else diffTxt7 = "-" & Format(Abs(diff7) / 1440, "h:mm")
    If diff30 >= 0 Then diffTxt30 = "+" & Format(Abs(diff30) / 1440, "h:mm") Else diffTxt30 = "-" & Format(Abs(diff30) / 1440, "h:mm")
    If diff180 >= 0 Then diffTxt180 = "+" & Format(Abs(diff180) / 1440, "h:mm") Else diffTxt180 = "-" & Format(Abs(diff180) / 1440, "h:mm")
    ' Časy pro tučné formátování
    Dim time7 As String: time7 = Format(avg7 / 1440, "h:mm")
    Dim time30 As String: time30 = Format(avg30 / 1440, "h:mm")
    Dim time180 As String: time180 = Format(avg180 / 1440, "h:mm")
    
    ' Sestavení poznámky s mezerami pro zarovnání
    Dim label7 As String: label7 = "Posledních 7 dní:"
    Dim label30 As String: label30 = "Posledních 30 dní:"
    Dim label180 As String: label180 = "Posledních 6 měsíců:"
    
    ' Zarovnání na délku nejdelšího labelu + 1 mezera
    Dim maxLen As Long: maxLen = Len(label180) + 1
    label7 = label7 & Space(maxLen - Len(label7))
    label30 = label30 & Space(maxLen - Len(label30))
    label180 = label180 & " "
    
    Dim noteTxt As String
    noteTxt = label7 & time7 & " " & arrow7 & " (" & diffTxt7 & ")" & vbLf & _
              label30 & time30 & " " & arrow30 & " (" & diffTxt30 & ")" & vbLf & _
              label180 & time180 & " " & arrow180 & " (" & diffTxt180 & ")"
    
    ' Zápis poznámky do FG2
    On Error Resume Next
    wsObecne.Range("FG2").Comment.Delete
    On Error GoTo 0
    
    Dim cmt As Comment
    Set cmt = wsObecne.Range("FG2").AddComment(noteTxt)
    
    ' AutoSize + monospace font pro zarovnání
    cmt.Shape.TextFrame.AutoSize = True
    cmt.Shape.TextFrame.Characters.Font.Name = "Consolas"
    cmt.Shape.TextFrame.Characters.Font.size = 10
    
    ' Tučné formátování časů (hledáme InStr v každém řádku)
    Dim cmtText As String: cmtText = cmt.Shape.TextFrame.Characters.text
    Dim pos As Long, searchFrom As Long
    
    ' Řádek 1
    searchFrom = 1
    pos = InStr(searchFrom, cmtText, time7)
    If pos > 0 Then cmt.Shape.TextFrame.Characters(pos, Len(time7)).Font.Bold = True
    
    ' Řádek 2 (hledat za prvním vbLf)
    searchFrom = InStr(1, cmtText, vbLf) + 1
    pos = InStr(searchFrom, cmtText, time30)
    If pos > 0 Then cmt.Shape.TextFrame.Characters(pos, Len(time30)).Font.Bold = True
    
    ' Řádek 3 (hledat za druhým vbLf)
    searchFrom = InStr(searchFrom, cmtText, vbLf) + 1
    pos = InStr(searchFrom, cmtText, time180)
    If pos > 0 Then cmt.Shape.TextFrame.Characters(pos, Len(time180)).Font.Bold = True
End Sub

Sub VypoctiStreak(ByRef dictD As Object, ByVal fStart As Date, ByVal fEnd As Date)
    Dim wsObecne As Worksheet: Set wsObecne = ThisWorkbook.Worksheets("OBECNÉ INFORMACE")
    
    ' Najdi nejstarší datum v datech pro správný výpočet streaku přes hranice období
    Dim minDate As Date: minDate = fStart
    Dim kk As Variant
    For Each kk In dictD.keys
        If CDate(kk) < minDate Then minDate = CDate(kk)
    Next kk
    
    Dim d As Date
    Dim currStreak As Long, maxStreak As Long
    Dim currStart As Date, currEnd As Date
    Dim maxStart As Date, maxEnd As Date
    Dim tempStart As Date
    
    currStreak = 0: maxStreak = 0
    Dim inStreak As Boolean: inStreak = False
    
    ' Oříznutí budoucnosti pro výpočet streaku
    Dim effectiveEndDate As Date
    Dim maxDataDate As Date: maxDataDate = minDate
    Dim k As Variant
    For Each k In dictD.keys
        If CDate(k) > maxDataDate And dictD(k) >= 20 Then maxDataDate = CDate(k)
    Next k
    
    If maxDataDate > Date Then effectiveEndDate = maxDataDate Else effectiveEndDate = Date
    If effectiveEndDate > fEnd Then effectiveEndDate = fEnd
    
    ' 1. Výpočet (Historický nejdelší)
    For d = minDate To effectiveEndDate
        If dictD.Exists(d) And dictD(d) >= 20 Then
            If Not inStreak Then
                inStreak = True
                tempStart = d
                currStreak = 1
            Else
                currStreak = currStreak + 1
            End If
            
            If currStreak > maxStreak Then
                maxStreak = currStreak
                maxStart = tempStart
                maxEnd = d
            End If
        Else
            inStreak = False
            currStreak = 0
        End If
    Next d
    If inStreak And currStreak > maxStreak Then
        maxStreak = currStreak
        maxStart = tempStart
        maxEnd = effectiveEndDate
    End If

    ' 2. Výpočet AKTUÁLNÍHO streaku
    Dim actStreak As Long: actStreak = 0
    Dim actStart As Date, actEnd As Date
    Dim lastDate As Date: lastDate = 0
    
    If dictD.Exists(effectiveEndDate) And dictD(effectiveEndDate) >= 20 Then
        lastDate = effectiveEndDate
    ElseIf dictD.Exists(effectiveEndDate - 1) And dictD(effectiveEndDate - 1) >= 20 Then
        lastDate = effectiveEndDate - 1
    End If
    
    If lastDate > 0 Then
        actEnd = lastDate
        d = lastDate
        Do While d >= minDate
            If dictD.Exists(d) And dictD(d) >= 20 Then
                actStreak = actStreak + 1
                actStart = d
                d = d - 1
            Else
                Exit Do
            End If
        Loop
    End If
    
    ' 3. Zápis do buňky
    Dim txt As String
    txt = "Watch Streak: " & actStreak & " " & SkloňujDen(actStreak) & " (nejdelší: " & maxStreak & " " & SkloňujDen(maxStreak) & ")"
    wsObecne.Range("FF2").Value = txt
    
    ' 4. Práce s KOMENTÁŘEM (Zachování velikosti + Text + Formátování)
    Dim cmt As Comment
    Dim existingText As String
    Dim customText As String
    Dim newNote As String
    
    ' Proměnné pro uložení velikosti
    Dim oldWidth As Double
    Dim oldHeight As Double
    Dim hadComment As Boolean
    
    On Error Resume Next
    Set cmt = wsObecne.Range("FF2").Comment
    
    ' ZÁLOHA VELIKOSTI A TEXTU
    If Not cmt Is Nothing Then
        existingText = cmt.text
        oldWidth = cmt.Shape.Width   ' <--- Uložíme šířku
        oldHeight = cmt.Shape.Height ' <--- Uložíme výšku
        hadComment = True
    Else
        existingText = ""
        hadComment = False
    End If
    On Error GoTo ErrWatchStreak
    
    ' Filtrujeme staré automatické řádky, zbytek (tvůj text) si necháme
    If Len(existingText) > 0 Then
        Dim lines() As String
        lines = Split(existingText, vbLf)
        Dim i As Long
        For i = 0 To UBound(lines)
            Dim ln As String: ln = Trim(lines(i))
            ' Hledáme klíčová slova, která značí automatický text
            If InStr(1, ln, "Current Watch Streak") = 0 And _
               InStr(1, ln, "Longest Watch Streak") = 0 And _
               InStr(1, ln, "Current:") = 0 And _
               InStr(1, ln, "Longest:") = 0 And _
               ln <> "" Then
                customText = customText & lines(i) & vbLf
            End If
        Next i
    End If
    
    ' Sestavení nové poznámky
    Dim strAct As String, strMax As String
    
    ' Formát: "Current Watch Streak: 01.01.2025 - 05.01.2025"
    If actStreak > 0 Then
        strAct = "Current Watch Streak: " & Format(actStart, "dd.mm.yyyy") & " - " & Format(actEnd, "dd.mm.yyyy")
    Else
        strAct = "Current Watch Streak: N/A"
    End If
    
    If maxStreak > 0 Then
        strMax = "Longest Watch Streak: " & Format(maxStart, "dd.mm.yyyy") & " - " & Format(maxEnd, "dd.mm.yyyy")
    Else
        strMax = "Longest Watch Streak: N/A"
    End If
    
    newNote = strAct & vbLf & strMax
    If Len(customText) > 0 Then newNote = newNote & vbLf & left(customText, Len(customText) - 1)
    
    ' Aplikace nového textu
    On Error Resume Next
    wsObecne.Range("FF2").Comment.Delete
    Set cmt = wsObecne.Range("FF2").AddComment(newNote)
    
    ' OBNOVENÍ VELIKOSTI
    If hadComment Then
        cmt.Shape.TextFrame.AutoSize = False ' Musíme vypnout AutoSize, aby platily rozměry
        cmt.Shape.Width = oldWidth
        cmt.Shape.Height = oldHeight
    Else
        cmt.Shape.TextFrame.AutoSize = True ' Pokud komentář nebyl, necháme ho automaticky
    End If
    
    ' Tučné písmo JEN PRO DATUMY
    With cmt.Shape.TextFrame
        Dim pStart As Long
        
        ' Tučně pro Current
        If actStreak > 0 Then
            ' Najdi první datum
            pStart = InStr(1, .Characters.text, Format(actStart, "dd.mm.yyyy"))
            If pStart > 0 Then .Characters(pStart, 10).Font.Bold = True
            ' Najdi druhé datum
            pStart = InStr(pStart + 10, .Characters.text, Format(actEnd, "dd.mm.yyyy"))
            If pStart > 0 Then .Characters(pStart, 10).Font.Bold = True
        End If
        
        ' Tučně pro Longest
        If maxStreak > 0 Then
            Dim startLongest As Long
            startLongest = InStr(1, .Characters.text, "Longest Watch Streak")
            
            If startLongest > 0 Then
                pStart = InStr(startLongest, .Characters.text, Format(maxStart, "dd.mm.yyyy"))
                If pStart > 0 Then .Characters(pStart, 10).Font.Bold = True
                
                pStart = InStr(pStart + 10, .Characters.text, Format(maxEnd, "dd.mm.yyyy"))
                If pStart > 0 Then .Characters(pStart, 10).Font.Bold = True
            End If
        End If
    End With
    
    Exit Sub
ErrWatchStreak:
    ' Tichý exit
End Sub

' --- CENTRÁLNÍ ANALÝZA DAT (JÁDRO SINGLE-PASS) ---
Private Sub AnalyzujDataCentralne(ByRef dataArr As Variant, ByRef CommentsI() As String, ByRef CommentsN() As String, ByRef CommentsJ() As String, _
                                  ByVal lastRow As Long, ByVal fStart As Date, ByVal fEnd As Date, _
                                  ByRef dictD As Object, ByRef dictM As Object, _
                                  ByRef dictStreakD As Object)
    
    Dim i As Long, k As Long
    Dim regEp As Object, regN As Object, matches As Object
    
    ' Inicializace Regexu jednou
    Set regEp = CreateObject("VBScript.RegExp"): regEp.Pattern = "EP:\s*(\d+)\s*(?:-\s*(\d+))?;\s*(\d{1,2}\.\d{1,2}\.\d{4})": regEp.Global = True
    Set regN = CreateObject("VBScript.RegExp"): regN.Pattern = "\d{1,2}\.\d{1,2}\.\d{4}": regN.Global = True
    
    Dim epDur As Double, totEps As Long
    Dim watchTime As Double, adjTime As Double, rwCount As Long
    Dim useCmtI As Boolean
    Dim curDate As Date, monthKey As String
    Dim minsToAdd As Double
    Dim useCmtIStreak As Boolean
    
    For i = 2 To lastRow
        ' Načtení základních údajů z pole
        epDur = 0: totEps = 0
        If IsNumeric(Nz(dataArr(i, 10))) Then epDur = CDbl(Nz(dataArr(i, 10))) ' J
        If IsNumeric(Nz(dataArr(i, 9))) Then totEps = CLng(Nz(dataArr(i, 9))) ' I
        
        Dim epOverridesM As Object: Set epOverridesM = Nothing
        If Len(CommentsJ(i)) > 0 Then Set epOverridesM = ParseEpDurationsFromComment(CommentsJ(i))
        
        useCmtI = False
        useCmtIStreak = False
        
        ' --- A. ZPRACOVÁNÍ SLOUPCE I (EPISODES) ---
        If Len(CommentsI(i)) > 0 And epDur > 0 Then
            Dim lines As Variant, ln As Variant
            lines = Split(CommentsI(i), vbLf)
            For Each ln In lines
                Set matches = regEp.Execute(ln)
                If matches.count > 0 Then
                    Dim epStart As Long, epEnd As Long, epDate As Date
                    epStart = CLng(Nz(matches(0).SubMatches(0)))
                    If Nz(matches(0).SubMatches(1)) <> "" Then epEnd = CLng(Nz(matches(0).SubMatches(1))) Else epEnd = epStart
                    On Error Resume Next
                    epDate = CDate(Nz(matches(0).SubMatches(2)))
                    On Error GoTo 0
                    
                    If epDate <= fEnd Then
                        minsToAdd = CalcEpMinutes(epStart, epEnd, epOverridesM, epDur, totEps, totEps * epDur)
                        ' Streak slovník (bez dolní hranice)
                        dictStreakD(epDate) = dictStreakD(epDate) + minsToAdd
                        useCmtIStreak = True
                        ' Hlavní slovníky (s dolní hranicí)
                        If epDate >= fStart Then
                            dictD(epDate) = dictD(epDate) + minsToAdd
                            monthKey = Format(epDate, "mmmm yyyy")
                            dictM(monthKey) = dictM(monthKey) + minsToAdd
                            useCmtI = True
                        End If
                    End If
                End If
            Next ln
        End If
        
        ' --- B. ZPRACOVÁNÍ ZÁKLADNÍHO ČASU A REWATCHŮ (L, M, N, O) ---
        If IsDate(Nz(dataArr(i, 12))) And IsNumeric(Nz(dataArr(i, 15))) Then
            watchTime = CDbl(Nz(dataArr(i, 15))) ' O
            If watchTime > 0 Then
                rwCount = val(Nz(dataArr(i, 14))) ' N
                If rwCount > 0 Then adjTime = watchTime / (rwCount + 1) Else adjTime = watchTime
                
                ' Rewatch komentáře (N)
                If Len(CommentsN(i)) > 0 Then
                    Dim linesN As Variant, lnN As Variant
                    linesN = Split(CommentsN(i), vbLf)
                    For Each lnN In linesN
                        ' Logika pro "; EPs:"
                        If InStr(1, lnN, "; EPs:", vbTextCompare) > 0 Then
                            Dim parts() As String: parts = Split(lnN, "; EPs:")
                            Dim dPart As String: dPart = Trim(Mid(parts(0), InStr(parts(0), ".") + 1))
                            Dim epsArr() As String: epsArr = Split(Trim(parts(1)), ",")
                            
                            Dim rwS As Date, rwE As Date
                            If InStr(dPart, "-") > 0 Then
                                rwS = CDate(Trim(Split(dPart, "-")(0))): rwE = CDate(Trim(Split(dPart, "-")(1)))
                            Else
                                rwS = CDate(dPart): rwE = rwS
                            End If
                            
                            Dim nDays As Long: nDays = DateDiff("d", rwS, rwE) + 1
                            If UBound(epsArr) + 1 = nDays Then
                                Dim rwEpCtr As Long: rwEpCtr = 1
                                For k = 0 To nDays - 1
                                    curDate = rwS + k
                                    Dim epsForDay As Long: epsForDay = CLng(Trim(epsArr(k)))
                                    minsToAdd = CalcEpMinutes(rwEpCtr, rwEpCtr + epsForDay - 1, epOverridesM, epDur, totEps, totEps * epDur)
                                    rwEpCtr = rwEpCtr + epsForDay
                                    If curDate <= fEnd Then
                                        dictStreakD(curDate) = dictStreakD(curDate) + minsToAdd
                                        If curDate >= fStart Then
                                            dictD(curDate) = dictD(curDate) + minsToAdd
                                            monthKey = Format(curDate, "mmmm yyyy")
                                            dictM(monthKey) = dictM(monthKey) + minsToAdd
                                        End If
                                    End If
                                Next k
                            End If
                        ' Logika pro starý formát data
                        ElseIf regN.Test(lnN) Then
                            Set matches = regN.Execute(lnN)
                            Dim rS As Date, rE As Date
                            rS = CDate(Nz(matches(0)))
                            If matches.count > 1 Then rE = CDate(Nz(matches(1))) Else rE = rS
                            
                            Dim daysRW As Long: daysRW = DateDiff("d", rS, rE) + 1
                            If daysRW < 1 Then daysRW = 1
                            Dim totalRwTime As Double: totalRwTime = CalcEpMinutes(1, totEps, epOverridesM, epDur, totEps, totEps * epDur)
                            If totalRwTime = 0 Then totalRwTime = adjTime  ' Fallback
                            minsToAdd = totalRwTime / daysRW
                            For k = 0 To daysRW - 1
                                curDate = rS + k
                                If curDate <= fEnd Then
                                    dictStreakD(curDate) = dictStreakD(curDate) + minsToAdd
                                    If curDate >= fStart Then
                                        dictD(curDate) = dictD(curDate) + minsToAdd
                                        monthKey = Format(curDate, "mmmm yyyy")
                                        dictM(monthKey) = dictM(monthKey) + minsToAdd
                                    End If
                                End If
                            Next k
                        End If
                    Next lnN
                End If
                
                ' Základní rozsah (pokud nebyly EPs v I)
                If (Not useCmtI) Or (Not useCmtIStreak) Then
                    Dim sDate As Date: sDate = CDate(Nz(dataArr(i, 12)))
                    Dim eDate As Date
                    If IsDate(Nz(dataArr(i, 13))) Then eDate = CDate(Nz(dataArr(i, 13))) Else eDate = Date
                    
                    If sDate <= eDate Then
                        Dim diff As Long: diff = DateDiff("d", sDate, eDate) + 1
                        If diff < 1 Then diff = 1
                        Dim totalFirstWatch As Double: totalFirstWatch = CalcEpMinutes(1, totEps, epOverridesM, epDur, totEps, totEps * epDur)
                        If totalFirstWatch = 0 Then totalFirstWatch = adjTime ' Fallback
                        minsToAdd = totalFirstWatch / diff
                        For k = 0 To diff - 1
                            curDate = sDate + k
                            If curDate <= fEnd Then
                                If (Not useCmtIStreak) Then
                                    dictStreakD(curDate) = dictStreakD(curDate) + minsToAdd
                                End If
                                If (Not useCmtI) And curDate >= fStart Then
                                    dictD(curDate) = dictD(curDate) + minsToAdd
                                    monthKey = Format(curDate, "mmmm yyyy")
                                    dictM(monthKey) = dictM(monthKey) + minsToAdd
                                End If
                            End If
                        Next k
                    End If
                End If
            End If
        End If
    Next i
End Sub

' --- HLAVNÍ PROCEDURA (ŘÍDÍCÍ) ---
Public Sub SpustitAktualizaciObouGrafu()
    On Error GoTo ErrMaster
    
    ' 1. OPTIMALIZACE PROSTŘEDÍ
    Dim oldCalc As XlCalculation: oldCalc = Application.Calculation
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual
    Application.EnableEvents = False
    
    ' 2. PŘÍPRAVA DAT (ZDROJE)
    Dim wsAnime As Worksheet: Set wsAnime = ThisWorkbook.Worksheets("ANIME LIST")
    Dim lastRow As Long: lastRow = wsAnime.Cells(wsAnime.Rows.count, "L").End(xlUp).row
    
    If lastRow < 2 Then MsgBox "Žádná data.", vbExclamation: GoTo CleanExit
    
    ' Načtení hodnot do pole (Array)
    Dim dataArr As Variant: dataArr = wsAnime.Range("A1:Q" & lastRow).Value
    
    ' Načtení komentářů do paměti (Caching)
    Dim CommentsI() As String, CommentsN() As String, CommentsJ() As String
    ReDim CommentsI(1 To lastRow): ReDim CommentsN(1 To lastRow): ReDim CommentsJ(1 To lastRow)
    Dim rngCmt As Range, cell As Range
    
    On Error Resume Next
    Set rngCmt = wsAnime.Range("I2:I" & lastRow).SpecialCells(xlCellTypeComments)
    If Not rngCmt Is Nothing Then
        For Each cell In rngCmt: CommentsI(cell.row) = cell.Comment.text: Next cell
    End If
    Set rngCmt = Nothing
    Set rngCmt = wsAnime.Range("N2:N" & lastRow).SpecialCells(xlCellTypeComments)
    If Not rngCmt Is Nothing Then
        For Each cell In rngCmt: CommentsN(cell.row) = cell.Comment.text: Next cell
    End If
    Set rngCmt = Nothing
    Set rngCmt = wsAnime.Range("J2:J" & lastRow).SpecialCells(xlCellTypeComments)
    If Not rngCmt Is Nothing Then
        For Each cell In rngCmt: CommentsJ(cell.row) = cell.Comment.text: Next cell
    End If
    On Error GoTo ErrMaster

    ' 3. PŘÍPRAVA VÝSTUPNÍCH OBJEKTŮ (Slovníky)
    Dim dictDaily As Object: Set dictDaily = CreateObject("Scripting.Dictionary")
    Dim dictMonthly As Object: Set dictMonthly = CreateObject("Scripting.Dictionary")
    Dim dictStreakDaily As Object: Set dictStreakDaily = CreateObject("Scripting.Dictionary")
    
    ' 4. PŘÍPRAVA ČASOVÉHO FILTRU
    Dim wsObecne As Worksheet: Set wsObecne = ThisWorkbook.Worksheets("OBECNÉ INFORMACE")
    Dim filterStart As Date, filterEnd As Date
    If IsDate(wsObecne.Range("EN2").Value) Then filterStart = CDate(wsObecne.Range("EN2").Value) Else filterStart = DateSerial(2000, 1, 1)
    If IsDate(wsObecne.Range("EO2").Value) Then filterEnd = CDate(wsObecne.Range("EO2").Value) Else filterEnd = Date + 365

    ' <<< ZDE TVŮJ KÓD PRO REWATCHE >>>
    Call AktualizovatDataRewatchu
    Application.ScreenUpdating = False
    ' <<< KONEC >>>

    ' 5. JEDINÝ PRŮCHOD DATY (SINGLE PASS ANALÝZA)
    ' Tato procedura naplní dictDaily a dictMonthly najednou
    Call AnalyzujDataCentralne(dataArr, CommentsI, CommentsN, CommentsJ, lastRow, filterStart, filterEnd, dictDaily, dictMonthly, dictStreakDaily)
    
    ' Zjištění minut dnešního sledování pro optimalizaci bez zápisu do buněk
    If dictDaily.Exists(Date) Then
        g_WatchMinutesToday = dictDaily(Date)
    Else
        g_WatchMinutesToday = 0
    End If
    
    ' 6. VYKRESLENÍ GRAFŮ A VÝPOČTY (Používají už hotové slovníky)
    Call VykresliMesicniGraf(dictMonthly, filterEnd)
    Call VykresliDenniGraf(dictDaily, filterStart, filterEnd)
    Call VypoctiStreak(dictStreakDaily, filterStart, filterEnd)
    
    ' 7. EXTERNÍ LOGY
    Call SpustitVytvoreniHistoryLog
    Call GenerateAdvancedTimeline_V22
    ThisWorkbook.Save
    Call RunPythonOrchestrator(g_WatchMinutesToday > 1)
    'Call AktualizovatWebAplikaci
    Call VytvorGrafyPokracovani
    Call AktualizovatTagyZAnilistu_Silent

CleanExit:
    Application.EnableEvents = True
    Application.Calculation = oldCalc
    Application.ScreenUpdating = True
    MsgBox "Všechny grafy a výpočty byly úspěšně aktualizovány (Single-Pass).", vbInformation
    Exit Sub

ErrMaster:
    Application.EnableEvents = True
    Application.Calculation = xlCalculationAutomatic
    Application.ScreenUpdating = True
    MsgBox "Chyba: " & err.Description, vbCritical
End Sub


' =========================================================================================
' === PARSER PRO P?ESN? D?LKY EPIZOD (Sloupec J)
' =========================================================================================
Private Function ParseEpDurationsFromComment(ByVal cmtText As String) As Object
    Dim dict As Object: Set dict = CreateObject("Scripting.Dictionary")
    If Len(cmtText) = 0 Then
        Set ParseEpDurationsFromComment = Nothing
        Exit Function
    End If
    
    Dim reg As Object: Set reg = CreateObject("VBScript.RegExp")
    reg.Pattern = "(\d+)[\.\s]*EP:\s*(\d+):(\d+)"
    reg.Global = True
    reg.IgnoreCase = True
    
    Dim lines() As String: lines = Split(cmtText, vbLf)
    Dim vLine As Variant, matches As Object
    Dim epNum As Long, mins As Double, secs As Double
    
    For Each vLine In lines
        Set matches = reg.Execute(vLine)
        If matches.count > 0 Then
            epNum = CLng(matches(0).SubMatches(0))
            mins = CDbl(matches(0).SubMatches(1))
            secs = CDbl(matches(0).SubMatches(2))
            dict.Add epNum, mins + (secs / 60)
        End If
    Next vLine
    
    If dict.count > 0 Then Set ParseEpDurationsFromComment = dict Else Set ParseEpDurationsFromComment = Nothing
End Function

Private Function CalcEpMinutes(ByVal firstEp As Long, ByVal lastEp As Long, _
                               ByVal epOverrides As Object, _
                               ByVal defaultDuration As Double, _
                               ByVal totalEpisodes As Long, _
                               ByVal totalTime As Double) As Double
    If epOverrides Is Nothing Then
        CalcEpMinutes = (lastEp - firstEp + 1) * defaultDuration
        Exit Function
    End If

    Dim sumOverrides As Double: sumOverrides = 0
    Dim countOverrides As Long: countOverrides = epOverrides.count
    Dim key As Variant
    For Each key In epOverrides.keys
        sumOverrides = sumOverrides + epOverrides(key)
    Next key

    Dim avgRemaining As Double
    If totalEpisodes > countOverrides Then
        avgRemaining = (totalTime - sumOverrides) / (totalEpisodes - countOverrides)
        If avgRemaining < 0 Then avgRemaining = defaultDuration
    Else
        avgRemaining = defaultDuration
    End If

    Dim totalMins As Double: totalMins = 0
    Dim j As Long
    For j = firstEp To lastEp
        If epOverrides.Exists(j) Then
            totalMins = totalMins + epOverrides(j)
        Else
            totalMins = totalMins + avgRemaining
        End If
    Next j

    CalcEpMinutes = totalMins
End Function
