Sub Spravovat_Top10_HM()
    ' Deklarace proměnných
    Dim ws As Worksheet
    Dim shp As Shape
    Dim rankInput As Variant
    Dim rankNum As Long
    Dim animeName As Variant
    Dim currentName As String
    Dim imagePath As Variant
    Dim maxLimit As Long
    
    ' --- NOVÉ PROMĚNNÉ PRO VÝBĚR REŽIMU ---
    Dim editMode As VbMsgBoxResult
    Dim shapePrefix As String
    Dim promptTitle As String
    Dim rankPrompt As String
    ' ---------------------------------------
    
    ' --- Cíl na správný list ---
    ' Zde musí být přesný název tvého listu
    On Error Resume Next
    Set ws = ThisWorkbook.Worksheets("OBECNÉ INFORMACE")
    If ws Is Nothing Then
        MsgBox "Chyba: List 'OBECNÉ INFORMACE' nebyl nalezen.", vbCritical
        Exit Sub
    End If
    On Error GoTo 0 ' Reset error handling

' --- Krok 1: Zeptat se, co chceme měnit (Top 10 vs HM) ---
    editMode = MsgBox("Co chceš spravovat?" & vbCrLf & vbCrLf & _
                        "Ano = TOP 10" & vbCrLf & _
                        "Ne = Honourable Mentions", _
                        vbYesNoCancel + vbQuestion, "Výběr Režimu Úprav")
    
    Select Case editMode
        Case vbYes ' --- REŽIM TOP 10 ---
            shapePrefix = "Top10_Anime_"
            promptTitle = "TOP 10"
            maxLimit = 10 ' Top 10 zůstává 10
            rankPrompt = "Který rank (1-" & maxLimit & ") chceš změnit?"
            
        Case vbNo ' --- REŽIM HONOURABLE MENTIONS ---
            shapePrefix = "HM_Anime_"
            promptTitle = "Honourable Mention"
            maxLimit = 20 ' <--- ZDE SI ZMĚŇ POČET HM (např. na 20)
            rankPrompt = "Kterou pozici HM (1-" & maxLimit & ") chceš změnit?"
            
        Case vbCancel
            Exit Sub ' Uživatel stiskl Storno
    End Select

    ' --- Krok 2: Zeptat se na Rank / Pozici ---
    ' Použijeme Application.InputBox, protože umí lépe ověřit "Cancel"
    rankInput = Application.InputBox(prompt:=rankPrompt, _
                                     title:="Krok 2: Výběr Pozice (" & promptTitle & ")", _
                                     Type:=1) ' Type 1 = Číslo
    
    ' Kontrola, zda uživatel klikl na "Cancel" (vrátí False)
    If rankInput = False Then Exit Sub
    
    rankNum = CLng(rankInput)
    
' Kontrola platného rozsahu
    If rankNum < 1 Or rankNum > maxLimit Then
        MsgBox "Neplatná pozice. Zadej prosím číslo od 1 do " & maxLimit & ".", vbCritical, "Chyba"
        Exit Sub
    End If



' --- Krok 3: Najít objekt a zeptat se na Název ---
    ' Pokusíme se najít objekt se jménem (např. "Top10_Anime_5" nebo "HM_Anime_3")
    On Error Resume Next
    ' ZDE JE TA KLÍČOVÁ ZMĚNA:
    Set shp = ws.Shapes(shapePrefix & rankNum)
    On Error GoTo 0
    
    ' Pokud objekt neexistuje, makro skončí
    If shp Is Nothing Then
        ' I TATO HLÁŠKA JE UPRAVENÁ:
        MsgBox "Chyba: Objekt '" & shapePrefix & rankNum & "' nebyl na listu 'OBECNÉ INFORMACE' nalezen." & vbCrLf & _
               "Zkontroluj, zda jsi objekt správně pojmenoval.", vbCritical, "Chyba Objektu"
        Exit Sub
    End If
    
    ' Načtení současného Alt Textu, aby se zobrazil v InputBoxu
    currentName = shp.AlternativeText
    
    animeName = Application.InputBox(prompt:="Zadej název pro " & promptTitle & " #" & rankNum & ":", _
                                     title:="Krok 3: Název Anime", _
                                     Default:=currentName, _
                                     Type:=2) ' Type 2 = Text
                                     
    ' Kontrola, zda uživatel klikl na "Cancel" (vrátí "False" jako string)
    If animeName = "False" Then Exit Sub

    ' --- Krok 4: Zeptat se na Obrázek ---
    imagePath = Application.GetOpenFilename(FileFilter:="Obrázky (*.jpg; *.jpeg; *.png; *.bmp),*.jpg;*.jpeg;*.png;*.bmp", _
                                          title:="Krok 4: Vyber obrázek pro " & promptTitle & " #" & rankNum)
                                          
    ' Kontrola, zda uživatel klikl na "Cancel" (vrátí False)
    If imagePath = False Then Exit Sub

    ' --- Krok 3: Najít objekt a zeptat se na Název ---
    ' Pokusíme se najít objekt se jménem (např. "Top10_Anime_5" nebo "HM_Anime_3")
    On Error Resume Next
    Set shp = ws.Shapes(shapePrefix & rankNum)
    On Error GoTo 0
    
    ' Pokud objekt neexistuje, makro skončí
    If shp Is Nothing Then
        MsgBox "Chyba: Objekt '" & shapePrefix & rankNum & "' nebyl na listu 'OBECNÉ INFORMACE' nalezen." & vbCrLf & _
               "Zkontroluj, zda jsi objekt správně pojmenoval.", vbCritical, "Chyba Objektu"
        Exit Sub
    End If

    ' --- Krok 4: Aktualizace Objektu ---
    ' Zablokování aktualizace obrazovky, aby vkládání neprobliklo
    Application.ScreenUpdating = False
    
    On Error Resume Next ' Ošetření chyb při práci s objektem
    With shp
        ' Nastavení nového názvu
        .AlternativeText = CStr(animeName)
        
        ' Vložení nového obrázku jako výplň
        .Fill.Visible = msoTrue
        .Fill.UserPicture CStr(imagePath)
        ' Tento řádek zajistí, že se obrázek přizpůsobí, ale neroztáhne
        .Fill.TextureTile = msoFalse
    End With
    
    ' Kontrola, jestli vložení proběhlo v pořádku
    If err.Number <> 0 Then
        MsgBox "Došlo k chybě při vkládání obrázku nebo nastavování textu." & vbCrLf & "Error: " & err.Description, vbCritical
    Else
        ' Potvrzení o úspěchu
        MsgBox promptTitle & " #" & rankNum & " byl úspěšně aktualizován na '" & animeName & "'.", vbInformation, "Hotovo"
    End If
    
    On Error GoTo 0
    Application.ScreenUpdating = True
    
End Sub
