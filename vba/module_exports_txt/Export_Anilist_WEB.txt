Option Explicit

' -------------------------------------------------------------------------
' Makro pro automaticky export dat z Excelu a push na GitHub pro Webovou aplikaci
' -------------------------------------------------------------------------

Sub AktualizovatWebAplikaci()
    ' Nejdřív přegenerovat HISTORY LOG list — jinak by export poslal na web
    ' zastaralou historii bez epizod zapsaných od posledního přegenerování
    ' (přesně tohle způsobovalo "chybějící epizody nově sledovaného anime").
    Call SpustitVytvoreniHistoryLog

    ' Automaticky uložit sešit před exportem, aby Python viděl nejnovější data
    ThisWorkbook.Save
    
    Dim pyScriptPath As String
    Dim webAppPath As String
    Dim excelFilePath As String
    Dim pythonCommand As String
    Dim wsh As Object
    
    ' =========================================================
    ' 1. NASTAVENÍ CEST
    ' =========================================================
    
    ' Zjistíme lokální cestu k tomuto sešitu (řeší i OneDrive chybku 'http...')
    excelFilePath = ThisWorkbook.FullName
    If left(excelFilePath, 4) = "http" Then
        ' Pokud je Excel na OneDrive a vrací URL, natvrdo nastavíme lokální cestu
        excelFilePath = "C:\Users\macou\OneDrive - ŠKODA AUTO VYSOKÁ ŠKOLA o.p.s\Osobní PC\Excel Projekt - nemazat\Anime_List\" & ThisWorkbook.Name
    End If
    
    ' Kde je uložený Python skript a složka s webovou aplikací?
    Dim workspacePath As String
    workspacePath = "c:\Users\macou\.gemini\antigravity-ide\scratch\Anime List WEB"
    
    pyScriptPath = workspacePath & "\export_data.py"
    webAppPath = workspacePath & "\anime-list-web"
    
    ' =========================================================
    
    ' Kontrola, zda existuje python skript
    If Dir(pyScriptPath) = "" Then
        MsgBox "Nenalezen skript: " & pyScriptPath, vbCritical, "Chyba Exportu"
        Exit Sub
    End If
    
    ' Kontrola, zda existuje lokální Excel soubor pro Python
    If Dir(excelFilePath) = "" Then
        MsgBox "Nepodařilo se najít lokální cetu k Excelu: " & excelFilePath & vbCrLf & "Python ho potřebuje přečíst.", vbCritical, "Chyba Cesty"
        Exit Sub
    End If

    Set wsh = CreateObject("WScript.Shell")
    
    ' 2. Krok: Spustit Python export na pozadí asynchronně (False = nečekat na dokončení)
    ' Předáme pythonu cestu k Excelu a složku s webem jako argumenty
    Application.StatusBar = "Spouštím aktualizaci webu na pozadí..."
    
    pythonCommand = "cmd.exe /c start ""WebUpdateLauncher"" /min cmd /c ""python """ & pyScriptPath & """ """ & excelFilePath & """ """ & webAppPath & "\public\data""" & " && timeout /t 2 && exit || pause"""
    
    wsh.Run pythonCommand, 0, False
    
    ' Okamžitě vymazat stavový řádek a uvolnit Excel uživateli
    Application.StatusBar = False
    
End Sub
