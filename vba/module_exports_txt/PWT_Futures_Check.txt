' ==========================================================================================
' MODUL: Stats_PWT_TextBox
' POPIS: Vytváří/aktualizuje textové pole se základními statistikami PWT listu.
'        Formát: "PWT: X Anime / X epizod / X,Y hodin (X,Y dní)"
'        DŮLEŽITÉ: Po prvním vytvoření si můžeš upravit výplň, ohraničení a font.
'                  Při dalších aktualizacích se formátování NERESETUJE.
'
' AUTOMATICKÁ AKTUALIZACE:
'   Pro automatickou aktualizaci při přepnutí na list, vlož tento kód do modulu listu:
'   (VBA Editor -> Dvojklik na list "ANIME PLAN TO WATCH + FUTURES" -> Vlož kód)
'
'   Private Sub Worksheet_Activate()
'       AktualizovatPWTStats
'   End Sub
'
' ==========================================================================================
Option Explicit

' Název textového pole (konstantní pro identifikaci)
Private Const TEXTBOX_NAME As String = "PWT_Stats_TextBox"

Sub AktualizovatPWTStats()
    ' ==========================================================================================
    ' Hlavní procedura: Spočítá statistiky a aktualizuje textové pole
    ' ==========================================================================================
    
    Dim ws As Worksheet
    Dim lastRow As Long, i As Long
    Dim pocetAnime As Long
    Dim celkemEpizod As Long
    Dim celkemMinut As Double
    Dim celkemHodin As Double
    Dim celkemDni As Double
    Dim statsText As String
    Dim shp As Shape
    
    ' --- NASTAVENÍ ---
    Const COL_NAZEV As String = "G"      ' Sloupec s názvem anime
    Const COL_EP_POCET As String = "I"   ' Sloupec s počtem epizod
    Const COL_CAS_TOTAL As String = "K"  ' Sloupec s celkovým časem (v minutách)
    
    ' Nastavení listu
    Set ws = ThisWorkbook.Sheets("ANIME PLAN TO WATCH + FUTURES")
    lastRow = ws.Cells(ws.Rows.count, COL_NAZEV).End(xlUp).row
    
    ' Reset počítadel
    pocetAnime = 0
    celkemEpizod = 0
    celkemMinut = 0
    
    ' --- VÝPOČET STATISTIK ---
    ' Začínáme od řádku 3 (přeskakujeme hlavní i sloupcové záhlaví)
    For i = 3 To lastRow
        ' Načtení názvu a ořezání mezer
        Dim animeName As String
        animeName = Trim(ws.Range(COL_NAZEV & i).Value)
        
        ' Kontrola, zda řádek obsahuje název anime
        ' A zároveň má vyplněné "Pořadí" v sloupci F (tím odfiltrujeme nadpisy kategorií)
        If Len(animeName) > 0 And animeName <> "Název" And Len(Trim(ws.Range("F" & i).Value)) > 0 Then
            pocetAnime = pocetAnime + 1
            
            ' Počet epizod
            If IsNumeric(ws.Range(COL_EP_POCET & i).Value) And Not IsEmpty(ws.Range(COL_EP_POCET & i).Value) Then
                celkemEpizod = celkemEpizod + CLng(ws.Range(COL_EP_POCET & i).Value)
            End If
            
            ' Celkový čas (ošetření chyb #HODNOTA!)
            If Not IsError(ws.Range(COL_CAS_TOTAL & i).Value) Then
                If IsNumeric(ws.Range(COL_CAS_TOTAL & i).Value) And Not IsEmpty(ws.Range(COL_CAS_TOTAL & i).Value) Then
                    celkemMinut = celkemMinut + CDbl(ws.Range(COL_CAS_TOTAL & i).Value)
                End If
            End If
        End If
    Next i
    
    ' Přepočet na hodiny a dny
    celkemHodin = Round(celkemMinut / 60, 1)
    celkemDni = Round(celkemMinut / 60 / 24, 1)
    
    ' --- SESTAVENÍ TEXTU ---
    ' Formát: "PWT: X Anime / X epizod / X,Y hodin (X,Y dní)"
    statsText = "PTW: " & pocetAnime & " Anime / " & _
                celkemEpizod & " epizod / " & _
                Replace(CStr(celkemHodin), ".", ",") & " hodin (" & _
                Replace(CStr(celkemDni), ".", ",") & " dní)"
    
    ' --- VYTVOŘENÍ NEBO AKTUALIZACE TEXTOVÉHO POLE ---
    Set shp = GetOrCreateTextBox(ws, "PWT_Stats_TextBox", 10, 10, 350, 30)
    
    ' Aktualizace pouze textu (zachová formátování)
    shp.TextFrame2.TextRange.text = statsText
    
    ' Centrování textu (horizontálně a vertikálně)
    With shp.TextFrame2
        .HorizontalAnchor = msoAnchorCenter
        .VerticalAnchor = msoAnchorMiddle
        .TextRange.ParagraphFormat.Alignment = msoAlignCenter
    End With
    
    ' --- SELEKTIVNÍ TUČNÉ FORMÁTOVÁNÍ PRO ČÍSLA ---
    ' Nejdříve celý text nastavíme na normální
    shp.TextFrame2.TextRange.Font.Bold = msoFalse
    
    ' Pomocné proměnné pro pozice
    Dim strAnime As String, strEpizod As String, strHodin As String, strDni As String
    Dim pos As Long
    
    strAnime = CStr(pocetAnime)
    strEpizod = CStr(celkemEpizod)
    strHodin = Replace(CStr(celkemHodin), ".", ",")
    strDni = Replace(CStr(celkemDni), ".", ",")
    
    ' Tučně: počet anime (po "PWT: ")
    pos = InStr(statsText, strAnime)
    If pos > 0 Then
        shp.TextFrame2.TextRange.Characters(pos, Len(strAnime)).Font.Bold = msoTrue
    End If
    
    ' Tučně: počet epizod (hledáme za " / ")
    pos = InStr(statsText, strEpizod & " epizod")
    If pos > 0 Then
        shp.TextFrame2.TextRange.Characters(pos, Len(strEpizod)).Font.Bold = msoTrue
    End If
    
    ' Tučně: počet hodin
    pos = InStr(statsText, strHodin & " hodin")
    If pos > 0 Then
        shp.TextFrame2.TextRange.Characters(pos, Len(strHodin)).Font.Bold = msoTrue
    End If
    
    ' Tučně: počet dní
    pos = InStr(statsText, strDni & " dní)")
    If pos > 0 Then
        shp.TextFrame2.TextRange.Characters(pos, Len(strDni)).Font.Bold = msoTrue
    End If
    
End Sub

' ==========================================================================================
' POMOCNÉ FUNKCE (Nahraď původní funkci GetOrCreateTextBox touto novou verzí)
' ==========================================================================================

Private Function GetOrCreateTextBox(ws As Worksheet, boxName As String, _
                                    posX As Double, posY As Double, _
                                    w As Double, h As Double) As Shape
    Dim shp As Shape
    
    On Error Resume Next
    Set shp = ws.Shapes(boxName)
    On Error GoTo 0
    
    If shp Is Nothing Then
        Set shp = ws.Shapes.AddTextbox(msoTextOrientationHorizontal, posX, posY, w, h)
        shp.Name = boxName
        
        ' Výchozí styl
        With shp
            .Fill.Visible = msoFalse
            .line.Visible = msoTrue
            .line.ForeColor.RGB = RGB(100, 100, 100)
            .line.weight = 1
            With .TextFrame2.TextRange.Font
                .size = 11
                .Fill.ForeColor.RGB = RGB(0, 0, 0)
                .Name = "Calibri"
            End With
        End With
    End If
    
    Set GetOrCreateTextBox = shp
End Function

Private Sub ZtucnitText(shp As Shape, textToBold As String, afterText As String)
    ' Pomocná procedura: Ztuční 'textToBold', který následuje po 'afterText'
    Dim fullText As String
    Dim startPos As Long
    Dim contextPos As Long
    
    fullText = shp.TextFrame2.TextRange.text
    contextPos = InStr(fullText, afterText)
    
    If contextPos > 0 Then
        startPos = contextPos + Len(afterText)
        If Mid(fullText, startPos, Len(textToBold)) = textToBold Then
            shp.TextFrame2.TextRange.Characters(startPos, Len(textToBold)).Font.Bold = msoTrue
        End If
    End If
End Sub

Sub VymazatPWTStatsTextBox()
    ' ==========================================================================================
    ' Pomocná procedura: Smaže textové pole (pro reset)
    ' ==========================================================================================
    
    Dim ws As Worksheet
    Dim shp As Shape
    
    Set ws = ThisWorkbook.Sheets("ANIME PLAN TO WATCH + FUTURES")
    
    For Each shp In ws.Shapes
        If shp.Name = TEXTBOX_NAME Then
            shp.Delete
            Exit For
        End If
    Next shp
    
End Sub

Sub AktualizovatFuturesStats()
    ' ==========================================================================================
    ' NOVÁ PROCEDURA: Futures (Vychází do 3 měsíců)
    ' ==========================================================================================
    Dim ws As Worksheet
    Dim i As Long
    Dim pocetX As Long ' Počet Anime v A3:A1000
    Dim pocetY As Long ' Počet dat v D3:D1000 (do 3 měsíců)
    Dim statsText As String
    Dim shp As Shape
    Dim cellVal As Variant
    
    ' Název textového pole pro Futures
    Const TEXTBOX_FUTURES_NAME As String = "Futures_Stats_TextBox"
    
    Set ws = ThisWorkbook.Sheets("ANIME PLAN TO WATCH + FUTURES")
    
    pocetX = 0
    pocetY = 0
    
    ' --- VÝPOČET ---
    For i = 3 To 1000
        ' 1. Počet Anime (Sloupec A - není prázdný)
        If Len(Trim(ws.Range("A" & i).Value)) > 0 Then
            pocetX = pocetX + 1
        End If
        
        ' 2. Vychází do 3 měsíců (Sloupec D)
        cellVal = ws.Range("D" & i).Value
        If IsDate(cellVal) Then
            ' Logika: Datum je dnes nebo v budoucnu A ZÁROVEŇ méně než za 90 dní
            If CDate(cellVal) >= Date And CDate(cellVal) <= DateAdd("m", 3, Date) Then
                pocetY = pocetY + 1
            End If
        End If
    Next i
    
    ' --- SESTAVENÍ TEXTU ---
    statsText = "Futures: " & pocetX & " Anime" & vbNewLine & _
                pocetY & " vychází do 3 měsíců"
                
    ' --- VYKRESLENÍ ---
    ' Pozice: Left=370 (vedle prvního boxu), Top=10, Šířka=250, Výška=45
    Set shp = GetOrCreateTextBox(ws, TEXTBOX_FUTURES_NAME, 370, 10, 250, 45)
    
    ' Nastavení textu
    shp.TextFrame2.TextRange.text = statsText
    With shp.TextFrame2
        .HorizontalAnchor = msoAnchorCenter
        .VerticalAnchor = msoAnchorMiddle
        .TextRange.ParagraphFormat.Alignment = msoAlignCenter
    End With
    
    ' --- ZTUČNĚNÍ ČÍSEL (X a Y) ---
    ' Reset na normální písmo
    shp.TextFrame2.TextRange.Font.Bold = msoFalse
    
    ' Tučně X (hledáme číslo za "Futures: ")
    ZtucnitText shp, CStr(pocetX), "Futures: "
    
    ' Tučně Y (hledáme číslo na začátku druhého řádku)
    Dim posY As Long
    posY = InStr(statsText, vbNewLine & pocetY & " vychází")
    If posY > 0 Then
        ' Posuneme se o 1 znak (vbNewLine), abychom trefili začátek čísla
        shp.TextFrame2.TextRange.Characters(posY + 1, Len(CStr(pocetY))).Font.Bold = msoTrue
    End If

End Sub
