Option Explicit

' ===================================================================
' === KONSTANTY ===
' ===================================================================

Private Const EXE_FILE_NAME As String = "Anime EP AVG Length Calc.exe"
Private Const TEMP_FILE_NAME As String = "mediainfo_temp.txt"
Private Const SW_HIDE As Long = 0

Private Const ROOT_FOLDER_PATH As String = "C:\Users\macou\Desktop\Anime watching or planned to watch"
Private Const TARGET_SHEET_NAME As String = "ANIME LIST"
Private Const COLOR_SUMMARY As Long = &HDD9EED
Private Const COLOR_FOLDER_NAME As Long = &HC0FF&  ' 49407 = RGB(255, 192, 0) - oranžová
Private Const COLOR_DEFAULT As Long = vbWhite

' ===================================================================
' === VÝPOČET PRŮMĚRNÉ DÉLKY EPIZOD ===
' ===================================================================

Public Sub SpocitejPrumerDelkyEpizod()
    Dim fd As fileDialog
    Dim selectedFile As Variant
    Dim totalMilliseconds As Double
    Dim fileCount As Long
    Dim fso As Object
    Dim shell As Object
    Dim mediaInfoPath As String
    Dim tempFilePath As String
    Dim commandParams As String
    Dim fileStream As Object
    Dim durationStr As String
    Dim averageMinutes As Double
    Dim formattedResult As String
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set shell = CreateObject("WScript.Shell")
    
    ' Krok 1: Najít EXE vedle sešitu
    ' Pokus 1: ThisWorkbook.Path (funguje lokálně, na OneDrive vrací URL)
    mediaInfoPath = ThisWorkbook.path & "\" & EXE_FILE_NAME
    
    ' Pokus 2: Známá lokální cesta OneDrive (fallback pro OneDrive Business)
    If Not fso.fileExists(mediaInfoPath) Then
        mediaInfoPath = "C:\Users\macou\OneDrive - ŠKODA AUTO VYSOKÁ ŠKOLA o.p.s\Osobní PC\Excel Projekt - nemazat\Anime_List\" & EXE_FILE_NAME
    End If
    
    ' Pokus 3: Ruční výběr uživatelem
    If Not fso.fileExists(mediaInfoPath) Then
        MsgBox "Soubor '" & EXE_FILE_NAME & "' nebyl nalezen vedle sešitu." & vbCrLf & _
               "Prosím, v dalším okně ho najděte ručně.", vbInformation, "Najít nástroj"
        
        Set fd = Application.fileDialog(3)
        With fd
            .title = "Najděte soubor " & EXE_FILE_NAME
            .AllowMultiSelect = False
            .Filters.Clear
            .Filters.Add "Aplikace", "*.exe"
            
            If .Show = -1 Then
                mediaInfoPath = .SelectedItems(1)
                If LCase(fso.GetFileName(mediaInfoPath)) <> LCase(EXE_FILE_NAME) Then
                    MsgBox "Nebyl vybrán soubor '" & EXE_FILE_NAME & "'. Akce zrušena.", vbExclamation
                    Exit Sub
                End If
            Else
                Exit Sub
            End If
        End With
    End If
    
    tempFilePath = fso.GetParentFolderName(mediaInfoPath) & "\" & TEMP_FILE_NAME
    
    ' Krok 2: Výběr .mkv souborů
    Set fd = Application.fileDialog(3)
    With fd
        .title = "Vyberte .mkv soubory pro výpočet (Ctrl+A pro všechny)"
        .AllowMultiSelect = True
        .Filters.Clear
        .Filters.Add "MKV Soubory", "*.mkv"
        
        If .Show = -1 Then
            totalMilliseconds = 0
            fileCount = 0
            
            For Each selectedFile In .SelectedItems
                commandParams = "--Output=""General;%Duration%"" """ & selectedFile & """ > """ & tempFilePath & """"
                shell.Run "cmd /c """"" & mediaInfoPath & """ " & commandParams & """", SW_HIDE, True
                
                If fso.fileExists(tempFilePath) Then
                    Set fileStream = fso.OpenTextFile(tempFilePath, 1)
                    durationStr = ""
                    If Not fileStream.AtEndOfStream Then
                        durationStr = fileStream.ReadAll
                    End If
                    fileStream.Close
                    
                    If IsNumeric(durationStr) Then
                        totalMilliseconds = totalMilliseconds + CDbl(durationStr)
                        fileCount = fileCount + 1
                    End If
                    
                    fso.DeleteFile tempFilePath
                End If
            Next selectedFile
        Else
            Exit Sub
        End If
    End With
    
    ' Krok 3: Uklidit – zabít případné visící procesy a uvolnit objekty
    On Error Resume Next
    shell.Run "taskkill /f /im """ & EXE_FILE_NAME & """", SW_HIDE, True
    On Error GoTo 0
    Set shell = Nothing
    Set fso = Nothing
    
    ' Krok 4: Zobrazení výsledku
    If fileCount > 0 Then
        averageMinutes = (totalMilliseconds / 1000) / 60 / fileCount
        formattedResult = Format(averageMinutes, "0.000")
        
        MsgBox "Zpracováno souborů: " & fileCount & vbCrLf & vbCrLf & _
               "Průměrná délka epizody je: " & formattedResult & " min", vbInformation, "Výpočet dokončen"
    Else
        MsgBox "Nepodařilo se zpracovat žádný z vybraných souborů.", vbExclamation, "Nezpracováno"
    End If
End Sub

' ===================================================================
' === ANALÝZA SLOŽEK SE STAŽENÝM ANIME ===
' ===================================================================

Public Sub VypisStazeneAnime()
    Dim fso As Object, rootFolder As Object, subFolder As Object
    Dim outputBox As Shape, targetSheet As Worksheet, tr As Object
    Dim totalSizeAllFolders As Double, totalFilesAllFolders As Long
    Dim folderInfo As Collection
    Dim results As Variant, folderData As Variant
    Dim summaryPart As String
    Dim folderNamePart As String, subfolderCount As Long, fileCount As Long
    Dim dilyText As String, lineSuffix As String
    
    ' --- Inicializace a kontroly ---
    Set fso = CreateObject("Scripting.FileSystemObject")
    If Not fso.FolderExists(ROOT_FOLDER_PATH) Then
        MsgBox "Složka nenalezena: " & ROOT_FOLDER_PATH, vbCritical, "Chyba"
        Exit Sub
    End If
    
    On Error Resume Next
    Set targetSheet = ThisWorkbook.Sheets(TARGET_SHEET_NAME)
    Set outputBox = targetSheet.Shapes("StazeneAnime")
    On Error GoTo 0
    If targetSheet Is Nothing Or outputBox Is Nothing Then
        MsgBox "List '" & TARGET_SHEET_NAME & "' nebo textové pole 'StazeneAnime' nebylo nalezeno.", vbCritical, "Chyba"
        Exit Sub
    End If

    Application.ScreenUpdating = False
    
    ' --- Krok 1: Sběr všech dat ---
    Set folderInfo = New Collection
    totalSizeAllFolders = 0
    totalFilesAllFolders = 0
    
    Set rootFolder = fso.GetFolder(ROOT_FOLDER_PATH)
    
    If rootFolder.SubFolders.count = 0 Then
        Application.ScreenUpdating = True
        MsgBox "Složka neobsahuje žádné podsložky.", vbInformation, "Prázdná složka"
        Exit Sub
    End If
    
    For Each subFolder In rootFolder.SubFolders
        results = GetFolderStats(subFolder.path, fso)
        folderInfo.Add Array(subFolder.Name, subFolder.SubFolders.count, results(1))
        totalSizeAllFolders = totalSizeAllFolders + results(0)
        totalFilesAllFolders = totalFilesAllFolders + results(1)
    Next subFolder
    
    ' --- Krok 2: Zápis a formátování textu ---
    outputBox.TextFrame2.TextRange.text = ""
    Set tr = outputBox.TextFrame2.TextRange

    summaryPart = Format(totalSizeAllFolders / (1024 ^ 3), "0.0") & " GB, " & GetCzechPluralEPs(totalFilesAllFolders)
    
    With tr.InsertAfter("Stažené Anime: ")
        .Font.Fill.ForeColor.RGB = COLOR_DEFAULT
    End With
    With tr.InsertAfter(summaryPart)
        .Font.Fill.ForeColor.RGB = COLOR_SUMMARY
    End With

    For Each folderData In folderInfo
        folderNamePart = folderData(0)
        subfolderCount = folderData(1)
        fileCount = folderData(2)
        
        If subfolderCount = 0 And fileCount > 0 Then
            dilyText = "1 díl"
        Else
            dilyText = GetCzechPluralDily(subfolderCount)
        End If
        
        If dilyText = "1 díl" Then
            lineSuffix = " (" & GetCzechPluralEPs(fileCount) & ")"
        Else
            lineSuffix = " (" & dilyText & ", " & GetCzechPluralEPs(fileCount) & ")"
        End If
        
        With tr.InsertAfter(vbCrLf & "- ")
            .Font.Fill.ForeColor.RGB = COLOR_DEFAULT
        End With
        With tr.InsertAfter(folderNamePart)
            .Font.Fill.ForeColor.RGB = COLOR_FOLDER_NAME
        End With
        With tr.InsertAfter(lineSuffix)
            .Font.Fill.ForeColor.RGB = COLOR_DEFAULT
        End With
    Next folderData

    Application.ScreenUpdating = True
End Sub

' ===================================================================
' === Pomocné funkce ===
' ===================================================================

Private Function GetCzechPluralDily(ByVal count As Long) As String
    Select Case count
        Case 1
            GetCzechPluralDily = "1 díl"
        Case 2 To 4
            GetCzechPluralDily = count & " díly"
        Case Else
            GetCzechPluralDily = count & " dílů"
    End Select
End Function

Private Function GetCzechPluralEPs(ByVal count As Long) As String
    If count = 1 Then
        GetCzechPluralEPs = "1 EP"
    Else
        GetCzechPluralEPs = count & " EPs"
    End If
End Function

Private Function GetFolderStats(ByVal sPath As String, fso As Object) As Variant
    Dim totalSize As Double
    Dim fileCount As Long
    Dim currentFolder As Object, subFolder As Object, file As Object
    Dim subFolderStats As Variant
    
    totalSize = 0
    fileCount = 0
    
    On Error Resume Next
    Set currentFolder = fso.GetFolder(sPath)
    If err.Number <> 0 Then
        GetFolderStats = Array(0, 0)
        Exit Function
    End If
    On Error GoTo 0

    For Each file In currentFolder.Files
        totalSize = totalSize + file.size
        If LCase(fso.GetExtensionName(file.Name)) = "mkv" Then
            fileCount = fileCount + 1
        End If
    Next file
    
    For Each subFolder In currentFolder.SubFolders
        subFolderStats = GetFolderStats(subFolder.path, fso)
        totalSize = totalSize + subFolderStats(0)
        fileCount = fileCount + subFolderStats(1)
    Next subFolder
    
    GetFolderStats = Array(totalSize, fileCount)
End Function
