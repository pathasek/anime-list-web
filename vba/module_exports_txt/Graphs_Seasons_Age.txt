'================================================================================================
'##### HLAVNÍ SPOUŠTĚCÍ MAKRO #####
'================================================================================================
Public Sub AktualizovatVsechnyGrafy()
    On Error GoTo ErrorHandler
    
    ' ——— OPTIMALIZACE PROSTŘEDÍ (CENTRÁLNÍ ŘÍZENÍ) ———
    With Application
        .ScreenUpdating = False
        .Calculation = xlCalculationManual
        .EnableEvents = False
    End With
    
    ' Zavolá první proceduru pro vytvoření grafů Věk/Sezóna
    Call VytvorGrafyVekuSezony
    
    ' Zavolá druhou proceduru pro vytvoření grafu Hodnocení v čase
    Call CreateFavoriteYearsChart_The_Final_One
    
    ' Zavolá třetí proceduru pro vytvoření přehledu Nejlepších sezón
    Call VytvorPrehled_NejlepsiSezony
    
    ' ——— OBNOVENÍ PROSTŘEDÍ ———
CleanExit:
    With Application
        .ScreenUpdating = True
        .Calculation = xlCalculationAutomatic
        .EnableEvents = True
    End With
    
    'MsgBox "Všechny grafy byly úspěšně aktualizovány!", vbInformation, "Hotovo"
    Exit Sub

ErrorHandler:
    MsgBox "Nastala chyba při aktualizaci grafů: " & err.Description, vbCritical
    Resume CleanExit
End Sub

    rng.ColumnWidth = 27.14
    rng.RowHeight = 24.75

        ws.Shapes("btnAktualizaceVekSezona").Delete

    ' Vytvořit nové tlačítko
    Set btn = ws.Shapes.AddShape(msoShapeRectangle, rng.left, rng.top, rng.Width, rng.Height)

            .Visible = msoTrue
            .weight = 1.5
            .ForeColor.RGB = RGB(0, 0, 0)
        End With
        With .TextFrame2
            .VerticalAnchor = msoAnchorMiddle
            .HorizontalAnchor = msoAnchorCenter

            With .TextRange.Font

'================================================================================================
'##### PROCEDURA 1: GRAFY VĚK/SEZÓNA (THE ABSOLUTE PEAK PERFORMANCE) #####
'================================================================================================
Public Sub VytvorGrafyVekuSezony()
    On Error GoTo Chyba

    ' --- DEKLARACE PROMĚNNÝCH ---
    Dim wsData As Worksheet, wsChart As Worksheet
    Dim chtAge As ChartObject, chtSez As ChartObject
    Dim chtAgeAvg As ChartObject ' Přejmenováno z chtSezAvg
    
    Dim leftAge As Double, topAge As Double, wAge As Double, hAge As Double
    Dim leftSez As Double, topSez As Double, wSez As Double, hSez As Double
    Dim leftAvg As Double, topAvg As Double, wAvg As Double, hAvg As Double
    
    Dim lastRow As Long, i As Long
    Dim dataArr As Variant
    Dim valDate As Variant, valRating As Variant
    
    Dim todayDate As Date
    Dim d As Date, daysDiff As Long
    Dim sGrp As Long
    Dim dummyYear As Long
    
    Dim cntAge(1 To 4) As Long
    Dim cntSez(1 To 4) As Long
    Dim cntAgeRated(1 To 4) As Long ' Počet hodnocených anime v age-group
    Dim sumAge(1 To 4) As Double    ' Suma hodnocení pro age-group
    Dim seasonColors(1 To 4) As Long
    
    Dim ageLabels(1 To 4) As String, sezLabels(1 To 4) As String
    Dim totalAge As Long, totalSez As Long
    
    ' --- INITIALIZACE ---
    Set wsData = ThisWorkbook.Sheets("ANIME LIST")
    Set wsChart = ThisWorkbook.Sheets("OBECNÉ INFORMACE")
    
    todayDate = Date
    For i = 1 To 4: seasonColors(i) = -1: Next i

    ' ——— OPTIMALIZACE PROSTŘEDÍ (ŘÍZENO Z HLAVNÍHO MAKRA) ———
    ' With Application ... End With - ODSTRANĚNO

    ' --- MAZÁNÍ / PŘÍPRAVA GRAFŮ ---
    On Error Resume Next
        wsChart.ChartObjects("GrafPrumerneHodnoceniVeku").Delete
    On Error GoTo Chyba
    
    On Error Resume Next
      Set chtAge = wsChart.ChartObjects("GrafAnimeVeku")
      Set chtSez = wsChart.ChartObjects("GrafAnimeSezony")
      ' Zkusíme najít nový název, pokud ne, zkusíme starý a přejmenujeme
      Set chtAgeAvg = wsChart.ChartObjects("GrafPrumerVeku")
      If chtAgeAvg Is Nothing Then
          Set chtAgeAvg = wsChart.ChartObjects("GrafPrumerSezony")
          If Not chtAgeAvg Is Nothing Then chtAgeAvg.Name = "GrafPrumerVeku"
      End If
    On Error GoTo 0

    ' Pozice Grafu Věk
    If Not chtAge Is Nothing Then
        leftAge = chtAge.left: topAge = chtAge.top: wAge = chtAge.Width: hAge = chtAge.Height
    Else
        leftAge = 100: topAge = 700: wAge = 400: hAge = 300
    End If
    
    ' Pozice Grafu Sezón
    If Not chtSez Is Nothing Then
        leftSez = chtSez.left: topSez = chtSez.top: wSez = chtSez.Width: hSez = chtSez.Height
    Else
        leftSez = 550: topSez = 700: wSez = 300: hSez = 300
    End If
    
    ' Pozice Grafu Průměrného Hodnocení Věku (pod grafem Sezón)
    If Not chtAgeAvg Is Nothing Then
        leftAvg = chtAgeAvg.left: topAvg = chtAgeAvg.top: wAvg = chtAgeAvg.Width: hAvg = chtAgeAvg.Height
    Else
        leftAvg = leftSez
        topAvg = topSez + hSez + 20 ' 20 bodů pod grafem sezón
        wAvg = wSez
        hAvg = hSez
    End If

    ageLabels(1) = "0–2 roky": sezLabels(1) = "Winter"
    ageLabels(2) = "2–5 let": sezLabels(2) = "Spring"
    ageLabels(3) = "5–15 let": sezLabels(3) = "Summer"
    ageLabels(4) = "15+ let": sezLabels(4) = "Fall"

    ' --- NAČTENÍ DAT DO PAMĚTI ---
    lastRow = wsData.Cells(wsData.Rows.count, "F").End(xlUp).row
    If lastRow < 2 Then GoTo Konec
    
    ' Načítáme data od F do K (F=1, K=6 v poli dataArr)
    dataArr = wsData.Range("F2:K" & lastRow).Value

    ' --- HLAVNÍ SMYČKA (MATEMATICKÁ OPTIMALIZACE) ---
    For i = 1 To UBound(dataArr, 1)
        valDate = dataArr(i, 1) ' Sloupec F
        valRating = dataArr(i, 6) ' Sloupec K
        
        If IsDate(valDate) Then
            d = valDate
            daysDiff = todayDate - d
            
            ' 1) Věk
            Select Case daysDiff
                Case Is <= 730:  aGrp = 1: cntAge(1) = cntAge(1) + 1
                Case Is <= 1825: aGrp = 2: cntAge(2) = cntAge(2) + 1
                Case Is <= 5475: aGrp = 3: cntAge(3) = cntAge(3) + 1
                Case Else:       aGrp = 4: cntAge(4) = cntAge(4) + 1
            End Select
            
            ' Suma hodnocení pro věkovou skupinu
            If IsNumeric(valRating) Then
                If valRating > 0 Then ' Předpokládáme pouze hodnocená anime
                    sumAge(aGrp) = sumAge(aGrp) + CDbl(valRating)
                    cntAgeRated(aGrp) = cntAgeRated(aGrp) + 1
                End If
            End If
            
            ' 2) Sezóna
            Call GetSeasonInfo(d, sGrp, dummyYear)
            
            cntSez(sGrp) = cntSez(sGrp) + 1
            
            ' 3) Barva
            If seasonColors(sGrp) = -1 Then
                seasonColors(sGrp) = wsData.Cells(i + 1, "F").DisplayFormat.Interior.Color
            End If
        End If
    Next i

    ' --- VYKRESLENÍ GRAFŮ ---
    
    ' 1) Graf Věku - Změna na Bar Chart
    If chtAge Is Nothing Then
        Set chtAge = wsChart.ChartObjects.Add(leftAge, topAge, wAge, hAge)
        chtAge.Name = "GrafAnimeVeku"
    End If
    
    With chtAge.Chart
        .ChartType = xlBarClustered ' Změna typu na Bar
        .HasTitle = True: .ChartTitle.text = "Počet Anime podle stáří věkových skupin"
        .ChartTitle.Font.Bold = True: .HasLegend = False
        
        ' --- UPDATE MÍSTO SMAZÁNÍ (Zachování formátování) ---
        Dim srsAge As Series
        If .SeriesCollection.count > 0 Then
            Set srsAge = .SeriesCollection(1)
        Else
            Set srsAge = .SeriesCollection.NewSeries
        End If
        
        srsAge.XValues = ageLabels
        srsAge.values = Array(cntAge(1), cntAge(2), cntAge(3), cntAge(4))
        
        ' Data Labels - text aktualizujeme, formát necháme
        If Not srsAge.HasDataLabels Then srsAge.ApplyDataLabels
    End With
    
    totalAge = cntAge(1) + cntAge(2) + cntAge(3) + cntAge(4)
    If totalAge > 0 Then
        With chtAge.Chart.SeriesCollection(1)
            For i = 1 To .Points.count
                ' Data Labels na konci pruhů
                .Points(i).DataLabel.text = cntAge(i) & " (" & Format(cntAge(i) / totalAge, "0,0%") & ")"
            Next i
        End With
    End If
    
    ' 2) Graf Sezón - Změna z Pie na Bar Chart
    If chtSez Is Nothing Then
        Set chtSez = wsChart.ChartObjects.Add(leftSez, topSez, wSez, hSez)
        chtSez.Name = "GrafAnimeSezony"
    End If
    
    With chtSez.Chart
        .ChartType = xlBarClustered ' Změna na Bar
        .HasTitle = True: .ChartTitle.text = "Počet Anime podle sezóny"
        .ChartTitle.Font.Bold = True: .HasLegend = False
        
        ' --- SEŘAZENÍ DAT (Largest to Smallest) ---
        Dim sezKeys(1 To 4) As String, sezCnts(1 To 4) As Long
        Dim sezCols(1 To 4) As Long
        Dim tmpKey As String, tmpCnt As Long, tmpCol As Long
        Dim j As Long
        
        ' Naplnění pomocných polí
        For i = 1 To 4
            sezKeys(i) = sezLabels(i)
            sezCnts(i) = cntSez(i)
            sezCols(i) = seasonColors(i)
        Next i
        
        ' Bubble Sort (Sestupně podle počtu)
        For i = 1 To 3
            For j = i + 1 To 4
                If sezCnts(j) < sezCnts(i) Then
                     ' Swap
                    tmpCnt = sezCnts(i): sezCnts(i) = sezCnts(j): sezCnts(j) = tmpCnt
                    tmpKey = sezKeys(i): sezKeys(i) = sezKeys(j): sezKeys(j) = tmpKey
                    tmpCol = sezCols(i): sezCols(i) = sezCols(j): sezCols(j) = tmpCol
                End If
            Next j
        Next i
        
        ' --- UPDATE MÍSTO SMAZÁNÍ (Zachování formátování) ---
        Dim srsSez As Series
        If .SeriesCollection.count > 0 Then
            Set srsSez = .SeriesCollection(1)
        Else
            Set srsSez = .SeriesCollection.NewSeries
        End If

        srsSez.XValues = sezKeys
        srsSez.values = sezCnts
    End With

    totalSez = cntSez(1) + cntSez(2) + cntSez(3) + cntSez(4)
    If totalSez > 0 Then
        With chtSez.Chart.SeriesCollection(1)
            .ApplyDataLabels
            For i = 1 To .Points.count
                .Points(i).DataLabel.text = _
                  sezCnts(i) & " (" & Format(sezCnts(i) / totalSez, "0,0%") & ")"
                  
                ' Aplikace barev (nyní seřazených)
                If sezCols(i) <> -1 Then
                    .Points(i).Format.Fill.ForeColor.RGB = sezCols(i)
                End If
            Next i
        End With
    End If
    
    ' 3) NOVÝ GRAF: Průměrné hodnocení PODLE VĚKU (Bar Chart)
    If chtAgeAvg Is Nothing Then
        Set chtAgeAvg = wsChart.ChartObjects.Add(leftAvg, topAvg, wAvg, hAvg)
        chtAgeAvg.Name = "GrafPrumerVeku"
    End If
    
    ' Vypočat průměrů a příprava na sort
    Dim avgAge(1 To 4) As Double
    Dim avgKeys(1 To 4) As String
    Dim tmpAvg As Double
    
    For i = 1 To 4
        If cntAgeRated(i) > 0 Then
            avgAge(i) = sumAge(i) / cntAgeRated(i)
        Else
            avgAge(i) = 0
        End If
        avgKeys(i) = ageLabels(i)
    Next i
    
    ' Bubble Sort (Vzestupně podle průměru, aby největší byl nahoře)
    For i = 1 To 3
        For j = i + 1 To 4
            If avgAge(j) < avgAge(i) Then
                ' Swap
                tmpAvg = avgAge(i): avgAge(i) = avgAge(j): avgAge(j) = tmpAvg
                tmpKey = avgKeys(i): avgKeys(i) = avgKeys(j): avgKeys(j) = tmpKey
            End If
        Next j
    Next i

    With chtAgeAvg.Chart
        .ChartType = xlBarClustered
        .HasTitle = True: .ChartTitle.text = "Průměrné hodnocení věkových skupin"
        .ChartTitle.Font.Bold = True: .HasLegend = False
        
        ' --- UPDATE MÍSTO SMAZÁNÍ ---
        Dim srsAvg As Series
        If .SeriesCollection.count > 0 Then
            Set srsAvg = .SeriesCollection(1)
        Else
            Set srsAvg = .SeriesCollection.NewSeries
            ' Pouze při prvním vytvoření série/grafu nastavíme osu, jinak necháme uživatelské nastavení
             .Axes(xlValue).MinimumScale = 0
             .Axes(xlValue).MaximumScale = 10
        End If
        
        srsAvg.XValues = avgKeys
        srsAvg.values = Array(avgAge(1), avgAge(2), avgAge(3), avgAge(4))
    End With
    
    With chtAgeAvg.Chart.SeriesCollection(1)
        If Not .HasDataLabels Then .ApplyDataLabels
        For i = 1 To .Points.count
            .Points(i).DataLabel.text = Format(avgAge(i), "0.00")
            
            ' Barvy necháme defaultní (nebo modré), protože nemáme specifické barvy pro věk
             .Points(i).Format.Fill.ForeColor.RGB = RGB(91, 155, 213) ' Výchozí modrá
        Next i
    End With

Konec:
    ' Application state obnovuje hlavní makro
    Exit Sub

Chyba:
    ' Application state obnovuje hlavní makro
    MsgBox "Chyba při aktualizaci grafů Věk/Sezóna: " & err.Description, vbCritical
End Sub

'##### PROCEDURA 2: GRAF HODNOCENÍ V ČASE #####
'================================================================================================
Sub CreateFavoriteYearsChart_The_Final_One()
    '================================================================================================
    ' Autor: Gemini (Verze 9.9 - Plná kontrola uživatele nad formátem os)
    ' Popis: Makro neresetuje meze osy Y napravo a respektuje smazání titulku
    '        osy Y nalevo. Formátování je plně v rukou uživatele.
    '================================================================================================

    ' --- Fáze 1: Zpracování dat ---
    Dim wb As Workbook: Set wb = ThisWorkbook
    Dim wsAnimeList As Worksheet, wsChartTarget As Worksheet
    On Error Resume Next
    Set wsAnimeList = wb.Sheets("ANIME LIST")
    Set wsChartTarget = wb.Sheets("OBECNÉ INFORMACE")
    On Error GoTo 0
    If wsAnimeList Is Nothing Or wsChartTarget Is Nothing Then Exit Sub

    Dim yearlyScores As Object: Set yearlyScores = CreateObject("Scripting.Dictionary")
    Dim yearlyCounts As Object: Set yearlyCounts = CreateObject("Scripting.Dictionary")
    Dim decadeScores As Object: Set decadeScores = CreateObject("Scripting.Dictionary")
    Dim decadeCounts As Object: Set decadeCounts = CreateObject("Scripting.Dictionary")
    
    Dim lastRow As Long
    lastRow = wsAnimeList.Cells.Find("*", SearchOrder:=xlByRows, SearchDirection:=xlPrevious).row
    
    ' --- NAČTENÍ DAT DO POLE (Optimalizace) ---
    If lastRow < 2 Then Exit Sub
    Dim dataArr As Variant
    dataArr = wsAnimeList.Range("F2:K" & lastRow).Value

    Dim i As Long
    For i = 1 To UBound(dataArr, 1)
        Dim valDate As Variant, valScore As Variant
        valDate = dataArr(i, 1) ' Sloupec F (Datum)
        valScore = dataArr(i, 6) ' Sloupec K (Hodnocení)

        If IsDate(valDate) And IsNumeric(valScore) Then
            Dim releaseYear As Long, score As Double, decade As Long
            releaseYear = year(valDate)
            score = CDbl(valScore)
            
            If score > 0 Then
                yearlyScores(releaseYear) = yearlyScores(releaseYear) + score
                yearlyCounts(releaseYear) = yearlyCounts(releaseYear) + 1
                decade = Int(releaseYear / 10) * 10
                decadeScores(decade) = decadeScores(decade) + score
                decadeCounts(decade) = decadeCounts(decade) + 1
            End If
        End If
    Next i
    If yearlyScores.count = 0 Then
        MsgBox "Nebyly nalezena žádná platná data k vytvoření grafu.", vbExclamation, "Žádná data"
        Exit Sub
    End If

    ' --- FÁZE 2: Příprava dat ---
    Dim sortedYears As Object: Set sortedYears = CreateObject("System.Collections.ArrayList")
    Dim y As Variant
    For Each y In yearlyScores.keys: sortedYears.Add y: Next
    sortedYears.Sort
    
    Dim minDataYear As Long: minDataYear = sortedYears(0)
    Dim maxDataYear As Long: maxDataYear = sortedYears(sortedYears.count - 1)
    
    Dim dataSize As Long: dataSize = sortedYears.count
    Dim yearsArray() As Variant, yearlyAvgArray() As Double, yearlyCountsArray() As Long
    ReDim yearsArray(1 To dataSize), yearlyAvgArray(1 To dataSize), yearlyCountsArray(1 To dataSize)
    
    Dim overallMin As Double: overallMin = 10
    Dim overallMax As Double: overallMax = 0
    
    Dim idx As Long: idx = 1
    For Each y In sortedYears
        yearsArray(idx) = y
        yearlyAvgArray(idx) = Round(yearlyScores(y) / yearlyCounts(y), 2)
        yearlyCountsArray(idx) = yearlyCounts(y)
        If yearlyAvgArray(idx) < overallMin Then overallMin = yearlyAvgArray(idx)
        If yearlyAvgArray(idx) > overallMax Then overallMax = yearlyAvgArray(idx)
        idx = idx + 1
    Next y
    
    Dim sortedDecades As Object: Set sortedDecades = CreateObject("System.Collections.ArrayList")
    For Each y In decadeScores.keys: sortedDecades.Add y: Next
    sortedDecades.Sort
    Dim decadeDataSize As Long: decadeDataSize = sortedDecades.count
    Dim decadeYearsArray() As Variant, decadeValuesArray() As Double
    ReDim decadeYearsArray(1 To decadeDataSize), decadeValuesArray(1 To decadeDataSize)
    idx = 1
    For Each y In sortedDecades
        decadeYearsArray(idx) = y + 5
        decadeValuesArray(idx) = Round(decadeScores(y) / decadeCounts(y), 2)
        If decadeValuesArray(idx) < overallMin Then overallMin = decadeValuesArray(idx)
        If decadeValuesArray(idx) > overallMax Then overallMax = decadeValuesArray(idx)
        idx = idx + 1
    Next y

    ' --- FÁZE 3: AKTUALIZACE GRAFU ---
    wsChartTarget.Activate
    Dim chrtObj As ChartObject
    On Error Resume Next
    Set chrtObj = wsChartTarget.ChartObjects("AnimeHodnoceniVCaseGraf")
    On Error GoTo 0
    If chrtObj Is Nothing Then
        Set chrtObj = wsChartTarget.ChartObjects.Add(left:=250, top:=50, Width:=800, Height:=450)
        chrtObj.Name = "AnimeHodnoceniVCaseGraf"
    End If
    
    Dim chrt As Chart
    Set chrt = chrtObj.Chart
        
    With chrt
        .PlotArea.Format.Fill.Visible = msoFalse
        
        For i = .SeriesCollection.count To 1 Step -1
            If left(.SeriesCollection(i).Name, 4) = "Řada" Or left(.SeriesCollection(i).Name, 7) = "Pozadí " Then
                .SeriesCollection(i).Delete
            End If
        Next i
        
        With .Axes(xlCategory, xlPrimary)
            .MinimumScale = minDataYear - 1: .MaximumScale = maxDataYear + 1
            .HasMajorGridlines = False: .HasMinorGridlines = False
        End With

        Dim dynamicYMin As Double: dynamicYMin = Application.WorksheetFunction.Floor(overallMin, 1)
        Dim dynamicYMax As Double: dynamicYMax = Application.WorksheetFunction.Ceiling(overallMax, 1)
        
        ' ===== ZMĚNA 1: Formátování levé osy Y nyní respektuje smazání titulku =====
        With .Axes(xlValue, xlPrimary)
             ' Měřítko se stále aktualizuje podle dat
             .MinimumScale = dynamicYMin
             .MaximumScale = dynamicYMax
             ' Titulek se aktualizuje, jen pokud existuje. Pokud ho smažeš, zůstane smazaný.
             If .HasTitle Then
                .AxisTitle.text = "Průměrné hodnocení"
             End If
             .HasMajorGridlines = False
        End With
        
        Dim srs As Series
        Dim seriesName As String
        
        seriesName = "Roční průměr"
        On Error Resume Next
        Set srs = .SeriesCollection(seriesName)
        On Error GoTo 0
        If srs Is Nothing Then
            Set srs = .SeriesCollection.NewSeries
            srs.Name = seriesName
            srs.ChartType = xlXYScatterSmooth
            srs.MarkerStyle = 8: srs.MarkerSize = 6
            srs.Format.line.ForeColor.RGB = RGB(255, 0, 0)
            srs.Format.line.weight = 2.5
        End If
        srs.AxisGroup = xlPrimary
        srs.XValues = yearsArray
        srs.values = yearlyAvgArray
        Set srs = Nothing
        
        seriesName = "Dekádový průměr"
        On Error Resume Next
        Set srs = .SeriesCollection(seriesName)
        On Error GoTo 0
        If srs Is Nothing Then
            Set srs = .SeriesCollection.NewSeries
            srs.Name = seriesName
            srs.ChartType = xlXYScatterSmooth
            srs.MarkerStyle = 2: srs.MarkerSize = 8
            srs.Format.line.ForeColor.RGB = RGB(255, 192, 0)
            srs.Format.line.weight = 2
        End If
        srs.AxisGroup = xlPrimary
        srs.XValues = decadeYearsArray
        srs.values = decadeValuesArray
        Set srs = Nothing

        seriesName = "Počet anime"
        On Error Resume Next
        Set srs = .SeriesCollection(seriesName)
        On Error GoTo 0
        If srs Is Nothing Then
            Set srs = .SeriesCollection.NewSeries
            srs.Name = seriesName
            srs.ChartType = xlLine
            srs.MarkerStyle = xlMarkerStyleNone
            srs.Format.line.ForeColor.RGB = RGB(112, 173, 71)
            srs.Format.line.weight = 2
            srs.Format.line.DashStyle = msoLineDash
        End If
        srs.AxisGroup = xlSecondary
        srs.XValues = yearsArray
        srs.values = yearlyCountsArray
        Set srs = Nothing
        
        .HasTitle = True: .ChartTitle.text = "Průměrné hodnocení a počet zhlédnutých Anime v každém roce"
        .HasLegend = True
        
        Dim shp As Shape
        Dim shapeName As String
        For Each y In sortedDecades
            If y + 9.5 > .Axes(xlCategory).MinimumScale And y + 0.5 < .Axes(xlCategory).MaximumScale Then
                shapeName = "DecadeLabel_" & y
                On Error Resume Next
                Set shp = .Shapes(shapeName)
                On Error GoTo 0
                
                ' === ÚPRAVA POZICE POPISKU ===
                ' Standardně chceme popisek uprostřed dekády (y + 5)
                calcYear = y + 5
                
                ' Pokud je střed dekády (např. 1985) "schovaný" vlevo před začátkem grafu (1987),
                ' posuneme ho na začátek grafu + 1 rok (tedy nad 1988).
                If calcYear < .Axes(xlCategory).MinimumScale Then
                    calcYear = .Axes(xlCategory).MinimumScale + 1
                End If
                
                ' Výpočet pixelové pozice podle calcYear
                lblXpos = .PlotArea.InsideLeft + ((calcYear - .Axes(xlCategory).MinimumScale) / (.Axes(xlCategory).MaximumScale - .Axes(xlCategory).MinimumScale) * .PlotArea.InsideWidth) - 20
                ' =============================
                
                If shp Is Nothing Then
                    Set shp = .Shapes.AddTextbox(msoTextOrientationHorizontal, lblXpos, .PlotArea.InsideTop - 20, 40, 20)
                    shp.Name = shapeName
                    shp.TextFrame2.TextRange.text = y & "s"
                    With shp.TextFrame2.TextRange.Font
                        .size = 10: .Fill.ForeColor.RGB = RGB(220, 220, 220): .Bold = False
                    End With
                    shp.line.Visible = msoFalse: shp.Fill.Visible = msoFalse
                Else
                    shp.left = lblXpos
                    shp.top = .PlotArea.InsideTop - 20
                End If
                Set shp = Nothing
            End If
        Next y
        
        .HasAxis(xlCategory, xlSecondary) = True
        .HasAxis(xlValue, xlSecondary) = True
        
        With .Axes(xlCategory, xlSecondary)
             .MinimumScale = minDataYear - 1: .MaximumScale = maxDataYear + 1
             .TickLabelPosition = xlNone
             .HasTitle = False: .HasMajorGridlines = False: .HasMinorGridlines = False
        End With

        ' ===== ZMĚNA 2: Meze (minimum/maximum) pravé osy Y se již neaktualizují =====
        With .Axes(xlValue, xlSecondary)
            ' Následující dva řádky jsou smazány, aby se neresetovaly tvé manuální meze.
            ' .MinimumScale = 0
            ' .MaximumScale = Application.WorksheetFunction.Ceiling(maxCount, 5)
            
            ' UPOZORNĚNÍ: Pokud přidáš data, která přesáhnou tvé manuální nastavení,
            ' nemusí být v grafu vidět! Budeš muset meze osy upravit ručně.
            
            .HasTitle = False
            .HasMajorGridlines = False
        End With
    End With
    
End Sub


'================================================================================================
'##### POMOCNÉ FUNKCE PRO PROCEDURU 3 #####
'================================================================================================
Private Function GetSeriesNameFromComment(cell As Range) As String
    If Not cell.Comment Is Nothing Then
        Dim lines As Variant
        lines = Split(cell.Comment.text, vbLf)
        Dim line As Variant
        For Each line In lines
            If InStr(1, Trim(line), "Název série:", vbTextCompare) = 1 Then
                GetSeriesNameFromComment = Trim(Mid(line, InStr(1, line, ":") + 1))
                Exit Function
            End If
        Next line
    End If
    GetSeriesNameFromComment = ""
End Function


'================================================================================================
'##### PROCEDURA 3: NEJLEPŠÍ ANIME SEZÓNY (Top 8) #####
'================================================================================================
Private Sub VytvorPrehled_NejlepsiSezony()
    ' --- KONFIGURACE ---
    Const POCET_SEZON As Long = 8
    
    Const COL_DATE_IDX As Long = 4      ' F
    Const COL_SCORE_IDX As Long = 9     ' K
    Const COL_NAME_IDX As Long = 1      ' C
    Const COL_STATUS_IDX As Long = 12   ' N
    Const COL_GENRES_IDX As Long = 6    ' H
    Const COL_EPS_IDX As Long = 7       ' I
    Const COL_DUR_IDX As Long = 8       ' J
    Const OUTLIER_RATING_THRESHOLD As Double = 2
    
    Dim wsData As Worksheet, wsTarget As Worksheet
    Set wsData = ThisWorkbook.Sheets("ANIME LIST")
    Set wsTarget = ThisWorkbook.Sheets("OBECNÉ INFORMACE")
    
    Dim lastRow As Long
    lastRow = wsData.Cells(wsData.Rows.count, "F").End(xlUp).row
    If lastRow < 2 Then Exit Sub
    
    Dim arrData As Variant
    arrData = wsData.Range("C2:Q" & lastRow).Value
    
    ' --- 1) HLAVNÍ POMOCNÉ SLOVNÍKY ---
    Dim commentsC As Object: Set commentsC = CreateObject("Scripting.Dictionary")
    Dim hmAnimeList As Object: Set hmAnimeList = CreateObject("Scripting.Dictionary")
    Dim top10AnimeList As Object: Set top10AnimeList = CreateObject("Scripting.Dictionary")
    
    Dim i As Long
    For i = 1 To UBound(arrData, 1)
        If Not wsData.Cells(i + 1, "C").Comment Is Nothing Then
            commentsC(i) = GetSeriesNameFromComment(wsData.Cells(i + 1, "C"))
        Else
            commentsC(i) = ""
        End If
    Next i
    
    Dim hmShape As Shape, hmIndex As Long
    For hmIndex = 1 To 20
        On Error Resume Next
        Set hmShape = wsTarget.Shapes("HM_Anime_" & hmIndex)
        On Error GoTo 0
        If Not hmShape Is Nothing Then
            If hmShape.AlternativeText <> "" Then hmAnimeList(LCase(hmShape.AlternativeText)) = True
        End If
        Set hmShape = Nothing
    Next hmIndex
    
    Dim topShape As Shape, topIndex As Long
    For topIndex = 1 To 10
        On Error Resume Next
        Set topShape = wsTarget.Shapes("Top10_Anime_" & topIndex)
        On Error GoTo 0
        If Not topShape Is Nothing Then
            If topShape.AlternativeText <> "" Then top10AnimeList(LCase(topShape.AlternativeText)) = True
        End If
        Set topShape = Nothing
    Next topIndex
    
    ' --- 2) SBĚR DAT A NALEZENÍ PRVNÍHO VÝSKYTU SÉRIE ---
    Dim dictSeasons As Object: Set dictSeasons = CreateObject("Scripting.Dictionary")
    Dim valDate As Variant, valScore As Variant, valName As String, score As Double
    Dim sYear As Long, sId As Long, sKey As String, sName As String, sData As Variant
    
    For i = 1 To UBound(arrData, 1)
        valDate = arrData(i, COL_DATE_IDX)
        valScore = arrData(i, COL_SCORE_IDX)
        valName = CStr(arrData(i, COL_NAME_IDX))
        
        If IsDate(valDate) Then
            Call GetSeasonInfo(CDate(valDate), sId, sYear)
            sKey = sYear & "-" & sId
            
            Dim sNameC As String: sNameC = commentsC(i)
            
            If IsNumeric(valScore) Then
                score = CDbl(valScore)
                If score > 0 Then
                    Select Case sId
                        Case 1: sName = "Winter " & sYear
                        Case 2: sName = "Spring " & sYear
                        Case 3: sName = "Summer " & sYear
                        Case 4: sName = "Fall " & sYear
                    End Select
                    
                    If Not dictSeasons.Exists(sKey) Then
                        Dim dGen As Object: Set dGen = CreateObject("Scripting.Dictionary")
                        Dim colAnim As Collection: Set colAnim = New Collection
                        dictSeasons.Add sKey, Array(0#, 0, -1#, 11#, "", 0#, 0, 0, dGen, sName, colAnim)
                    End If
                    sData = dictSeasons(sKey)
                    sData(0) = sData(0) + score: sData(1) = sData(1) + 1
                    If score > sData(2) Then sData(2) = score
                    If score < sData(3) Then sData(3) = score
                    
                    Dim eps As Long: eps = 0: Dim dur As Double: dur = 0
                    If IsNumeric(arrData(i, COL_EPS_IDX)) Then eps = CLng(arrData(i, COL_EPS_IDX))
                    If IsNumeric(arrData(i, COL_DUR_IDX)) Then dur = val(arrData(i, COL_DUR_IDX))
                    sData(5) = sData(5) + (eps * dur): sData(6) = sData(6) + eps
                    
                    Dim status As String: status = LCase(CStr(arrData(i, COL_STATUS_IDX)))
                    If InStr(status, "completed") > 0 Or InStr(status, "dokončeno") > 0 Then sData(7) = sData(7) + 1
                    
                    Dim sGenres As String: sGenres = CStr(arrData(i, COL_GENRES_IDX))
                    If Len(sGenres) > 0 Then
                        Dim gArr() As String, g As Variant: gArr = Split(sGenres, "; ")
                        Dim dG As Object: Set dG = sData(8)
                        For Each g In gArr
                            Dim trG As String: trG = Trim(g)
                            If Len(trG) > 0 Then dG(trG) = dG(trG) + 1
                        Next g
                        Set sData(8) = dG
                    End If
                    
                    Dim animeData(1 To 3) As Variant
                    animeData(1) = valName: animeData(2) = sNameC: animeData(3) = score
                    sData(10).Add animeData
                    dictSeasons(sKey) = sData
                End If
            End If
        End If
    Next i
    If dictSeasons.count = 0 Then Exit Sub
    
    ' --- 3) CHRONOLOGICKÝ PRŮCHOD (ŘEŠENÍ MVP) ---
    Dim sortedSeasonKeys As Object: Set sortedSeasonKeys = CreateObject("System.Collections.ArrayList")
    Dim tk As Variant
    For Each tk In dictSeasons.keys: sortedSeasonKeys.Add tk: Next tk
    sortedSeasonKeys.Sort ' e.g., "2023-1", "2023-2"... Text sort handles dates formatting fine.
    
    For Each tk In sortedSeasonKeys
        sKey = tk
        sData = dictSeasons(sKey)
        Dim monthAnimeCollection As Collection: Set monthAnimeCollection = sData(10)
        
        ' 1. Najdeme absolutní maximální skóre v sezóně
        Dim maxScore As Double: maxScore = -1
        Dim aItem As Variant
        For Each aItem In monthAnimeCollection
             If IsNumeric(aItem(3)) Then
                 If CDbl(aItem(3)) > maxScore Then maxScore = CDbl(aItem(3))
             End If
        Next aItem
        
        ' 2. Filtrace kandidátů s tímto maximema
        Dim topCandidates As Collection: Set topCandidates = New Collection
        If maxScore > -1 Then
             For Each aItem In monthAnimeCollection
                 If IsNumeric(aItem(3)) Then
                     If CDbl(aItem(3)) = maxScore Then topCandidates.Add aItem
                 End If
             Next aItem
        End If
        
        ' 3. Vyhodnocení vítěze (Tie-breaker přes Top10 a HM)
        If topCandidates.count > 0 Then
            Dim winnerIndex As Long: winnerIndex = 1
            Dim highestPrestige As Long: highestPrestige = 0 ' 0=Normal, 1=HM, 2=Top10
            
            Dim iCand As Long
            For iCand = 1 To topCandidates.count
                 Dim currPrestige As Long: currPrestige = 0
                 Dim cItem As Variant: cItem = topCandidates.item(iCand)
                 
                 Dim sNameCand As String: sNameCand = ""
                 If cItem(2) <> "" Then sNameCand = LCase(cItem(2)) Else sNameCand = LCase(cItem(1))
                 
                 If top10AnimeList.Exists(sNameCand) Then
                     currPrestige = 2
                 ElseIf hmAnimeList.Exists(sNameCand) Then
                     currPrestige = 1
                 End If
                 
                 If currPrestige > highestPrestige Then
                     highestPrestige = currPrestige
                     winnerIndex = iCand
                 End If
            Next iCand
            
            Dim finalWinnerName As String: finalWinnerName = topCandidates.item(winnerIndex)(1)
            Dim sScore As String
            If maxScore = Int(maxScore) Then sScore = Format(maxScore, "0") Else sScore = Replace(Format(maxScore, "0.0"), ",0", "")
            sData(4) = finalWinnerName & " (" & sScore & "/10)"
        Else
            sData(4) = "N/A (-/10)"
        End If
        
        dictSeasons(sKey) = sData
    Next tk
    
    ' --- PŘEVOD NA POLE PRO ŘAZENÍ A VÝPIS ---
    Const MIN_ANIME_COUNT As Long = 4
    
    Dim validCount As Long: validCount = 0
    Dim k As Variant
    For Each k In dictSeasons.keys
        sData = dictSeasons(k)
        If sData(1) >= MIN_ANIME_COUNT Then validCount = validCount + 1
    Next k
    
    If validCount = 0 Then Exit Sub
    
    Dim arrExport() As Variant
    ReDim arrExport(1 To validCount, 1 To 10)
    Dim idx As Long: idx = 0
    
    For Each k In dictSeasons.keys
        sData = dictSeasons(k)
        
        If sData(1) >= MIN_ANIME_COUNT Then
            idx = idx + 1
            
            Dim avgR As Double: avgR = sData(0) / sData(1)
            Dim topG As String: topG = GetTopGenre(sData(8))
            Dim compPct As Double
            If sData(1) > 0 Then compPct = (sData(7) / sData(1)) Else compPct = 0
            
            arrExport(idx, 1) = avgR            ' Sort Key
            arrExport(idx, 2) = sData(9)        ' Name
            arrExport(idx, 3) = sData(4)        ' MVP
            arrExport(idx, 4) = sData(1)        ' Count (Local List Count)
            arrExport(idx, 5) = compPct         ' Completed % (Initially Local)
            arrExport(idx, 6) = sData(2)        ' Max
            arrExport(idx, 7) = sData(3)        ' Min
            arrExport(idx, 8) = sData(5)        ' Sum Duration (Min)
            arrExport(idx, 9) = sData(6)        ' Sum Ep
            arrExport(idx, 10) = topG           ' Top Genre
        End If
    Next k
    
    ' --- SORT (Bubble Sort Descending by AvgRating) ---
    BubbleSort2D arrExport, 1, True
    
    ' --- PRE-FETCH API DATA FOR TOP X ---
    Dim rowsToShow As Long: rowsToShow = IIf(UBound(arrExport, 1) > POCET_SEZON, POCET_SEZON, UBound(arrExport, 1))
    Dim iter As Long
    
    For iter = 1 To rowsToShow
        Dim tYear As Long, tSeason As String
        Dim sNameParts() As String
        sNameParts = Split(arrExport(iter, 2), " ")
        If UBound(sNameParts) >= 1 Then
            tSeason = sNameParts(0)
            tYear = CLng(sNameParts(1))
            
            ' Fetch Total Anime in Season from AniList API (Format: TV)
            Dim totalInSeason As Long
            totalInSeason = GetAnilistSeasonTotal(tYear, tSeason)
            
            ' Calculate "Completed %" based on API Total
            ' Logic: (My Rated Count) / (API Total Count)
            Dim myCompletedC As Long
            myCompletedC = arrExport(iter, 4)
            
            If totalInSeason > 0 Then
                arrExport(iter, 5) = myCompletedC / totalInSeason
                ' Display count as "MyCount / APITotal"
                arrExport(iter, 4) = arrExport(iter, 4) & " / " & totalInSeason
            Else
                ' If API fails or returns 0 (or huge error), we keep implicit local %
            End If
        End If
        DoEvents
    Next iter
    
    ' --- VYKRESLENÍ ---
    Dim tbHeader As Shape, tbContent As Shape
    Dim defaultLeftPos As Double, defaultTopPos As Double
    Dim borderColor As Long: borderColor = RGB(0, 0, 0)
    
    ' Default Position (Right of charts)
    defaultLeftPos = 900
    defaultTopPos = 50
    
    ' --- HEADER ---
    On Error Resume Next
    Set tbHeader = wsTarget.Shapes("txtBestSeasonsHeader")
    On Error GoTo 0
    
    If tbHeader Is Nothing Then
        ' Create New Header
        Set tbHeader = wsTarget.Shapes.AddTextbox(msoTextOrientationHorizontal, defaultLeftPos, defaultTopPos, 400, 30)
        tbHeader.Name = "txtBestSeasonsHeader"
        With tbHeader
            .line.Visible = msoFalse
            .Fill.Visible = msoTrue
            .Fill.ForeColor.RGB = RGB(85, 89, 107) ' Default Dark Grey
            .Fill.Transparency = 0
            With .TextFrame2
                .TextRange.text = "Nejlepší Anime sezóny"
                .TextRange.Font.Name = "Aptos Narrow"
                .TextRange.Font.size = 18
                .TextRange.Font.Bold = msoTrue
                .TextRange.Font.Fill.Visible = msoTrue
                .TextRange.Font.Fill.Solid
                .TextRange.Font.Fill.ForeColor.RGB = RGB(255, 255, 255) ' Default White
                .VerticalAnchor = msoAnchorMiddle
                .HorizontalAnchor = msoAnchorCenter
            End With
        End With
    Else
        ' Update Existing Header (Preserve Size, Pos, Color)
        ' Ensure text is correct (in case it was changed)
        tbHeader.TextFrame2.TextRange.text = "Nejlepší Anime sezony"
    End If
    
    ' Redraw Header Lines (Delete/Recreate or Update?)
    ' Easiest to delete and recreate lines to match current Header/Content size/pos
    On Error Resume Next
    wsTarget.Shapes("lineBSHeaderTop").Delete: wsTarget.Shapes("lineBSHeaderLeft").Delete
    wsTarget.Shapes("lineBSHeaderRight").Delete
    On Error GoTo 0
    
    DrawLine wsTarget, "lineBSHeaderTop", tbHeader.left, tbHeader.top, tbHeader.left + tbHeader.Width, tbHeader.top, borderColor, 1.5
    DrawLine wsTarget, "lineBSHeaderLeft", tbHeader.left, tbHeader.top, tbHeader.left, tbHeader.top + tbHeader.Height, borderColor, 1.5
    
    ' --- CONTENT ---
    On Error Resume Next
    Set tbContent = wsTarget.Shapes("txtBestSeasonsContent")
    On Error GoTo 0
    
    ' Capture User Preferred Colors from Existing or Defaults
    Dim userFontColor As Long: userFontColor = RGB(220, 220, 220)
    
    If tbContent Is Nothing Then
        ' Create New Content Box
        Dim calcHeight As Double
        calcHeight = (rowsToShow * 3 * 14) + (rowsToShow * 10) + 50
        If calcHeight < 100 Then calcHeight = 100
        
        Set tbContent = wsTarget.Shapes.AddTextbox(msoTextOrientationHorizontal, defaultLeftPos, defaultTopPos + 30, 400, calcHeight)
        tbContent.Name = "txtBestSeasonsContent"
        
        With tbContent
            .Fill.Visible = msoTrue
            .Fill.ForeColor.RGB = RGB(85, 89, 107) ' Default
            .Fill.Transparency = 0
            .line.Visible = msoFalse
            .TextFrame2.TextRange.text = ""
            .TextFrame2.MarginLeft = 10
            .TextFrame2.MarginTop = 10
        End With
    Else
        ' Update Existing Content Box (PRESERVE SIZE & COLORS)
        ' We do NOT change .Height or .Width
        
        ' Capture Font Color if possible (from first char)
        If tbContent.TextFrame2.TextRange.Characters.count > 0 Then
            userFontColor = tbContent.TextFrame2.TextRange.Characters(1, 1).Font.Fill.ForeColor.RGB
        End If
        
        ' Clear Text
        tbContent.TextFrame2.TextRange.text = ""
    End If
    
    ' Redraw Content Lines
    On Error Resume Next
    wsTarget.Shapes("lineBSContentLeft").Delete: wsTarget.Shapes("lineBSContentRight").Delete
    wsTarget.Shapes("lineBSContentBottom").Delete
    On Error GoTo 0
    
    Dim tr As Object
    For i = 1 To rowsToShow
        If i > 1 Then tbContent.TextFrame2.TextRange.Characters.InsertAfter vbCrLf & vbCrLf
        
        ' Data Prep
        Dim dValScore As Double: dValScore = arrExport(i, 1)
        Dim sAvg As String
        If dValScore = Int(dValScore) Then
            sAvg = Format(dValScore, "0")
        Else
            sAvg = Format(dValScore, "0.00") ' User requested "X,YZ"
            sAvg = Replace(sAvg, ".", ",")
            If Right(sAvg, 3) = ",00" Then sAvg = left(sAvg, Len(sAvg) - 3)
            If Right(sAvg, 1) = "0" Then sAvg = left(sAvg, Len(sAvg) - 1) ' Handle 8,90 -> 8,9
        End If
        
        ' Max/Min formatting (same 0.## logic)
        Dim dValMax As Double: dValMax = arrExport(i, 6)
        Dim sMax As String
        If dValMax = Int(dValMax) Then sMax = Format(dValMax, "0") Else sMax = Replace(Format(dValMax, "0.##"), ".", ",")
        
        Dim dValMin As Double: dValMin = arrExport(i, 7)
        Dim sMin As String
        If dValMin = Int(dValMin) Then sMin = Format(dValMin, "0") Else sMin = Replace(Format(dValMin, "0.##"), ".", ",")
        
        Dim sCom As String: sCom = Format(arrExport(i, 5), "0 %")
        Dim sMyCount As String: sMyCount = arrExport(i, 4)
        
        Dim parts() As String
        parts = Split(CStr(sMyCount), " / ")
        sMyCount = parts(0) ' Get the first part (My Count)
        
        ' Calculate hours/days
        Dim mins As Double: mins = arrExport(i, 8)
        Dim hours As Double: hours = mins / 60
        Dim sHours As String
        If hours = Int(hours) Then sHours = Format(hours, "0") Else sHours = Replace(Format(hours, "0.0"), ".", ",")
        
        Dim days As Double: days = hours / 24
        Dim sDays As String
        If days = Int(days) Then sDays = Format(days, "0") Else sDays = Replace(Format(days, "0.0"), ".", ",")
        
        ' --- LINE 1 ---
        ' "1. [Season] / MVP: [Name] ([Score]/10)"
        ' Format: 1. (Reg) Season (Bold+Italic) / MVP: (Reg) Name (Bold+Italic) (Score/10) (Bold+Italic)
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(i & ". ")
        FormatTxt tr, False, False, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrExport(i, 2)) ' Season
        FormatTxt tr, True, True, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / MVP: ")
        FormatTxt tr, False, False, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrExport(i, 3) & vbCrLf) ' Name (Score/10) - stored in arrExport(3)
        FormatTxt tr, True, True, userFontColor
        
        ' --- LINE 2 ---
        ' "Počet Anime: [MyCount] / dokončeno [Pct] % / O hodnocení: [Avg]/10 / hodnocení ^ [Max]/10 ˇ [Min]/10"
        
        ' Symboly pomocí ChrW pro bezpečnost (Unicode)
        Dim sUp As String: sUp = ChrW(9650) ' Triangle Up
        Dim sDown As String: sDown = ChrW(9660) ' Triangle Down
        Dim sAvgSym As String: sAvgSym = ChrW(216) ' O stroke (O)
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("Počet Anime: ")
        FormatTxt tr, False, False, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(sMyCount)
        FormatTxt tr, True, False, userFontColor
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / dokončeno ")
        FormatTxt tr, False, False, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(sCom)
        FormatTxt tr, True, False, userFontColor
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / " & sAvgSym & " hodnocení: ")
        FormatTxt tr, False, False, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(sAvg & "/10")
        FormatTxt tr, True, False, userFontColor
        
        ' Condition for Max/Min display
        If dValMax <> dValMin Then
            Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / hodnocení " & sUp & " ")
            FormatTxt tr, False, False, userFontColor
            Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(sMax & "/10")
            FormatTxt tr, True, False, userFontColor
            
            Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" " & sDown & " ")
            FormatTxt tr, False, False, userFontColor
            Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(sMin & "/10" & vbCrLf)
            FormatTxt tr, True, False, userFontColor
        Else
            ' Just append newline if not showing range
            Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(vbCrLf)
            FormatTxt tr, False, False, userFontColor
        End If
        
        ' --- LINE 3 ---
        ' "Délka sledování: [H] hodin ([D] dny) / počet epizod: [Ep] / top žánr: [Genre]"
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter("Délka sledování: ")
        FormatTxt tr, False, False, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(sHours & " hodin (" & sDays & " dny)")
        FormatTxt tr, True, False, userFontColor
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / počet epizod: ")
        FormatTxt tr, False, False, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrExport(i, 9))
        FormatTxt tr, True, False, userFontColor
        
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(" / top žánr: ")
        FormatTxt tr, False, False, userFontColor
        Set tr = tbContent.TextFrame2.TextRange.Characters.InsertAfter(arrExport(i, 10))
        FormatTxt tr, True, False, userFontColor
        
    Next i
    
    DrawLine wsTarget, "lineBSContentLeft", tbContent.left, tbContent.top, tbContent.left, tbContent.top + tbContent.Height, borderColor, 1.5
    DrawLine wsTarget, "lineBSContentBottom", tbContent.left, tbContent.top + tbContent.Height, tbContent.left + tbContent.Width, tbContent.top + tbContent.Height, borderColor, 1.5

End Sub

Private Sub DrawLine(ws As Worksheet, sName As String, x1 As Double, y1 As Double, x2 As Double, y2 As Double, col As Long, w As Double)
    Dim ln As Shape
    Set ln = ws.Shapes.AddLine(x1, y1, x2, y2)
    With ln
        .Name = sName
        .line.ForeColor.RGB = col
        .line.weight = w

' funkce pro AniList API (nahrazuje Jikan)
Private Function GetAnilistSeasonTotal(ByVal year As Long, ByVal seasonName As String) As Long
    ' AniList GraphQL: Use PageInfo to get Total
    ' Query: Multi-Page to bypass 5000 limit issue and get accurate counts for all formats
    ' Formats: TV, OVA, MOVIE, SPECIAL, ONA, TV_SHORT
    
    Dim url As String
    url = "https://graphql.anilist.co"
    
    Dim sSeason As String
    sSeason = UCase(seasonName) ' AniList expects UPPERCASE (WINTER, SPRING, SUMMER, FALL)
    
    ' Construct complex query for all formats separately to ensure accurate summing
    ' Each alias (tv, ova, etc.) gets its own Page context with filters
    Dim qCore As String
    qCore = "media(season: " & sSeason & ", seasonYear: " & year & ", type: ANIME, isAdult: false, countryOfOrigin: \""JP\"") { id }"
    
    Dim query As String
    query = "{""query"":""query { " & _
            "tv: Page { pageInfo { total } " & Replace(qCore, "media(", "media(format: TV, ") & " } " & _
            "ova: Page { pageInfo { total } " & Replace(qCore, "media(", "media(format: OVA, ") & " } " & _
            "mov: Page { pageInfo { total } " & Replace(qCore, "media(", "media(format: MOVIE, ") & " } " & _
            "spe: Page { pageInfo { total } " & Replace(qCore, "media(", "media(format: SPECIAL, ") & " } " & _
            "ona: Page { pageInfo { total } " & Replace(qCore, "media(", "media(format: ONA, ") & " } " & _
            "tvs: Page { pageInfo { total } " & Replace(qCore, "media(", "media(format: TV_SHORT, ") & " } " & _
            "}""}"
            
    Dim xmlHttp As Object
    On Error Resume Next
    Set xmlHttp = CreateObject("MSXML2.XMLHTTP")
    If xmlHttp Is Nothing Then Set xmlHttp = CreateObject("Microsoft.XMLHTTP")
    On Error GoTo ErrAPI
    
    If xmlHttp Is Nothing Then Exit Function
    
    xmlHttp.Open "POST", url, False
    xmlHttp.setRequestHeader "Content-Type", "application/json"
    xmlHttp.sEnd query
    
    If xmlHttp.status = 200 Then
        Dim resp As String
        resp = xmlHttp.responseText
        ' Response: {"data":{"tv":{"pageInfo":{"total":41}}, ...}}
        
        Dim totalSum As Long: totalSum = 0
        Dim pStart As Long: pStart = 1
        Dim pTotal As Long, pColon As Long, pEnd As Long
        Dim sVal As String
        
        ' Loop through all "total": occurrences
        Do
            pTotal = InStr(pStart, resp, """total""")
            If pTotal > 0 Then
                pColon = InStr(pTotal, resp, ":")
                If pColon > 0 Then
                    pEnd = InStr(pColon, resp, "}") ' Value ends at } in pageInfo
                    If pEnd > pColon Then
                        sVal = Mid(resp, pColon + 1, pEnd - pColon - 1)
                        If IsNumeric(sVal) Then
                            totalSum = totalSum + CLng(Trim(sVal))
                        End If
                    End If
                End If
                pStart = pTotal + 1 ' Continue searching
            Else
                Exit Do
            End If
        Loop
        
        ' Safeguard against unlikely huge numbers (e.g. error 5000 * 6 = 30000)
        If totalSum > 2000 Then
            GetAnilistSeasonTotal = 0
        Else
            GetAnilistSeasonTotal = totalSum
        End If
    End If
    
    Set xmlHttp = Nothing
    Exit Function

ErrAPI:
    GetAnilistSeasonTotal = 0
End Function

Private Function GetTopGenre(ByVal dGenres As Variant) As String
    Dim k As Variant, maxC As Long, topG As String
    maxC = 0
    topG = "N/A"
    
    For Each k In dGenres.keys
        If dGenres(k) > maxC Then
            maxC = dGenres(k)
            topG = k
        End If
    Next k
    GetTopGenre = topG
End Function

Private Sub BubbleSort2D(ByRef arr As Variant, ByVal SortColIndex As Long, Optional descending As Boolean = True)
    Dim i As Long, j As Long, k As Long
    Dim Temp As Variant
    Dim condition As Boolean
    
    For i = LBound(arr, 1) To UBound(arr, 1) - 1
        For j = i + 1 To UBound(arr, 1)
            If descending Then
                condition = (arr(j, SortColIndex) > arr(i, SortColIndex))
            Else
                condition = (arr(j, SortColIndex) < arr(i, SortColIndex))
            End If
            
            If condition Then
                For k = LBound(arr, 2) To UBound(arr, 2)
                    Temp = arr(i, k)
                    arr(i, k) = arr(j, k)
                    arr(j, k) = Temp
                Next k
            End If
        Next j
    Next i
End Sub


Private Sub GetSeasonInfo(ByVal d As Date, ByRef sId As Long, ByRef sYear As Long)
    Dim m As Long
    Dim dy As Long
    Dim y As Long
    
    m = Month(d)
    dy = Day(d)
    y = year(d)
    
    Select Case m
        Case 1, 2
            sId = 1
            sYear = y
        Case 3
            If dy < 20 Then
                sId = 1
                sYear = y
            Else
                sId = 2
                sYear = y
            End If
        Case 4, 5
            sId = 2
            sYear = y
        Case 6
            If dy < 20 Then
                sId = 2
                sYear = y
            Else
                sId = 3
                sYear = y
            End If
        Case 7, 8
            sId = 3
            sYear = y
        Case 9
            If dy < 20 Then
                sId = 3
                sYear = y
            Else
                sId = 4
                sYear = y
            End If
        Case 10, 11
            sId = 4
            sYear = y
        Case 12
            If dy < 20 Then
                sId = 4
                sYear = y
            Else
                sId = 1
                sYear = y + 1
            End If
    End Select
End Sub
