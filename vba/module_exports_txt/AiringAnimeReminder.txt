
Option Explicit

' ====================================================================
' Modul: AiringAnimeReminder
' Účel:  Zobrazí přehled AIRING anime s počtem čekajících epizod
' Autor: AntiGravity
' ====================================================================

' --- BARVY (HEX) ---
Private Const COLOR_TEXT_HEX      As String = "#280B03"
Private Const COLOR_REMAINING_HEX As String = "#47D359"  ' Zelená: 1+ EP
Private Const COLOR_ZERO_HEX      As String = "#FF0000"  ' Červená: 0 EP

' --- VÝJIMKY VYSÍLACÍCH ČASŮ ---
' Formát: "Název=den čas|Další=den čas"
Private Const BROADCAST_EXCEPTIONS As String = "Dorohedoro, S02=ve středu 17:00"

' --- KOREKCE POČTU EPIZOD ---
' Formát: "Název=±N|Další=±N"
' +N = premiérový multi-drop (COTE S04 vydal 4 EP místo 1 › +3)
' -N = pauza/recap week (1 týden bez EP › -1)
Private Const EPISODE_OFFSETS As String = "Mushoku Tensei: Jobless Reincarnation, S03=+1|That Time I Got Reincarnated as a Slime, S04=-1|Ascendance of a Bookworm: Adopted Daughter of an Archduke=-2"

' --- NAPOJENÍ ---
Private Const SHAPE_NAME  As String = "txtBoxAiringReminder"
Private Const DATA_SHEET  As String = "ANIME LIST"
Private Const UI_SHEET    As String = "HODNOCENÍ ANIME"

' ====================================================================
' HLAVNÍ MAKRO (optimalizováno na minimální lag)
' ====================================================================
Public Sub UpdateAiringReminder()
    On Error GoTo ErrorHandler
    
    ' !! NEMĚNÍME Application.Calculation !!
    ' Tento modul NEMĚNÍ buňky, jen čte data a píše do Shape.
    ' Přepínání Calculation režimu způsobovalo přepočet celého sešitu = hlavní zdroj lagu.
    Application.ScreenUpdating = False
    
    Dim wsData As Worksheet, wsUI As Worksheet
    Set wsData = ThisWorkbook.Sheets(DATA_SHEET)
    Set wsUI = ThisWorkbook.Sheets(UI_SHEET)
    
    ' Načtení cache a výjimek (čisté slovníky z paměti, rychlé)
    Dim cachedTimes As Object:    Set cachedTimes = GetAiringCache(ThisWorkbook)
    Dim exceptionsDict As Object: Set exceptionsDict = LoadExceptions(BROADCAST_EXCEPTIONS)
    Dim offsetsDict As Object:    Set offsetsDict = LoadExceptions(EPISODE_OFFSETS)
    
    ' Bulk načtení dat + hyperlinků
    Dim lastRow As Long
    lastRow = wsData.Cells(wsData.Rows.count, "Q").End(xlUp).row
    If lastRow < 2 Then GoTo Cleanup
    
    Dim dataArr As Variant
    dataArr = wsData.Range("C2:Q" & lastRow).Value
    
    ' Pre-build: MAL ID mapa z hyperlinků (1 průchod místo N×COM volání)
    Dim hlMap As Object
    Set hlMap = BuildHyperlinkMap(wsData.Range("C2:C" & lastRow))
    
    ' ----------------------------------------------------------------
    ' FÁZE 1: Sběr AIRING položek do předalokovaného pole
    ' ----------------------------------------------------------------
    Dim maxItems As Long: maxItems = UBound(dataArr, 1)
    Dim names() As String:    ReDim names(1 To maxItems)
    Dim remaining() As Long:  ReDim remaining(1 To maxItems)
    Dim times() As String:    ReDim times(1 To maxItems)
    Dim sortKeys() As Long:   ReDim sortKeys(1 To maxItems)
    Dim watched() As Long:    ReDim watched(1 To maxItems)
    
    Dim i As Long, animeCount As Long: animeCount = 0
    Dim nowDbl As Double: nowDbl = CDbl(Now) ' Now() voláme pouze 1×
    
    For i = 1 To maxItems
        ' Rychlý skip pro prázdné/non-AIRING řádky
        If IsEmpty(dataArr(i, 15)) Then GoTo NextRow
        If UCase(Trim(CStr(dataArr(i, 15)))) = "AIRING!" Then
            animeCount = animeCount + 1
            
            Dim animeName As String:     animeName = Trim(CStr(dataArr(i, 1)))
            Dim startDateVar As Variant: startDateVar = dataArr(i, 4)
            Dim watchedVar As Variant:   watchedVar = dataArr(i, 7)
            
            Dim watchedEp As Long: watchedEp = 0
            If IsNumeric(watchedVar) Then watchedEp = CLng(watchedVar)
            
            ' Vysílací čas: výjimka › cache › N/A
            Dim broadcastTime As String
            If exceptionsDict.Exists(animeName) Then
                broadcastTime = exceptionsDict(animeName)
            Else
                Dim rowKey As Long: rowKey = i + 1  ' row v sheetu (data začínají od řádku 2)
                If hlMap.Exists(rowKey) Then
                    Dim malStr As String: malStr = CStr(hlMap(rowKey))
                    If cachedTimes.Exists(malStr) Then
                        broadcastTime = cachedTimes(malStr)
                    Else
                        broadcastTime = "N/A"
                    End If
                Else
                    broadcastTime = "N/A"
                End If
            End If
            
            ' Přesný počet vydaných EP (matematika + ruční korekce)
            Dim releasedEp As Long
            releasedEp = CalcReleased(startDateVar, broadcastTime, nowDbl)
            If offsetsDict.Exists(animeName) Then
                releasedEp = releasedEp + CLng(offsetsDict(animeName))
            End If
            If releasedEp < 0 Then releasedEp = 0
            
            Dim pendingEp As Long: pendingEp = releasedEp - watchedEp
            If pendingEp < 0 Then pendingEp = 0
            
            names(animeCount) = animeName
            remaining(animeCount) = pendingEp
            times(animeCount) = broadcastTime
            sortKeys(animeCount) = GetSortKey(broadcastTime)
            watched(animeCount) = watchedEp
        End If
NextRow:
    Next i

    ' Získej/vytvoř textové pole
    Dim shp As Shape
    Set shp = GetOrCreateTextBox(wsUI)

    If animeCount = 0 Then
        With shp.TextFrame2.TextRange
            .text = "Žádné Airing Anime."
            .Font.Bold = msoFalse
            .Font.UnderlineStyle = msoNoUnderline
            .Font.Fill.ForeColor.RGB = HexToRGB(COLOR_TEXT_HEX)
        End With
        GoTo Cleanup
    End If

    ' ----------------------------------------------------------------
    ' FÁZE 2: Seřazení Po›Ne (insertion sort, rychlejší pro malé N)
    ' ----------------------------------------------------------------
    If animeCount > 1 Then
        Dim j As Long
        For i = 2 To animeCount
            Dim kN As String:  kN = names(i)
            Dim kR As Long:    kR = remaining(i)
            Dim kT As String:  kT = times(i)
            Dim kS As Long:    kS = sortKeys(i)
            Dim kW As Long:    kW = watched(i)
            j = i - 1
            Do While j >= 1
                If sortKeys(j) <= kS Then Exit Do
                names(j + 1) = names(j)
                remaining(j + 1) = remaining(j)
                times(j + 1) = times(j)
                sortKeys(j + 1) = sortKeys(j)
                watched(j + 1) = watched(j)
                j = j - 1
            Loop
            names(j + 1) = kN
            remaining(j + 1) = kR
            times(j + 1) = kT
            sortKeys(j + 1) = kS
            watched(j + 1) = kW
        Next i
    End If

    ' ----------------------------------------------------------------
    ' FÁZE 3: Sestavení textu + pozice pro formátování
    ' ----------------------------------------------------------------
    ' Předalokujeme pole pozic (3 typy × animeCount)
    Dim hdrPos() As Long, hdrLen() As Long  ' hlavičky
    Dim epPos() As Long, epLen() As Long    ' EP počet
    Dim epIsZero() As Boolean               ' je EP = 0?
    Dim tmPos() As Long, tmLen() As Long    ' čas vysílání
    Dim brPos() As Long, brLen() As Long    ' závorky
    ReDim hdrPos(1 To animeCount): ReDim hdrLen(1 To animeCount)
    ReDim epPos(1 To animeCount):  ReDim epLen(1 To animeCount)
    ReDim epIsZero(1 To animeCount)
    ReDim tmPos(1 To animeCount):  ReDim tmLen(1 To animeCount)
    ReDim brPos(1 To animeCount):  ReDim brLen(1 To animeCount)
    
    Dim fullText As String: fullText = ""
    Dim sPre As String:  sPre = "Epizody k zhlédnutí: "
    Dim sMid As String:  sMid = " / nová epizoda: "
    Dim preLen As Long:  preLen = Len(sPre)
    Dim midLen As Long:  midLen = Len(sMid)
    
    For i = 1 To animeCount
        If i > 1 Then fullText = fullText & vbLf
        
        ' Hlavička
        Dim hdr As String: hdr = i & ". " & names(i)
        hdrPos(i) = Len(fullText) + 1
        hdrLen(i) = Len(hdr)
        fullText = fullText & hdr & vbLf
        
        ' EP řetězec
        Dim sep As String: sep = CStr(remaining(i)) & " EP"
        epPos(i) = Len(fullText) + preLen + 1
        epLen(i) = Len(sep)
        epIsZero(i) = (remaining(i) = 0)
        
        ' Závorka
        Dim sBrack As String
        If remaining(i) > 1 Then
            sBrack = " (EP " & CStr(watched(i) + 1) & "-" & CStr(watched(i) + remaining(i)) & ")"
        Else
            sBrack = " (EP " & CStr(watched(i) + 1) & ")"
        End If
        brPos(i) = epPos(i) + epLen(i) + 1
        brLen(i) = Len(sBrack) - 1
        
        ' Čas
        tmPos(i) = brPos(i) + brLen(i) + midLen
        tmLen(i) = Len(times(i))
        
        fullText = fullText & sPre & sep & sBrack & sMid & times(i)
    Next i

    ' ----------------------------------------------------------------
    ' FÁZE 4: Jednorázový zápis textu + formátování
    ' ----------------------------------------------------------------
    With shp.TextFrame2.TextRange
        ' 1× zápis celého textu
        .text = fullText
        
        ' 1× reset formátování (univerzální barva)
        .Font.Bold = msoFalse
        .Font.Italic = msoFalse
        .Font.UnderlineStyle = msoNoUnderline
        .Font.Fill.ForeColor.RGB = HexToRGB(COLOR_TEXT_HEX)
        
        ' Formátování pozic
        Dim clrGreen As Long: clrGreen = HexToRGB(COLOR_REMAINING_HEX)
        Dim clrRed As Long:   clrRed = HexToRGB(COLOR_ZERO_HEX)
        Dim clrHdr As Long:   clrHdr = HexToRGB(COLOR_TEXT_HEX)
        Dim totalLen As Long: totalLen = Len(fullText)
        
        For i = 1 To animeCount
            ' Hlavička (tučné + kurzíva)
            If hdrPos(i) >= 1 And hdrPos(i) + hdrLen(i) - 1 <= totalLen Then
                With .Characters(hdrPos(i), hdrLen(i)).Font
                    .Bold = msoTrue
                    .Italic = msoTrue
                    .Fill.ForeColor.RGB = clrHdr
                End With
            End If
            
            ' EP počet (tučné + barva)
            If epPos(i) >= 1 And epPos(i) + epLen(i) - 1 <= totalLen Then
                With .Characters(epPos(i), epLen(i)).Font
                    .Bold = msoTrue
                    If epIsZero(i) Then
                        .Fill.ForeColor.RGB = clrRed
                    Else
                        .Fill.ForeColor.RGB = clrGreen
                    End If
                End With
            End If
            
            ' Závorka (kurzíva)
            If brLen(i) > 0 And brPos(i) >= 1 And brPos(i) + brLen(i) - 1 <= totalLen Then
                With .Characters(brPos(i), brLen(i)).Font
                    .Italic = msoTrue
                End With
            End If
            
            ' Čas (tučné + podtržené)
            If tmLen(i) > 0 And tmPos(i) >= 1 And tmPos(i) + tmLen(i) - 1 <= totalLen Then
                With .Characters(tmPos(i), tmLen(i)).Font
                    .Bold = msoTrue
                    .UnderlineStyle = msoUnderlineSingleLine
                End With
            End If
        Next i
    End With

Cleanup:
    Application.ScreenUpdating = True
    Exit Sub
ErrorHandler:
    Application.ScreenUpdating = True
    MsgBox "Chyba AiringAnimeReminder: " & err.Description, vbCritical
End Sub

' ====================================================================
' VÝPOČET EPIZOD (čistá matematika, žádné COM volání)
' ====================================================================
Private Function CalcReleased(ByVal startDateVar As Variant, ByVal broadcastTime As String, ByVal nowDbl As Double) As Long
    CalcReleased = 0
    If Not IsDate(startDateVar) Then Exit Function
    
    Dim startDate As Date: startDate = CDate(startDateVar)
    Dim targetWd As Long:  targetWd = ParseWeekday(broadcastTime)
    
    If targetWd = 0 Then
        ' Fallback bez přesného času
        If startDate <= Date Then CalcReleased = (Date - startDate) \ 7 + 1
        Exit Function
    End If
    
    Dim h As Long, m As Long
    ParseTime broadcastTime, h, m
    
    ' Nejbližší targetWd k startDate (±3 dny)
    Dim diff As Long
    diff = targetWd - Weekday(startDate, vbSunday)
    If diff < -3 Then diff = diff + 7
    If diff > 3 Then diff = diff - 7
    
    Dim firstAirDT As Double
    firstAirDT = CDbl(startDate + diff) + CDbl(TimeSerial(h, m, 0))
    
    If nowDbl >= firstAirDT Then
        CalcReleased = Int((nowDbl - firstAirDT) / 7) + 1
    End If
End Function

' Batch extrakce MAL ID z hyperlinků — 1 průchod místo N×Cells().Hyperlinks
Private Function BuildHyperlinkMap(ByVal rng As Range) As Object
    Dim dict As Object
    Set dict = CreateObject("Scripting.Dictionary")
    On Error Resume Next
    
    Dim hl As Hyperlink
    For Each hl In rng.Hyperlinks
        Dim parts() As String
        parts = Split(hl.Address, "/")
        Dim k As Long
        For k = UBound(parts) To LBound(parts) Step -1 ' ID je obvykle na konci URL
            If IsNumeric(parts(k)) And Len(parts(k)) > 0 Then
                dict(hl.Range.row) = CLng(parts(k))
                Exit For
            End If
        Next k
    Next hl
    
    On Error GoTo 0
    Set BuildHyperlinkMap = dict
End Function

Private Function ParseWeekday(ByVal s As String) As Long
    Dim tL As String: tL = LCase(s)
    If InStr(tL, "ponděl") > 0 Then ParseWeekday = vbMonday: Exit Function
    If InStr(tL, "úter") > 0 Then ParseWeekday = vbTuesday: Exit Function
    If InStr(tL, "střed") > 0 Then ParseWeekday = vbWednesday: Exit Function
    If InStr(tL, "čtvrtek") > 0 Then ParseWeekday = vbThursday: Exit Function
    If InStr(tL, "pátek") > 0 Then ParseWeekday = vbFriday: Exit Function
    If InStr(tL, "sobot") > 0 Then ParseWeekday = vbSaturday: Exit Function
    If InStr(tL, "neděl") > 0 Then ParseWeekday = vbSunday: Exit Function
    ParseWeekday = 0
End Function

Private Sub ParseTime(ByVal s As String, ByRef h As Long, ByRef m As Long)
    h = 0: m = 0
    Dim tp As String: tp = Trim(Right(s, 5))
    On Error Resume Next
    h = CLng(Split(tp, ":")(0))
    m = CLng(Split(tp, ":")(1))
    On Error GoTo 0
End Sub

Private Function GetSortKey(ByVal timeStr As String) As Long
    If InStr(timeStr, "N/A") > 0 Or Len(timeStr) = 0 Then
        GetSortKey = 9999999
        Exit Function
    End If
    
    Dim dVal As Long
    Select Case ParseWeekday(timeStr)
        Case vbMonday:    dVal = 1
        Case vbTuesday:   dVal = 2
        Case vbWednesday: dVal = 3
        Case vbThursday:  dVal = 4
        Case vbFriday:    dVal = 5
        Case vbSaturday:  dVal = 6
        Case vbSunday:    dVal = 7
        Case Else:        dVal = 8
    End Select
    
    Dim h As Long, m As Long
    ParseTime timeStr, h, m
    GetSortKey = (dVal * 10000) + (h * 100) + m
End Function

Private Function LoadExceptions(ByVal exStr As String) As Object
    Dim dict As Object
    Set dict = CreateObject("Scripting.Dictionary")
    If Len(exStr) = 0 Then GoTo Done
    
    Dim pairs() As String: pairs = Split(exStr, "|")
    Dim j As Long, parts() As String
    For j = LBound(pairs) To UBound(pairs)
        If InStr(pairs(j), "=") > 0 Then
            parts = Split(pairs(j), "=")
            dict(Trim(parts(0))) = Trim(parts(1))
        End If
    Next j
Done:
    Set LoadExceptions = dict
End Function

Private Function GetAiringCache(ByVal wb As Workbook) As Object
    Dim dict As Object
    Set dict = CreateObject("Scripting.Dictionary")
    
    On Error Resume Next
    Dim wsInfo As Worksheet
    Set wsInfo = wb.Sheets("OBECNÉ INFORMACE")
    If wsInfo Is Nothing Then GoTo Done
    
    Dim shp As Shape
    Set shp = wsInfo.Shapes("txtBoxAiringAnimeContent")
    If shp Is Nothing Then GoTo Done
    On Error GoTo 0
    
    Dim metaString As String
    metaString = shp.AlternativeText
    If Len(metaString) = 0 Then GoTo Done
    
    If left(metaString, 4) = "DST=" Then
        Dim dsPos As Long: dsPos = InStr(metaString, "|")
        If dsPos > 0 Then metaString = Mid(metaString, dsPos + 1)
    End If
    
    Dim pairs() As String: pairs = Split(metaString, "|")
    Dim i As Long, parts() As String
    For i = LBound(pairs) To UBound(pairs)
        If InStr(pairs(i), "=") > 0 Then
            parts = Split(pairs(i), "=")
            dict(Trim(parts(0))) = Trim(parts(1))
        End If
    Next i
Done:
    Set GetAiringCache = dict
End Function

Private Function GetOrCreateTextBox(ByVal ws As Worksheet) As Shape
    On Error Resume Next
    Set GetOrCreateTextBox = ws.Shapes(SHAPE_NAME)
    On Error GoTo 0
    If GetOrCreateTextBox Is Nothing Then
        Set GetOrCreateTextBox = ws.Shapes.AddTextbox(msoTextOrientationHorizontal, 50, 50, 400, 150)
        GetOrCreateTextBox.Name = SHAPE_NAME
        GetOrCreateTextBox.line.Visible = msoFalse
        GetOrCreateTextBox.Fill.Visible = msoFalse
        GetOrCreateTextBox.TextFrame2.TextRange.Font.size = 11
        GetOrCreateTextBox.TextFrame2.TextRange.Font.Name = "Aptos Narrow"
    End If
End Function

Private Function HexToRGB(ByVal HexStr As String) As Long
    HexStr = Replace(HexStr, "#", "")
    If Len(HexStr) = 6 Then
        HexToRGB = RGB( _
            val("&H" & left(HexStr, 2)), _
            val("&H" & Mid(HexStr, 3, 2)), _
            val("&H" & Right(HexStr, 2)))
    End If
End Function
